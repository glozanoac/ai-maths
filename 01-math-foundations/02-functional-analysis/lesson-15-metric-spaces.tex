\section{Motivation}
\label{sec:motivation}

Reinforcement learning algorithms seek fixed points of operators acting on
spaces of functions. The Bellman optimality operator
\begin{equation}
\label{eq:bellman}
(TV)(s) = \max_{a \in \mathcal{A}} \Bigl[ r(s,a) + \gamma
  \sum_{s'} p(s' \mid s,a)\, V(s') \Bigr]
\end{equation}
maps a value function $V \colon \mathcal{S} \to \R$ to another value function
$TV$. To prove that value iteration $V_{n+1} = TV_n$ converges, we need
precise answers to four questions:
\begin{enumerate}[nosep]
  \item What does ``distance between two value functions'' mean?
  \item What does ``$V_n$ converges to $V^*$'' mean?
  \item Does the limit $V^*$ actually belong to the same space as the $V_n$?
  \item Is the limit unique?
\end{enumerate}

\noindent
The theory of \emph{metric spaces} provides the framework for all four. A
metric formalizes distance (question 1), convergence of sequences is defined
via the metric (question 2), completeness guarantees that Cauchy sequences
have limits in the space (question 3), and the Hausdorff property of metric
spaces ensures uniqueness (question 4). The Banach contraction mapping
theorem -- which we prove in a later lesson -- ties everything together, but
its hypotheses are stated entirely in the language of this lesson.

\begin{rlconnection}[title={The Bellman Operator on a Metric Space}]
The Bellman operator $T$ acts on the space of bounded real-valued functions
$b(\mathcal{S})$ equipped with the supremum metric
$d_\infty(V, W) = \sup_{s \in \mathcal{S}} |V(s) - W(s)|$. One shows that
$d_\infty(TV, TW) \le \gamma\, d_\infty(V, W)$ for any discount factor
$\gamma \in [0,1)$, making $T$ a \emph{contraction}. The contraction mapping
theorem then yields existence, uniqueness, and geometric convergence of value
iteration -- but only if $(b(\mathcal{S}), d_\infty)$ is a \emph{complete}
metric space. Every concept in this lesson (metric, convergence, Cauchy
sequence, completeness) is indispensable to that argument.
\end{rlconnection}

\section{Metric Spaces}
\label{sec:metric}

\subsection{Definition and First Properties}
\label{subsec:metric_def}

\begin{definition}[Metric Space]
\label{def:metric}
A \emph{metric} on a set $X$ is a function $d \colon X \times X \to [0,\infty)$
satisfying, for all $x, y, z \in X$:
\begin{enumerate}[nosep,label=(M\arabic*)]
  \item \label{M1} $d(x,y) = 0$ if and only if $x = y$.
    \hfill (identity of indiscernibles)
  \item \label{M2} $d(x,y) = d(y,x)$.
    \hfill (symmetry)
  \item \label{M3} $d(x,z) \le d(x,y) + d(y,z)$.
    \hfill (triangle inequality)
\end{enumerate}
The pair $(X,d)$ is called a \emph{metric space}.
\end{definition}

\begin{remark}
\label{rem:nonneg}
Non-negativity $d(x,y) \ge 0$ follows from the axioms and need not be
postulated separately. Indeed, by \ref{M3} and \ref{M2},
$0 = d(x,x) \le d(x,y) + d(y,x) = 2\,d(x,y)$, so $d(x,y) \ge 0$.
\end{remark}

\subsection{Examples}
\label{subsec:examples}

\begin{example}[Euclidean Metric on $\R^n$]
\label{ex:euclidean}
On $X = \R^n$, define
$d_2(x,y) = \bigl(\sum_{i=1}^n (x_i - y_i)^2\bigr)^{1/2}$.
Axioms \ref{M1} and \ref{M2} are immediate. The triangle inequality follows
from the Cauchy--Schwarz inequality: for $u, v \in \R^n$,
$\bigl(\sum_i u_i v_i\bigr)^2 \le \bigl(\sum_i u_i^2\bigr)
\bigl(\sum_i v_i^2\bigr)$, which implies
$\|u + v\|_2 \le \|u\|_2 + \|v\|_2$ by expanding
$\|u+v\|_2^2 = \|u\|_2^2 + 2\langle u,v\rangle + \|v\|_2^2
\le (\|u\|_2 + \|v\|_2)^2$.
\end{example}

\begin{example}[$p$-Metrics on $\R^n$]
\label{ex:p_metric}
For $1 \le p \le \infty$, define $d_p(x,y) = \|x - y\|_p$ where
\[
\|x\|_p =
\begin{cases}
\bigl(\sum_{i=1}^n |x_i|^p\bigr)^{1/p} & \text{if } 1 \le p < \infty,\\[4pt]
\max_{1 \le i \le n} |x_i| & \text{if } p = \infty.
\end{cases}
\]
Each $d_p$ is a metric. For $1 < p < \infty$, the triangle inequality is
Minkowski's inequality (proved via H\"older's inequality). The case
$p = \infty$ is straightforward:
$\max_i |x_i + y_i| \le \max_i(|x_i| + |y_i|) \le \max_i|x_i| + \max_i|y_i|$.
\end{example}

\begin{example}[Discrete Metric]
\label{ex:discrete}
On any set $X$, define $d(x,y) = 0$ if $x = y$ and $d(x,y) = 1$ if $x \ne y$.
All three axioms are easily verified; in particular, the triangle inequality
holds because $d(x,z) \le 1 \le d(x,y) + d(y,z)$ whenever $x \ne z$ (at least
one of $d(x,y), d(y,z)$ equals $1$). Every subset of $X$ is open: for any
$x \in A \subseteq X$, $B(x, 1/2) = \{x\} \subseteq A$.
\end{example}

\begin{example}[{The Sup Metric on $C[a,b]$}]
\label{ex:Cab}
Let $C[a,b]$ denote the set of continuous real-valued functions on $[a,b]$.
Define
\[
d_\infty(f,g) = \max_{t \in [a,b]} |f(t) - g(t)|.
\]
The maximum exists by the extreme value theorem (continuous functions on
compact sets attain their extrema). Axioms \ref{M1}--\ref{M3} follow from
the corresponding properties of $|\cdot|$ on $\R$.

One can also define $d_1(f,g) = \int_a^b |f(t) - g(t)|\, dt$, which is a
different metric on $C[a,b]$. These two metrics induce different topologies
and have different completeness properties (see Section~\ref{sec:completeness}).
\end{example}

\begin{example}[Sequence Spaces $\ell^p$]
\label{ex:ell_p}
For $1 \le p < \infty$, define
\[
\ell^p = \Bigl\{x = (x_n)_{n=1}^\infty \subset \R :
  \sum_{n=1}^\infty |x_n|^p < \infty\Bigr\}, \quad
d_p(x,y) = \Bigl(\sum_{n=1}^\infty |x_n - y_n|^p\Bigr)^{1/p}.
\]
For $p = \infty$,
$\ell^\infty = \{x = (x_n) : \sup_n |x_n| < \infty\}$ with
$d_\infty(x,y) = \sup_n |x_n - y_n|$.
The triangle inequality for $\ell^p$ ($1 < p < \infty$) is the infinite-sum
version of Minkowski's inequality.
\end{example}

\subsection{Basic Topology}
\label{subsec:topology}

\begin{definition}[Open Ball]
\label{def:open_ball}
In a metric space $(X,d)$, the \emph{open ball} of radius $r > 0$ centered at
$x \in X$ is
\[
B(x,r) = \{y \in X : d(x,y) < r\}.
\]
The \emph{closed ball} is $\overline{B}(x,r) = \{y \in X : d(x,y) \le r\}$.
\end{definition}

\begin{definition}[Open and Closed Sets]
\label{def:open_closed}
Let $(X,d)$ be a metric space.
\begin{enumerate}[nosep,label=(\roman*)]
  \item A set $G \subseteq X$ is \emph{open} if for every $x \in G$ there
    exists $r > 0$ such that $B(x,r) \subseteq G$.
  \item A set $F \subseteq X$ is \emph{closed} if $X \setminus F$ is open.
\end{enumerate}
\end{definition}

\begin{proposition}[Properties of Open and Closed Sets]
\label{prop:open_closed}
In a metric space $(X,d)$:
\begin{enumerate}[nosep,label=(\alph*)]
  \item $\emptyset$ and $X$ are both open and closed.
  \item Any union of open sets is open.
  \item Any finite intersection of open sets is open.
  \item Any intersection of closed sets is closed.
  \item Any finite union of closed sets is closed.
  \item Every open ball $B(x,r)$ is an open set.
\end{enumerate}
\end{proposition}

\begin{proof}
\textbf{(a)} $\emptyset$ is open vacuously (no point to check). $X$ is open
because $B(x,r) \subseteq X$ for every $x \in X$ and $r > 0$. Their
complements are $X$ and $\emptyset$, so both are also closed.

\textbf{(b)} Let $\{G_\alpha\}_{\alpha \in I}$ be open sets and
$x \in \bigcup_\alpha G_\alpha$. Then $x \in G_\beta$ for some $\beta \in I$,
so there exists $r > 0$ with $B(x,r) \subseteq G_\beta \subseteq
\bigcup_\alpha G_\alpha$.

\textbf{(c)} Let $G_1, \ldots, G_n$ be open and $x \in \bigcap_{i=1}^n G_i$.
For each $i$, choose $r_i > 0$ with $B(x,r_i) \subseteq G_i$. Set
$r = \min(r_1, \ldots, r_n) > 0$ (finite minimum of positive numbers). Then
$B(x,r) \subseteq B(x,r_i) \subseteq G_i$ for all $i$, so
$B(x,r) \subseteq \bigcap_{i=1}^n G_i$.

\textbf{(d)--(e)} These follow from (b)--(c) by De Morgan's laws:
$X \setminus \bigcap_\alpha F_\alpha = \bigcup_\alpha (X \setminus F_\alpha)$
and
$X \setminus \bigcup_{i=1}^n F_i = \bigcap_{i=1}^n (X \setminus F_i)$.

\textbf{(f)} Let $y \in B(x,r)$ and set $\varepsilon = r - d(x,y) > 0$. For
any $z \in B(y,\varepsilon)$, the triangle inequality gives
$d(x,z) \le d(x,y) + d(y,z) < d(x,y) + \varepsilon = r$, so
$z \in B(x,r)$. Hence $B(y,\varepsilon) \subseteq B(x,r)$.
\end{proof}

\begin{definition}[Interior, Closure, Boundary]
\label{def:int_cl_bd}
Let $A \subseteq X$ in a metric space $(X,d)$.
\begin{enumerate}[nosep,label=(\roman*)]
  \item The \emph{interior} of $A$ is
    $\interior(A) = \bigcup\{G \subseteq A : G \text{ is open}\}$,
    the largest open set contained in $A$.
  \item The \emph{closure} of $A$ is
    $\cl(A) = \bigcap\{F \supseteq A : F \text{ is closed}\}$,
    the smallest closed set containing $A$.
  \item The \emph{boundary} of $A$ is
    $\partial A = \cl(A) \setminus \interior(A)$.
\end{enumerate}
\end{definition}

\begin{proposition}[Closure Characterization]
\label{prop:closure_char}
Let $(X,d)$ be a metric space and $A \subseteq X$. The following are
equivalent:
\begin{enumerate}[nosep,label=(\roman*)]
  \item $x \in \cl(A)$.
  \item For every $r > 0$, $B(x,r) \cap A \ne \emptyset$.
  \item There exists a sequence $(a_n)$ in $A$ with $a_n \to x$.
\end{enumerate}
\end{proposition}

\begin{proof}
\textbf{(i) $\Rightarrow$ (ii).}
Suppose $x \in \cl(A)$ but there exists $r > 0$ with
$B(x,r) \cap A = \emptyset$. Then $A \subseteq X \setminus B(x,r)$. Since
$B(x,r)$ is open, $X \setminus B(x,r)$ is closed. By definition of closure,
$\cl(A) \subseteq X \setminus B(x,r)$. But $x \in B(x,r)$, so
$x \notin X \setminus B(x,r)$, contradicting $x \in \cl(A)$.

\textbf{(ii) $\Rightarrow$ (iii).}
For each $n \in \N$, choose $a_n \in B(x, 1/n) \cap A$. Then
$d(a_n, x) < 1/n \to 0$, so $a_n \to x$.

\textbf{(iii) $\Rightarrow$ (i).}
Let $(a_n)$ be a sequence in $A$ with $a_n \to x$. Let $F$ be any closed set
with $A \subseteq F$. Then each $a_n \in F$. Since $F$ is closed,
$X \setminus F$ is open. If $x \notin F$, then $x \in X \setminus F$, so
there exists $r > 0$ with $B(x,r) \subseteq X \setminus F$. But $a_n \to x$
means $a_n \in B(x,r)$ for large $n$, giving $a_n \notin F$, contradicting
$a_n \in A \subseteq F$. Hence $x \in F$. Since this holds for every closed
$F \supseteq A$, we get $x \in \cl(A)$.
\end{proof}

\section{Convergence and Completeness}
\label{sec:completeness}

\subsection{Convergent Sequences}
\label{subsec:convergence}

\begin{definition}[Convergence]
\label{def:convergence}
A sequence $(x_n)$ in a metric space $(X,d)$ \emph{converges} to $x \in X$ if
for every $\varepsilon > 0$ there exists $N \in \N$ such that
$d(x_n, x) < \varepsilon$ for all $n \ge N$. We write $x_n \to x$ or
$\lim_{n \to \infty} x_n = x$.
\end{definition}

\begin{proposition}[Uniqueness of Limits]
\label{prop:unique_limit}
In a metric space, limits are unique: if $x_n \to x$ and $x_n \to y$, then
$x = y$.
\end{proposition}

\begin{proof}
By the triangle inequality,
$0 \le d(x,y) \le d(x, x_n) + d(x_n, y)$ for every $n$. Given
$\varepsilon > 0$, choose $N_1$ with $d(x_n, x) < \varepsilon/2$ for
$n \ge N_1$ and $N_2$ with $d(x_n, y) < \varepsilon/2$ for $n \ge N_2$. For
$n \ge \max(N_1, N_2)$,
$d(x,y) < \varepsilon/2 + \varepsilon/2 = \varepsilon$. Since $\varepsilon > 0$
was arbitrary and $d(x,y) \ge 0$, we conclude $d(x,y) = 0$, hence $x = y$ by
\ref{M1}.
\end{proof}

\subsection{Cauchy Sequences and Completeness}
\label{subsec:cauchy}

\begin{definition}[Cauchy Sequence]
\label{def:cauchy}
A sequence $(x_n)$ in $(X,d)$ is a \emph{Cauchy sequence} if for every
$\varepsilon > 0$ there exists $N \in \N$ such that $d(x_m, x_n) < \varepsilon$
for all $m, n \ge N$.
\end{definition}

\begin{proposition}[Convergent Implies Cauchy]
\label{prop:conv_cauchy}
Every convergent sequence in a metric space is Cauchy.
\end{proposition}

\begin{proof}
Suppose $x_n \to x$. Given $\varepsilon > 0$, choose $N$ with
$d(x_n, x) < \varepsilon/2$ for all $n \ge N$. For $m, n \ge N$,
\[
d(x_m, x_n) \le d(x_m, x) + d(x, x_n) < \frac{\varepsilon}{2} +
\frac{\varepsilon}{2} = \varepsilon. \qedhere
\]
\end{proof}

\begin{warning}[title={Cauchy Does Not Imply Convergent}]
The converse fails in incomplete spaces. Consider $\Q$ with the usual metric
$d(x,y) = |x - y|$. Define $x_1 = 1$ and $x_{n+1} = (x_n + 2/x_n)/2$.
Each $x_n \in \Q$, and one shows by induction that $(x_n)$ is Cauchy. The
sequence converges to $\sqrt{2}$ in $\R$, but $\sqrt{2} \notin \Q$. So
$(x_n)$ is Cauchy in $\Q$ with no limit in $\Q$.
\end{warning}

\begin{definition}[Complete Metric Space]
\label{def:complete}
A metric space $(X,d)$ is \emph{complete} if every Cauchy sequence in $X$
converges to a limit in $X$.
\end{definition}

\begin{theorem}[Completeness of $\R$]
\label{thm:R_complete}
$(\R, |\cdot|)$ is a complete metric space.
\end{theorem}

\begin{proof}
Let $(x_n)$ be a Cauchy sequence in $\R$.

\textbf{Step 1: $(x_n)$ is bounded.} Choose $N$ with $|x_m - x_n| < 1$ for
all $m, n \ge N$. Then for $n \ge N$, $|x_n| \le |x_N| + 1$. Setting
$M = \max(|x_1|, \ldots, |x_{N-1}|, |x_N| + 1)$, we have $|x_n| \le M$ for
all $n$.

\textbf{Step 2: Extract a convergent subsequence.} By the Bolzano--Weierstrass
theorem, every bounded sequence in $\R$ has a convergent subsequence. Thus
there exist a subsequence $(x_{n_k})$ and a point $x \in \R$ with
$x_{n_k} \to x$.

\textbf{Step 3: The full sequence converges.} Given $\varepsilon > 0$, choose
$N_1$ with $|x_m - x_n| < \varepsilon/2$ for $m, n \ge N_1$, and $K$ with
$|x_{n_k} - x| < \varepsilon/2$ for $k \ge K$. For $n \ge N_1$, pick
$k \ge K$ with $n_k \ge N_1$. Then
\[
|x_n - x| \le |x_n - x_{n_k}| + |x_{n_k} - x| < \frac{\varepsilon}{2}
  + \frac{\varepsilon}{2} = \varepsilon.
\]
Hence $x_n \to x$.
\end{proof}

\begin{theorem}[Completeness of $\R^n$]
\label{thm:Rn_complete}
$(\R^n, d_p)$ is a complete metric space for every $1 \le p \le \infty$.
\end{theorem}

\begin{proof}
Let $(x^{(k)})_{k=1}^\infty$ be Cauchy in $(\R^n, d_p)$, with
$x^{(k)} = (x_1^{(k)}, \ldots, x_n^{(k)})$.

\textbf{Coordinate-wise convergence.} For each $i \in \{1, \ldots, n\}$,
\[
|x_i^{(k)} - x_i^{(m)}| \le \|x^{(k)} - x^{(m)}\|_p,
\]
since the $\ell^p$ norm dominates each coordinate. Hence
$(x_i^{(k)})_{k=1}^\infty$ is Cauchy in $\R$. By
Theorem~\ref{thm:R_complete}, $x_i^{(k)} \to x_i$ for some $x_i \in \R$.

\textbf{Convergence in $d_p$.} Let $x = (x_1, \ldots, x_n) \in \R^n$. Given
$\varepsilon > 0$, choose $K$ with $\|x^{(k)} - x^{(m)}\|_p < \varepsilon/2$
for $k, m \ge K$. For any finite set of coordinates and $k \ge K$, pass
$m \to \infty$ in $|x_i^{(k)} - x_i^{(m)}|$ to obtain
$|x_i^{(k)} - x_i| \le \varepsilon/2$. Thus
$\|x^{(k)} - x\|_p \le \varepsilon/2 < \varepsilon$ for $k \ge K$, and
$x^{(k)} \to x$ in $(\R^n, d_p)$.
\end{proof}

\begin{keyresult}[title={Completeness of $\R^n$}]
The spaces $(\R^n, d_p)$ for $1 \le p \le \infty$ are all complete. In RL with
a finite state space $|\mathcal{S}| = n$, value functions are vectors in
$\R^n$ and the space of all value functions is complete under any $p$-metric.
This is one reason finite-state MDPs are analytically tractable.
\end{keyresult}

\subsection{Dense Subsets and Separability}
\label{subsec:separable}

\begin{definition}[Dense Subset, Separable Space]
\label{def:separable}
Let $(X,d)$ be a metric space.
\begin{enumerate}[nosep,label=(\roman*)]
  \item A subset $D \subseteq X$ is \emph{dense} in $X$ if $\cl(D) = X$,
    equivalently, if for every $x \in X$ and every $r > 0$,
    $B(x,r) \cap D \ne \emptyset$.
  \item $(X,d)$ is \emph{separable} if it contains a countable dense subset.
\end{enumerate}
\end{definition}

\begin{example}
\label{ex:separable}
\begin{enumerate}[nosep,label=(\alph*)]
  \item $\R^n$ is separable: $\Q^n$ is countable and dense (every open ball
    in $\R^n$ contains a point with all rational coordinates).
  \item $(C[a,b], d_\infty)$ is separable. By the Weierstrass approximation
    theorem, every continuous function on $[a,b]$ can be uniformly
    approximated by polynomials. Polynomials with rational coefficients form a
    countable dense subset.
  \item $\ell^p$ is separable for $1 \le p < \infty$: the set of sequences
    with finitely many nonzero rational entries is countable and dense.
  \item $\ell^\infty$ is \emph{not} separable. The uncountable collection
    $\{x_A : A \subseteq \N\}$ where $(x_A)_n = \indicator_A(n)$ satisfies
    $d_\infty(x_A, x_B) = 1$ for $A \ne B$. Any dense subset must contain a
    point in each ball $B(x_A, 1/3)$, and these balls are pairwise disjoint,
    so the dense subset is uncountable.
\end{enumerate}
\end{example}

\section{The Completion Theorem}
\label{sec:completion}

The example of $\Q$ inside $\R$ suggests a general construction: given an
incomplete metric space, we can ``fill in the gaps'' by adding limits of Cauchy
sequences. The following theorem makes this precise.

\begin{theorem}[Completion of a Metric Space]
\label{thm:completion}
Let $(X,d)$ be a metric space. There exists a complete metric space
$(\hat{X}, \hat{d})$ and an isometry $\iota \colon X \to \hat{X}$ (meaning
$\hat{d}(\iota(x), \iota(y)) = d(x,y)$ for all $x, y \in X$) such that
$\iota(X)$ is dense in $\hat{X}$. Moreover, the completion is unique up to
isometry: if $(\hat{X}_1, \hat{d}_1, \iota_1)$ and
$(\hat{X}_2, \hat{d}_2, \iota_2)$ are two completions of $(X,d)$, then there
exists a surjective isometry $\Phi \colon \hat{X}_1 \to \hat{X}_2$ with
$\Phi \circ \iota_1 = \iota_2$.
\end{theorem}

\begin{proof}
We break the proof into five steps.

\textbf{Step 1: Construction of $\hat{X}$.}
Let $\mathcal{C}$ denote the set of all Cauchy sequences in $(X,d)$. Define
an equivalence relation on $\mathcal{C}$ by
\[
(x_n) \sim (y_n) \quad \iff \quad \lim_{n \to \infty} d(x_n, y_n) = 0.
\]
We verify that $\sim$ is an equivalence relation. Reflexivity: $d(x_n, x_n) = 0
\to 0$. Symmetry: $d(x_n, y_n) = d(y_n, x_n)$ by \ref{M2}. Transitivity: if
$(x_n) \sim (y_n)$ and $(y_n) \sim (z_n)$, then
$d(x_n, z_n) \le d(x_n, y_n) + d(y_n, z_n) \to 0$ by \ref{M3}.

Let $\hat{X} = \mathcal{C}/{\sim}$ and write $[(x_n)]$ for the equivalence
class of $(x_n)$.

\textbf{Step 2: The metric $\hat{d}$ is well defined.}
For Cauchy sequences $(x_n)$ and $(y_n)$, the real sequence $(d(x_n, y_n))$ is
Cauchy in $\R$: by the triangle inequality,
\[
|d(x_m, y_m) - d(x_n, y_n)| \le d(x_m, x_n) + d(y_m, y_n) \to 0
\quad \text{as } m, n \to \infty.
\]
Since $\R$ is complete (Theorem~\ref{thm:R_complete}), the limit
$\lim_{n \to \infty} d(x_n, y_n)$ exists. Define
\[
\hat{d}\bigl([(x_n)],\, [(y_n)]\bigr) = \lim_{n \to \infty} d(x_n, y_n).
\]

\emph{Independence of representatives.} If $(x_n) \sim (x_n')$ and
$(y_n) \sim (y_n')$, then
\[
|d(x_n, y_n) - d(x_n', y_n')| \le d(x_n, x_n') + d(y_n, y_n') \to 0,
\]
so the limits agree.

\emph{$\hat{d}$ is a metric.} \ref{M2} and non-negativity are inherited from
$d$. For \ref{M1}: $\hat{d}([(x_n)], [(y_n)]) = 0$ iff
$d(x_n, y_n) \to 0$ iff $(x_n) \sim (y_n)$ iff $[(x_n)] = [(y_n)]$.
For \ref{M3}:
$\hat{d}([(x_n)], [(z_n)]) = \lim d(x_n, z_n) \le \lim [d(x_n, y_n)
+ d(y_n, z_n)] = \hat{d}([(x_n)], [(y_n)]) + \hat{d}([(y_n)], [(z_n)])$.

\textbf{Step 3: $\iota$ is an isometric embedding and $\iota(X)$ is dense.}
Define $\iota \colon X \to \hat{X}$ by $\iota(x) = [(x, x, x, \ldots)]$
(the class of the constant sequence at $x$). Then
$\hat{d}(\iota(x), \iota(y)) = \lim_{n \to \infty} d(x, y) = d(x, y)$,
so $\iota$ is an isometry.

For density: let $p = [(x_n)] \in \hat{X}$ and $\varepsilon > 0$. Since
$(x_n)$ is Cauchy, there exists $N$ with $d(x_m, x_n) < \varepsilon/2$ for
$m, n \ge N$. Then
\[
\hat{d}(p,\, \iota(x_N)) = \lim_{n \to \infty} d(x_n, x_N) \le
\frac{\varepsilon}{2} < \varepsilon.
\]
Hence $\iota(X)$ is dense in $\hat{X}$.

\textbf{Step 4: $\hat{X}$ is complete.}
Let $(p_k)_{k=1}^\infty$ be a Cauchy sequence in $(\hat{X}, \hat{d})$. Since
$\iota(X)$ is dense, for each $k$ choose $z_k \in X$ with
$\hat{d}(p_k, \iota(z_k)) < 1/k$.

\emph{$(z_k)$ is Cauchy in $X$.} For $k, m \in \N$,
\begin{align*}
d(z_k, z_m) &= \hat{d}(\iota(z_k), \iota(z_m)) \\
  &\le \hat{d}(\iota(z_k), p_k) + \hat{d}(p_k, p_m) +
       \hat{d}(p_m, \iota(z_m)) \\
  &< \frac{1}{k} + \hat{d}(p_k, p_m) + \frac{1}{m}.
\end{align*}
Since $(p_k)$ is Cauchy in $\hat{X}$, the right-hand side tends to $0$ as
$k, m \to \infty$. Hence $(z_k)$ is Cauchy in $X$.

\emph{$(p_k)$ converges to $p = [(z_k)] \in \hat{X}$.} We have
\[
\hat{d}(p_k, p) \le \hat{d}(p_k, \iota(z_k)) + \hat{d}(\iota(z_k), p).
\]
The first term is at most $1/k \to 0$. For the second term,
\[
\hat{d}(\iota(z_k), p) = \hat{d}(\iota(z_k),\, [(z_j)]) =
\lim_{j \to \infty} d(z_k, z_j).
\]
Since $(z_j)$ is Cauchy, $\lim_{j \to \infty} d(z_k, z_j) \to 0$ as
$k \to \infty$ (given $\varepsilon > 0$, choose $K$ with
$d(z_j, z_m) < \varepsilon$ for $j, m \ge K$; then for $k \ge K$,
$\lim_{j} d(z_k, z_j) \le \varepsilon$). Therefore
$\hat{d}(p_k, p) \to 0$, and $(\hat{X}, \hat{d})$ is complete.

\textbf{Step 5: Uniqueness up to isometry.}
Let $(\hat{X}_1, \hat{d}_1, \iota_1)$ and $(\hat{X}_2, \hat{d}_2, \iota_2)$
be two completions of $(X,d)$. Define
$\varphi \colon \iota_1(X) \to \hat{X}_2$ by
$\varphi(\iota_1(x)) = \iota_2(x)$. Then
\[
\hat{d}_2(\varphi(\iota_1(x)),\, \varphi(\iota_1(y))) =
\hat{d}_2(\iota_2(x), \iota_2(y)) = d(x,y) =
\hat{d}_1(\iota_1(x), \iota_1(y)),
\]
so $\varphi$ is an isometry from $\iota_1(X)$ to $\iota_2(X)$. In particular,
$\varphi$ is uniformly continuous. Since $\iota_1(X)$ is dense in $\hat{X}_1$
and $\hat{X}_2$ is complete, $\varphi$ extends uniquely to a uniformly
continuous map $\Phi \colon \hat{X}_1 \to \hat{X}_2$ (by the extension theorem
for uniformly continuous maps into complete spaces). By continuity, $\Phi$
preserves the isometry property:
$\hat{d}_2(\Phi(p), \Phi(q)) = \hat{d}_1(p, q)$ for all
$p, q \in \hat{X}_1$.

Finally, $\Phi$ is surjective: $\Phi(\hat{X}_1) \supseteq
\Phi(\iota_1(X)) = \iota_2(X)$, and $\Phi(\hat{X}_1)$ is complete (as an
isometric image of a complete space), hence closed. Since $\iota_2(X)$ is
dense in $\hat{X}_2$, we get
$\hat{X}_2 = \cl(\iota_2(X)) \subseteq \cl(\Phi(\hat{X}_1)) =
\Phi(\hat{X}_1)$. Thus $\Phi$ is a surjective isometry with
$\Phi \circ \iota_1 = \iota_2$.
\end{proof}

\begin{rlconnection}[title={Completion and Function Approximation}]
In RL with function approximation, we often work with a parametric class of
value functions (e.g., linear combinations of features, or neural network
outputs) that may not be complete under the sup norm. The completion theorem
guarantees that this class can be isometrically embedded into a complete space.
Understanding the structure of the completed space helps determine whether
approximate value iteration converges and what it converges to.
\end{rlconnection}

\section{Exercises}
\label{sec:exercises}

\begin{exercise}[Metrics from Metrics]
\label{ex:derived_metrics}
Let $(X,d)$ be a metric space. Prove that the following are also metrics on $X$:
\begin{enumerate}[nosep,label=(\alph*)]
  \item $d_1(x,y) = \min(d(x,y),\, 1)$.
  \item $d_2(x,y) = \dfrac{d(x,y)}{1 + d(x,y)}$.
\end{enumerate}
For each, determine whether $(X,d_i)$ is complete whenever $(X,d)$ is
complete.

\emph{Hint:} For (b), show that $f(t) = t/(1+t)$ is increasing and
subadditive on $[0,\infty)$.
\end{exercise}

\begin{exercise}[Closed Balls Need Not Be Closures of Open Balls]
\label{ex:closed_ball}
In a general metric space, the closed ball
$\overline{B}(x,r) = \{y : d(x,y) \le r\}$ may differ from the closure
$\cl(B(x,r))$ of the open ball. Give an example by considering the discrete
metric on a set with at least two points, and verify explicitly that
$\cl(B(x,1)) \ne \overline{B}(x,1)$ for a suitable choice of $r$.
\end{exercise}

\begin{exercise}[Completeness of $\ell^\infty$]
\label{ex:ell_inf_complete}
Prove that $(\ell^\infty, d_\infty)$ is a complete metric space. That is,
show that every Cauchy sequence of bounded sequences converges to a bounded
sequence.

\emph{Hint:} Follow the structure of the proof of
Theorem~\ref{thm:Rn_complete}: use coordinate-wise convergence, then verify
that the limit is bounded and that convergence holds in $d_\infty$.
\end{exercise}

\begin{exercise}[Density of $\Q$ in $\R$]
\label{ex:Q_dense}
Prove that $\Q$ is dense in $(\R, |\cdot|)$ using only the Archimedean
property of $\R$ (for any $x > 0$, there exists $n \in \N$ with $n > x$).
That is, show that for every $x \in \R$ and every $\varepsilon > 0$, there
exists $q \in \Q$ with $|x - q| < \varepsilon$.
\end{exercise}

\begin{exercise}[The Completion of $\Q$]
\label{ex:Q_completion}
Let $(\hat{\Q}, \hat{d})$ be the completion of $(\Q, |\cdot|)$ as constructed
in Theorem~\ref{thm:completion}. Show that $\hat{\Q}$ is isometric to
$(\R, |\cdot|)$.

\emph{Hint:} Use uniqueness (Step 5 of the proof). It suffices to show that
$\R$ is complete and that $\Q$ embeds isometrically as a dense subset of $\R$.
\end{exercise}
