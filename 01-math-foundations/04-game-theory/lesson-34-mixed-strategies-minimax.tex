%% SECTION 1: MIXED STRATEGIES AND MIXED EXTENSIONS

\section{Mixed Strategies and Mixed Extensions}
\label{sec:mixed}

\subsection{Definitions}
\label{subsec:mixed_def}

In Lesson~33 we saw that some games (e.g., Matching Pennies) have no pure
strategy Nash equilibrium. To guarantee existence, we enlarge each player's
strategy space from finitely many pure strategies to the simplex of
probability distributions over them.

\begin{definition}[Mixed Strategy]
\label{def:mixed_strategy}
Let $\Gamma = (N, (S_i)_{i \in N}, (u_i)_{i \in N})$ be a finite game
with $S_i = \{s_i^1, \ldots, s_i^{m_i}\}$ for player~$i$. A \emph{mixed
strategy} for player~$i$ is a probability distribution
$\sigma_i \in \Delta(S_i)$, where
\[
    \Delta(S_i) = \bigg\{\sigma_i \in \R^{m_i} :
    \sigma_i(s_i^j) \geq 0 \text{ for all } j, \;\;
    \sum_{j=1}^{m_i} \sigma_i(s_i^j) = 1\bigg\}
\]
is the $(m_i - 1)$-simplex over $S_i$. A \emph{mixed strategy profile} is
$\sigma = (\sigma_1, \ldots, \sigma_n) \in \Delta := \prod_{i \in N} \Delta(S_i)$.
\end{definition}

\begin{definition}[Support]
\label{def:support}
The \emph{support} of a mixed strategy $\sigma_i$ is
$\supp(\sigma_i) = \{s_i \in S_i : \sigma_i(s_i) > 0\}$.
A pure strategy $s_i$ is identified with the degenerate distribution
placing probability~1 on $s_i$.
\end{definition}

\begin{definition}[Mixed Extension]
\label{def:mixed_extension}
The \emph{mixed extension} of a finite game $\Gamma = (N, (S_i), (u_i))$ is
the game $\bar{\Gamma} = (N, (\Delta(S_i)), (\bar{u}_i))$ where the expected
payoff under independent mixing is
\[
    \bar{u}_i(\sigma) = \sum_{s \in S} \bigg(\prod_{j \in N} \sigma_j(s_j)\bigg)
    u_i(s) = \E_{s \sim \sigma}[u_i(s)].
\]
We drop the bar and write $u_i(\sigma)$ when the context is clear.
\end{definition}

\begin{remark}[Multilinearity]
\label{rem:multilinear}
The expected payoff $u_i(\sigma)$ is \emph{multilinear}: linear in each
$\sigma_i$ when the other players' strategies are held fixed. Explicitly,
\[
    u_i(\sigma_i, \sigma_{-i}) = \sum_{s_i \in S_i} \sigma_i(s_i)
    \underbrace{\sum_{s_{-i} \in S_{-i}}
    \bigg(\prod_{j \neq i} \sigma_j(s_j)\bigg) u_i(s_i, s_{-i})}_{
    =:\, u_i(s_i, \sigma_{-i})}.
\]
The quantity $u_i(s_i, \sigma_{-i})$ is the expected payoff to player~$i$
from the pure strategy $s_i$ when opponents play $\sigma_{-i}$.
\end{remark}

\subsection{Mixed Nash Equilibrium and the Indifference Principle}
\label{subsec:mixed_ne}

\begin{definition}[Mixed Strategy Nash Equilibrium]
\label{def:mixed_ne}
A mixed strategy profile $\sigma^*$ is a \emph{Nash equilibrium} if
\[
    u_i(\sigma_i^*, \sigma_{-i}^*) \geq u_i(\sigma_i, \sigma_{-i}^*)
    \quad \text{for all } \sigma_i \in \Delta(S_i), \text{ for all } i \in N.
\]
\end{definition}

\begin{keyresult}
\begin{theorem}[Indifference Principle]
\label{thm:indifference}
Let $\sigma^*$ be a mixed strategy Nash equilibrium. For each player~$i$:
\begin{enumerate}[nosep,label=(\alph*)]
    \item If $s_i, s_i' \in \supp(\sigma_i^*)$, then
    $u_i(s_i, \sigma_{-i}^*) = u_i(s_i', \sigma_{-i}^*)$.
    \item If $s_i \in \supp(\sigma_i^*)$ and $t_i \notin \supp(\sigma_i^*)$,
    then $u_i(s_i, \sigma_{-i}^*) \geq u_i(t_i, \sigma_{-i}^*)$.
\end{enumerate}
That is, every pure strategy in the support yields the same expected payoff,
and this common payoff is at least as large as the payoff from any
strategy outside the support.
\end{theorem}
\end{keyresult}

\begin{proof}
By multilinearity (Remark~\ref{rem:multilinear}),
\[
    u_i(\sigma_i^*, \sigma_{-i}^*) = \sum_{s_i \in S_i} \sigma_i^*(s_i)\,
    u_i(s_i, \sigma_{-i}^*).
\]
Let $v_i^* = u_i(\sigma_i^*, \sigma_{-i}^*)$ denote the equilibrium payoff
for player~$i$.

\textbf{Part (b):} Suppose for contradiction that some $t_i \in S_i$
satisfies $u_i(t_i, \sigma_{-i}^*) > v_i^*$. Then the pure strategy $t_i$
(viewed as a degenerate mixed strategy) gives payoff
$u_i(t_i, \sigma_{-i}^*) > v_i^* = u_i(\sigma_i^*, \sigma_{-i}^*)$,
contradicting the NE condition. Hence $u_i(s_i, \sigma_{-i}^*) \leq v_i^*$
for all $s_i \in S_i$.

Now $v_i^*$ is a convex combination of the payoffs $u_i(s_i, \sigma_{-i}^*)$
with weights $\sigma_i^*(s_i) \geq 0$ summing to~1. Since every term in
the combination satisfies $u_i(s_i, \sigma_{-i}^*) \leq v_i^*$, if any
term in the support were strictly less than $v_i^*$, some other term would
need to be strictly greater (to make the average equal $v_i^*$), which we
just ruled out. Therefore $u_i(s_i, \sigma_{-i}^*) = v_i^*$ for all
$s_i \in \supp(\sigma_i^*)$, and $u_i(t_i, \sigma_{-i}^*) \leq v_i^*$
for all $t_i \notin \supp(\sigma_i^*)$.

\textbf{Part (a):} Immediate from the equality
$u_i(s_i, \sigma_{-i}^*) = v_i^*$ for all $s_i \in \supp(\sigma_i^*)$.
\end{proof}

\begin{example}[Mixed NE in Matching Pennies]
\label{ex:mixed_mp}
In Matching Pennies, let player~1 play $H$ with probability $p$ and player~2
play $H$ with probability $q$. By the indifference principle, player~1 is
indifferent when:
\[
    u_1(H, q) = u_1(T, q) \implies q - (1-q) = -(q) + (1-q)
    \implies 2q - 1 = 1 - 2q \implies q = \tfrac{1}{2}.
\]
Similarly, player~2 is indifferent when $p = 1/2$. The unique mixed NE is
$\sigma_1 = (1/2, 1/2)$, $\sigma_2 = (1/2, 1/2)$, with expected payoffs
$(0, 0)$.
\end{example}

\begin{example}[Mixed NE in Battle of the Sexes]
\label{ex:mixed_bos}
In the Battle of the Sexes, let player~1 play $O$ with probability $p$ and
player~2 play $O$ with probability $q$. By the indifference principle:
\begin{align*}
    u_1(O, q) &= u_1(F, q) \implies 3q = 1 - q \implies q = \tfrac{1}{4}, \\
    u_2(p, O) &= u_2(p, F) \implies p = 3(1-p) \implies p = \tfrac{3}{4}.
\end{align*}
The mixed NE is $\sigma_1 = (3/4, 1/4)$, $\sigma_2 = (1/4, 3/4)$, with
expected payoffs $(3/4, 3/4)$. Both players fare worse than in either pure NE.
\end{example}

\begin{rlconnection}
In RL, a stochastic policy $\pi(a|s)$ is precisely a mixed strategy in the
stage game at state $s$. The indifference principle has a direct analog: in
an optimal stochastic policy, every action in the support must yield the same
Q-value. If some action had a strictly higher Q-value, the agent should place
all probability on it.
\end{rlconnection}

%% SECTION 2: NASH'S EXISTENCE THEOREM

\section{Nash's Existence Theorem}
\label{sec:nash_existence}

The central result of this lesson guarantees that every finite game has at
least one Nash equilibrium in mixed strategies. The proof uses Kakutani's
fixed point theorem.

\subsection{Kakutani's Fixed Point Theorem}
\label{subsec:kakutani}

\begin{definition}[Upper Hemicontinuity]
\label{def:uhc}
Let $X \subseteq \R^m$ and $Y \subseteq \R^n$. A correspondence
(set-valued map) $F : X \rightrightarrows Y$ is \emph{upper hemicontinuous}
(uhc) at $x_0 \in X$ if for every open set $V \supseteq F(x_0)$, there
exists an open set $U \ni x_0$ such that $F(x) \subseteq V$ for all
$x \in U \cap X$.

Equivalently (when $Y$ is compact), $F$ is uhc at $x_0$ if whenever
$x_k \to x_0$ and $y_k \in F(x_k)$ with $y_k \to y_0$, we have
$y_0 \in F(x_0)$. This is the \emph{closed graph} characterization.
\end{definition}

\begin{theorem}[Kakutani's Fixed Point Theorem]
\label{thm:kakutani}
Let $K \subseteq \R^n$ be non-empty, compact, and convex. Let
$F : K \rightrightarrows K$ be a correspondence such that:
\begin{enumerate}[nosep,label=(\roman*)]
    \item $F(x)$ is non-empty for every $x \in K$.
    \item $F(x)$ is convex for every $x \in K$.
    \item $F$ is upper hemicontinuous.
\end{enumerate}
Then $F$ has a fixed point: there exists $x^* \in K$ with $x^* \in F(x^*)$.
\end{theorem}

\begin{proof}
We prove via Brouwer's fixed point theorem. For each $k \in \N$, let $T_k$
be a simplicial triangulation of $K$ with mesh size at most $1/k$. For each
vertex $v$ of $T_k$, choose $f_k(v) \in F(v)$ (possible since $F(v)$ is
non-empty). Extend $f_k$ to all of $K$ by barycentric interpolation: if
$x = \sum_j \lambda_j v_j$ lies in a simplex with vertices $v_1, \ldots, v_p$,
set $f_k(x) = \sum_j \lambda_j f_k(v_j)$.

Each $f_k : K \to K$ is continuous (linear on each simplex, consistent on
shared faces by uniqueness of barycentric coordinates). Since $K$ is convex
and each $f_k(x)$ is a convex combination of points in $K$, we have
$f_k(x) \in K$. By Brouwer's theorem, each $f_k$ has a fixed point
$x_k \in K$ with $f_k(x_k) = x_k$.

By compactness, $(x_k)$ has a convergent subsequence $x_{k_l} \to x^*$.
For each $l$, let $v_1^l, \ldots, v_p^l$ be the vertices of the simplex in
$T_{k_l}$ containing $x_{k_l}$, with barycentric coordinates $\lambda_j^l$.
Then $x_{k_l} = \sum_j \lambda_j^l f_{k_l}(v_j^l)$ and
$\|v_j^l - x_{k_l}\| \leq 1/k_l \to 0$, so $v_j^l \to x^*$.

Since $f_{k_l}(v_j^l) \in F(v_j^l) \subseteq K$ (compact), passing to a
further subsequence (diagonal argument over finitely many vertex indices),
we may assume $f_{k_l}(v_j^l) \to y_j$. By the closed-graph
characterization of upper hemicontinuity: $v_j^l \to x^*$ and
$f_{k_l}(v_j^l) \to y_j$ with $f_{k_l}(v_j^l) \in F(v_j^l)$ implies
$y_j \in F(x^*)$.

The barycentric coordinates $\lambda_j^l \in [0,1]$ (compact), so passing
to a further subsequence, $\lambda_j^l \to \lambda_j^*$ with
$\sum_j \lambda_j^* = 1$, $\lambda_j^* \geq 0$. Taking the limit:
$x^* = \sum_j \lambda_j^* y_j$. Since each $y_j \in F(x^*)$ and $F(x^*)$
is convex, we conclude $x^* \in F(x^*)$.
\end{proof}

\subsection{Nash's Theorem}
\label{subsec:nash_theorem}

\begin{keyresult}
\begin{theorem}[Nash, 1950]
\label{thm:nash_existence}
Every finite normal-form game has at least one mixed strategy Nash
equilibrium.
\end{theorem}
\end{keyresult}

\begin{proof}
Let $\Gamma = (N, (S_i)_{i \in N}, (u_i)_{i \in N})$ be a finite game with
$|S_i| = m_i$. Define the set of mixed strategy profiles
$\Delta = \prod_{i \in N} \Delta(S_i)$, which is compact and convex in
$\R^{m_1 + \cdots + m_n}$.

\textbf{Step 1: Define the best response correspondence.}
For each player~$i$, define $\BR_i : \Delta_{-i} \rightrightarrows \Delta(S_i)$ by
\[
    \BR_i(\sigma_{-i}) = \argmax_{\sigma_i \in \Delta(S_i)}
    u_i(\sigma_i, \sigma_{-i}).
\]
Define the joint best response $\BR : \Delta \rightrightarrows \Delta$ by
$\BR(\sigma) = \prod_{i \in N} \BR_i(\sigma_{-i})$.

\textbf{Step 2: $\BR(\sigma)$ is non-empty.}
For each $i$, the map $\sigma_i \mapsto u_i(\sigma_i, \sigma_{-i})$ is
continuous (linear) on the compact set $\Delta(S_i)$. By the extreme value
theorem, the maximum is attained, so $\BR_i(\sigma_{-i}) \neq \emptyset$.

\textbf{Step 3: $\BR(\sigma)$ is convex.}
Fix $i$ and $\sigma_{-i}$. Let $\sigma_i, \sigma_i' \in \BR_i(\sigma_{-i})$
and $\lambda \in [0,1]$. Set
$\hat{\sigma}_i = \lambda \sigma_i + (1-\lambda)\sigma_i'$. By linearity of
$u_i$ in $\sigma_i$:
\[
    u_i(\hat{\sigma}_i, \sigma_{-i})
    = \lambda\, u_i(\sigma_i, \sigma_{-i})
    + (1-\lambda)\, u_i(\sigma_i', \sigma_{-i})
    = v_i^*,
\]
where $v_i^* = \max_{\tau_i} u_i(\tau_i, \sigma_{-i})$. So
$\hat{\sigma}_i \in \BR_i(\sigma_{-i})$, and $\BR_i(\sigma_{-i})$ is convex.
Since $\BR(\sigma)$ is a product of convex sets, it is convex.

\textbf{Step 4: $\BR$ is upper hemicontinuous.}
We use the closed-graph characterization. Let $\sigma^k \to \sigma^*$ in
$\Delta$ and $\tau^k \in \BR(\sigma^k)$ with $\tau^k \to \tau^*$. Fix
player~$i$. Since $\tau_i^k \in \BR_i(\sigma_{-i}^k)$, for every
$\sigma_i \in \Delta(S_i)$:
\[
    u_i(\tau_i^k, \sigma_{-i}^k) \geq u_i(\sigma_i, \sigma_{-i}^k).
\]
Since $u_i$ is continuous on $\Delta$, taking $k \to \infty$:
\[
    u_i(\tau_i^*, \sigma_{-i}^*) \geq u_i(\sigma_i, \sigma_{-i}^*).
\]
This holds for all $\sigma_i$, so $\tau_i^* \in \BR_i(\sigma_{-i}^*)$.
Hence $\tau^* \in \BR(\sigma^*)$.

\textbf{Step 5: Apply Kakutani.}
The correspondence $\BR : \Delta \rightrightarrows \Delta$ satisfies all
hypotheses of Theorem~\ref{thm:kakutani}. Therefore there exists
$\sigma^* \in \Delta$ with $\sigma^* \in \BR(\sigma^*)$, meaning
$\sigma_i^* \in \BR_i(\sigma_{-i}^*)$ for all $i$, so $\sigma^*$ is a
Nash equilibrium.
\end{proof}

\begin{remark}[Alternative Proof via Brouwer]
\label{rem:brouwer_proof}
An alternative proof avoids Kakutani by constructing a continuous function
whose fixed points are NE. For each player $i$ and pure strategy $s_i^j$,
define the ``excess payoff''
$g_i^j(\sigma) = \max\{0,\, u_i(s_i^j, \sigma_{-i}) - u_i(\sigma)\}$,
and set
\[
    f_i^j(\sigma) = \frac{\sigma_i(s_i^j) + g_i^j(\sigma)}
    {1 + \sum_{k=1}^{m_i} g_i^k(\sigma)}.
\]
One verifies: (i) $f$ maps $\Delta$ to $\Delta$; (ii) $f$ is continuous;
(iii) $\sigma^*$ is a fixed point of $f$ iff $g_i^j(\sigma^*) = 0$ for all
$i, j$, iff $\sigma^*$ is a NE. Brouwer's theorem yields a fixed point.
\end{remark}

\begin{rlconnection}
Nash's theorem guarantees that every finite MARL problem (modeled as a
stochastic game with finite state and action spaces) has at least one Nash
equilibrium in the stage games. This justifies the goal of Nash-Q learning
\citep{shoham2008}. However, the theorem is non-constructive: computing a
Nash equilibrium is PPAD-complete, meaning no polynomial-time algorithm is
known.
\end{rlconnection}

%% SECTION 3: THE MINIMAX THEOREM

\section{Von Neumann's Minimax Theorem}
\label{sec:minimax}

\subsection{Two-Player Zero-Sum Games}
\label{subsec:zero_sum}

\begin{definition}[Two-Player Zero-Sum Game]
\label{def:zero_sum}
A two-player game $(N, (S_1, S_2), (u_1, u_2))$ is \emph{zero-sum} if
$u_1(s) + u_2(s) = 0$ for all $s \in S$. It is specified by a single payoff
matrix $A \in \R^{m \times n}$ where $u_1(i,j) = A_{ij}$ and
$u_2(i,j) = -A_{ij}$. Player~1 (row) maximizes; player~2 (column) minimizes.
\end{definition}

\begin{definition}[Maximin and Minimax Values]
\label{def:maximin}
For a zero-sum game with payoff matrix $A$, define:
\begin{align*}
    \underline{v} &= \max_{x \in \Delta_m} \min_{y \in \Delta_n}
    x^\top A y \quad \text{(maximin value)}, \\
    \overline{v} &= \min_{y \in \Delta_n} \max_{x \in \Delta_m}
    x^\top A y \quad \text{(minimax value)},
\end{align*}
where $\Delta_k$ denotes the $(k-1)$-simplex in $\R^k$.
\end{definition}

\begin{lemma}[Maximin $\leq$ Minimax]
\label{lem:maximin_leq}
For any matrix $A \in \R^{m \times n}$, $\underline{v} \leq \overline{v}$.
\end{lemma}

\begin{proof}
For any fixed $x \in \Delta_m$ and $y \in \Delta_n$:
\[
    \min_{y' \in \Delta_n} x^\top A y' \leq x^\top A y
    \leq \max_{x' \in \Delta_m} (x')^\top A y.
\]
Taking $\max$ over $x$ on the left side and $\min$ over $y$ on the right:
\[
    \underline{v} = \max_x \min_{y'} x^\top A y'
    \leq \min_y \max_{x'} (x')^\top A y = \overline{v}. \qedhere
\]
\end{proof}

\begin{keyresult}
\begin{theorem}[Von Neumann, 1928]
\label{thm:minimax}
For every matrix $A \in \R^{m \times n}$,
\[
    \max_{x \in \Delta_m} \min_{y \in \Delta_n} x^\top A y
    = \min_{y \in \Delta_n} \max_{x \in \Delta_m} x^\top A y.
\]
The common value $v^* = \underline{v} = \overline{v}$ is the \emph{value}
of the game.
\end{theorem}
\end{keyresult}

\begin{proof}
We prove $\underline{v} \geq \overline{v}$ using LP duality; combined with
Lemma~\ref{lem:maximin_leq}, equality follows.

\textbf{Step 1: Reformulate as linear programs.}
Since $y \mapsto x^\top A y$ is linear in $y$ and $\Delta_n$ is a simplex,
the minimum over $y$ is attained at a vertex $e_j$, giving
$\min_y x^\top A y = \min_{j=1,\ldots,n} (A^\top x)_j$. Thus
$\underline{v} = \max_{x \in \Delta_m} \min_j (A^\top x)_j$. Introducing
an auxiliary variable $v$:
\begin{align}
    \text{(P)} \quad \max\;&\; v \notag \\
    \text{s.t.}\;&\; A^\top x \geq v\, \mathbf{1}_n, \quad
    x \in \Delta_m, \quad v \in \R.
    \label{eq:primal}
\end{align}

\textbf{Step 2: The dual LP.}
Similarly, $\overline{v} = \min_{y \in \Delta_n} \max_i (Ay)_i$, which
gives:
\begin{align}
    \text{(D)} \quad \min\;&\; w \notag \\
    \text{s.t.}\;&\; Ay \leq w\, \mathbf{1}_m, \quad
    y \in \Delta_n, \quad w \in \R.
    \label{eq:dual}
\end{align}

\textbf{Step 3: Apply strong LP duality.}
Programs (P) and (D) are LP duals of each other. Both are feasible: take
$x = (1/m, \ldots, 1/m)^\top$ for (P) and $y = (1/n, \ldots, 1/n)^\top$
for (D). Both are bounded (the feasible sets are compact). By strong LP
duality, the optimal values coincide:
\[
    \underline{v} = \text{opt(P)} = \text{opt(D)} = \overline{v}. \qedhere
\]
\end{proof}

\begin{corollary}[NE in Zero-Sum Games]
\label{cor:ne_zero_sum}
In a two-player zero-sum game with payoff matrix $A$ and value $v^*$:
\begin{enumerate}[nosep,label=(\alph*)]
    \item $(x^*, y^*)$ is a NE iff $x^*$ is maximin and $y^*$ is minimax.
    \item All NE yield payoffs $u_1 = v^*$, $u_2 = -v^*$.
    \item NE strategies are interchangeable: if $(x^*, y^*)$ and
    $(x^{**}, y^{**})$ are both NE, then so are $(x^*, y^{**})$ and
    $(x^{**}, y^*)$.
\end{enumerate}
\end{corollary}

\begin{proof}
\textbf{(a) $(\Rightarrow)$:} Let $(x^*, y^*)$ be a NE. Then
$x^*$ best responds to $y^*$: $(x^*)^\top A y^* = \max_x x^\top A y^*$.
Also $y^*$ best responds to $x^*$: $(x^*)^\top A y^* = \min_y (x^*)^\top Ay$.
Therefore
$\min_y (x^*)^\top Ay \leq \underline{v} \leq \overline{v}
\leq \max_x x^\top Ay^*$,
and the minimax theorem gives $\underline{v} = \overline{v} = v^*$, so all
inequalities are equalities: $x^*$ achieves the maximin and $y^*$ achieves
the minimax.

$(\Leftarrow)$: If $x^*$ achieves the maximin and $y^*$ achieves the
minimax, then $(x^*)^\top Ay^* \geq \min_y (x^*)^\top Ay = v^*
= \max_x x^\top Ay^* \geq (x^*)^\top Ay^*$, forcing equality. For any $x$:
$x^\top Ay^* \leq v^* = (x^*)^\top Ay^*$, so $x^*$ best responds to $y^*$.
Similarly $y^*$ best responds to $x^*$. Hence $(x^*, y^*)$ is a NE.

\textbf{(b)} Immediate from part (a).

\textbf{(c)} If $x^*$ and $x^{**}$ are both maximin and $y^*$, $y^{**}$
both minimax, then any maximin-minimax pair forms a NE by part (a).
\end{proof}

\subsection{Solving Zero-Sum Games via LP}
\label{subsec:lp}

\begin{proposition}[Zero-Sum Games via LP]
\label{prop:lp_zerosum}
The maximin strategy $x^*$ and game value $v^*$ are computed by solving
LP~\eqref{eq:primal}. The minimax strategy $y^*$ is the dual solution from
LP~\eqref{eq:dual}. Both LPs have polynomial-time algorithms, so computing
NE in two-player zero-sum games is in~$\mathrm{P}$.
\end{proposition}

\begin{proof}
This follows from the reformulation in the proof of
Theorem~\ref{thm:minimax}. LP~\eqref{eq:primal} has $m + 1$ variables and
$n + m + 1$ constraints, solvable in polynomial time by interior point
methods.
\end{proof}

\begin{warning}
While NE in zero-sum games is polynomial via LP, computing a NE in a general
two-player game is PPAD-complete. This complexity barrier is a fundamental
challenge for MARL.
\end{warning}

\begin{rlconnection}
The minimax theorem is the foundation of: (i) \textbf{Minimax-Q learning}
\citep{shoham2008}, where the Bellman operator computes
$V(s) = \max_x \min_y x^\top Q(s, \cdot, \cdot) y$; (ii) \textbf{AlphaZero},
where MCTS approximates minimax values; (iii) \textbf{GANs}, where
$\min_G \max_D V(D,G)$ is a continuous zero-sum game whose saddle points are
justified by minimax theory.
\end{rlconnection}

\section{Correlated Equilibrium}
\label{sec:correlated}

\begin{definition}[Correlated Equilibrium]
\label{def:ce}
A \emph{correlated equilibrium} (CE) of a finite game
$\Gamma = (N, (S_i), (u_i))$ is a probability distribution $\pi$ over
$S = \prod_{i \in N} S_i$ such that for every player $i$ and every pair
$s_i, s_i' \in S_i$:
\[
    \sum_{s_{-i} \in S_{-i}} \pi(s_i, s_{-i})\, u_i(s_i, s_{-i})
    \geq \sum_{s_{-i} \in S_{-i}} \pi(s_i, s_{-i})\, u_i(s_i', s_{-i}).
\]
Equivalently, if a mediator draws $s$ from $\pi$ and privately recommends
$s_i$ to player $i$, no player has incentive to deviate.
\end{definition}

\begin{keyresult}
\begin{theorem}[Every NE Is a CE]
\label{thm:ne_is_ce}
If $\sigma^* = (\sigma_1^*, \ldots, \sigma_n^*)$ is a mixed strategy NE,
then the product distribution
$\pi(s) = \prod_{i \in N} \sigma_i^*(s_i)$ is a correlated equilibrium.
\end{theorem}
\end{keyresult}

\begin{proof}
Fix player $i$ and strategies $s_i, s_i' \in S_i$. If $\sigma_i^*(s_i) = 0$,
both sides of the CE constraint are~0 and the inequality holds trivially.
If $\sigma_i^*(s_i) > 0$, since $\pi$ is a product distribution,
$\pi(s_i, s_{-i}) = \sigma_i^*(s_i) \prod_{j \neq i} \sigma_j^*(s_j)$.
Dividing both sides of the CE constraint by $\sigma_i^*(s_i) > 0$ gives:
\[
    \sum_{s_{-i}} \bigg(\prod_{j \neq i} \sigma_j^*(s_j)\bigg) u_i(s_i, s_{-i})
    \geq \sum_{s_{-i}} \bigg(\prod_{j \neq i} \sigma_j^*(s_j)\bigg)
    u_i(s_i', s_{-i}),
\]
i.e., $u_i(s_i, \sigma_{-i}^*) \geq u_i(s_i', \sigma_{-i}^*)$.
Since $s_i \in \supp(\sigma_i^*)$, the indifference principle
(Theorem~\ref{thm:indifference}) gives exactly this inequality.
\end{proof}

\begin{proposition}[CE as a Linear Program]
\label{prop:ce_lp}
The set of correlated equilibria is a convex polytope defined by the linear
CE constraints, $\pi(s) \geq 0$, and $\sum_s \pi(s) = 1$. Optimizing any
linear objective over the CE set is an LP.
\end{proposition}

\begin{proof}
The CE constraints are linear in $\{\pi(s)\}_{s \in S}$:
\[
    \sum_{s_{-i}} \pi(s_i, s_{-i})\big[u_i(s_i, s_{-i})
    - u_i(s_i', s_{-i})\big] \geq 0
    \quad \text{for all } i, s_i, s_i'.
\]
Combined with the simplex constraints, the feasible set is a polytope.
\end{proof}

\begin{example}[CE in Battle of the Sexes]
\label{ex:ce_bos}
Consider the ``traffic light'' CE: a mediator recommends $(O,O)$ with
probability $1/2$ and $(F,F)$ with probability $1/2$. Then
$\pi(O,O) = \pi(F,F) = 1/2$ and $\pi(O,F) = \pi(F,O) = 0$.

Check for player~1 given recommendation $O$:
$\pi(O,O)\, u_1(O,O) = 3/2 \geq \pi(O,O)\, u_1(F,O) = 0$. Given $F$:
$\pi(F,F)\, u_1(F,F) = 1/2 \geq \pi(F,F)\, u_1(O,F) = 0$. Similarly for
player~2. Expected payoffs: $2$ for each player, Pareto-dominating the
mixed NE payoffs of $3/4$ each.
\end{example}

\begin{rlconnection}
Correlated equilibrium has key advantages for MARL: (i) finding a CE is an LP
(polynomial time), unlike NE (PPAD-complete); (ii) no-regret learning
dynamics (e.g., regret matching) converge to CE naturally; (iii) CE can
achieve better social welfare than any NE \citep{aumann1974}.
\end{rlconnection}

\section{Exercises}
\label{sec:exercises}

\begin{exercise}[Mixed NE Computation]
\label{ex:mixed_ne_compute}
Compute all Nash equilibria (pure and mixed) of the game:
\[
\begin{array}{c|cc}
 & L & R \\ \hline
T & (4,2) & (1,3) \\
B & (2,3) & (3,1)
\end{array}
\]
Verify the indifference principle for each mixed NE.
\end{exercise}

\begin{exercise}[Support Enumeration]
\label{ex:support_enum}
Consider the $3 \times 3$ game:
\[
\begin{array}{c|ccc}
 & L & C & R \\ \hline
T & (1,1) & (0,0) & (0,0) \\
M & (0,0) & (1,1) & (0,0) \\
B & (0,0) & (0,0) & (1,1)
\end{array}
\]
\begin{enumerate}[nosep,label=(\alph*)]
    \item Find all pure strategy NE.
    \item Find all mixed NE with support of size~2 for each player.
    \item Find the fully mixed NE (support of size~3 for each player).
    \item Show that the set of NE is not convex.
\end{enumerate}
\end{exercise}

\begin{exercise}[Minimax Computation]
\label{ex:minimax_compute}
For the zero-sum game with payoff matrix
\[
    A = \begin{pmatrix} 3 & -1 \\ -2 & 4 \end{pmatrix},
\]
compute the value $v^*$ and optimal mixed strategies for both players.
Formulate as an LP and verify the minimax theorem directly.
\end{exercise}

\begin{exercise}[Correlated Equilibrium Construction]
\label{ex:ce_construct}
In the game of Chicken:
\[
\begin{array}{c|cc}
 & D & S \\ \hline
D & (-5,-5) & (2,-1) \\
S & (-1,2) & (0,0)
\end{array}
\]
\begin{enumerate}[nosep,label=(\alph*)]
    \item Find all pure and mixed Nash equilibria.
    \item Construct a CE that Pareto-dominates the mixed NE.
    \item Write the CE incentive constraints explicitly and verify your
    solution satisfies them.
\end{enumerate}
\end{exercise}

\begin{exercise}[CE Polytope Structure]
\label{ex:ce_polytope}
Prove that the set of correlated equilibria is closed under convex
combinations: if $\pi$ and $\pi'$ are CE and $\lambda \in [0,1]$, then
$\lambda \pi + (1-\lambda)\pi'$ is a CE. Conclude that the CE set is
a convex polytope.
\end{exercise}

\begin{exercise}[Kakutani Conditions]
\label{ex:kakutani_verify}
Consider the two-player game where $S_1 = S_2 = \{A, B\}$ with payoffs
$(3,3)$ at $(A,A)$, $(0,0)$ at $(A,B)$ and $(B,A)$, and $(1,1)$ at $(B,B)$.
\begin{enumerate}[nosep,label=(\alph*)]
    \item Write the best response correspondence $\BR_i(\sigma_{-i})$
    explicitly as a function of the opponent's mixing probability.
    \item Verify that $\BR_i$ is non-empty-valued, convex-valued, and
    upper hemicontinuous.
    \item Find all fixed points of $\BR$ (i.e., all NE).
\end{enumerate}
\end{exercise}

\begin{exercise}[Minimax for $3 \times 3$]
\label{ex:minimax_3x3}
For the zero-sum game with payoff matrix
\[
    A = \begin{pmatrix} 1 & -1 & 0 \\ -1 & 1 & 0 \\ 0 & 0 & 0
    \end{pmatrix},
\]
show that the game value is $v^* = 0$ and find all maximin and minimax
strategies. \textit{Hint:} consider the role of the third row/column.
\end{exercise}

\begin{exercise}[NE $\subsetneq$ CE]
\label{ex:ne_strict_subset}
Give an example of a $2 \times 2$ game and a correlated equilibrium $\pi$
that is \emph{not} a product distribution (and hence not a NE viewed as a
joint distribution). Prove that your $\pi$ satisfies the CE constraints but
cannot be written as $\pi(s) = \sigma_1(s_1)\, \sigma_2(s_2)$ for any
mixed strategy profile $(\sigma_1, \sigma_2)$.
\end{exercise}

\noindent\textbf{Summary.} This lesson established: (1) mixed strategies and
the multilinear expected payoff; (2) the indifference principle
characterizing mixed NE support; (3) Nash's existence theorem via Kakutani's
fixed point theorem; (4) the minimax theorem for zero-sum games via LP
duality, with the corollary that zero-sum NE are polynomial to compute; and
(5) correlated equilibrium as a computationally tractable relaxation of NE.
These results form the strategic foundation for multi-agent RL in Phase~05.
