%% SECTION 1: MOTIVATION AND OVERVIEW

\section{Motivation and Overview}
\label{sec:motivation}

This lesson covers two topics central to multi-agent RL: \emph{signaling
games with perfect Bayesian equilibrium} and \emph{potential games}.

Signaling games extend the Bayesian framework of Lesson~37 to dynamic
settings where an informed player communicates through costly actions.
The equilibrium concept must combine sequential rationality with belief
updating, leading to PBE. The sender's message may fully reveal her type
(separating), hide it (pooling), or partially disclose it
(semi-separating).

Potential games are games where all incentives are captured by a single
scalar function. This guarantees pure NE existence and convergence of
simple learning dynamics---a rare property in multi-agent settings.

\begin{rlconnection}[title={Signaling and Potential Structure in MARL}]
In MARL, agents' actions serve dual purposes: achieving rewards and
signaling intentions. Understanding PBE predicts what information is
communicated in emergent protocols. Potential games provide one of the
few classes where independent learners are guaranteed to converge to NE.
Potential-based reward shaping leverages this structure to align
individual and team objectives.
\end{rlconnection}

%% SECTION 3: PERFECT BAYESIAN EQUILIBRIUM

\section{Perfect Bayesian Equilibrium}
\label{sec:beliefs-pbe}

In dynamic games of incomplete information, BNE (Lesson~37) does not
constrain behavior at unreached information sets. We need a refinement
combining sequential rationality with belief consistency.

\begin{definition}[System of Beliefs]
\label{def:beliefs}
A \emph{system of beliefs} $\mu$ assigns to each information set $h$ a
probability distribution $\mu(\cdot|h)$ over the nodes in $h$:
$\mu(x|h) \geq 0$ for $x \in h$ and $\sum_{x \in h}\mu(x|h) = 1$.
\end{definition}

\begin{definition}[Perfect Bayesian Equilibrium]
\label{def:pbe}
A \emph{perfect Bayesian equilibrium} (PBE) is a pair $(\sigma, \mu)$
consisting of a behavioral strategy profile $\sigma$ and a belief system
$\mu$ satisfying:
\begin{enumerate}[label=(\alph*)]
    \item \textbf{Sequential rationality:} At every information set $h$
    of every player $i$, the strategy $\sigma_i$ maximizes player $i$'s
    expected payoff given beliefs $\mu(\cdot|h)$ and the continuation
    strategies $\sigma_{-i}$.
    \item \textbf{Belief consistency:} At every information set $h$
    reached with positive probability under $\sigma$
    ($\Prob^\sigma(h) > 0$), beliefs are derived via Bayes' rule:
    \[
        \mu(x|h) = \frac{\Prob^\sigma(x \text{ is reached})}
        {\sum_{x' \in h}\Prob^\sigma(x' \text{ is reached})}
        \quad \text{for all } x \in h.
    \]
    \item At information sets reached with zero probability (off the
    equilibrium path), beliefs $\mu(\cdot|h)$ are unrestricted.
\end{enumerate}
\end{definition}

\begin{remark}
The freedom in specifying off-path beliefs is both a strength and a
weakness. It allows equilibria to be supported by pessimistic beliefs
(``if I see an unexpected message, I assume the worst''), but may permit
equilibria sustained by implausible beliefs. Refinements such as the
intuitive criterion address this but are beyond our scope.
\end{remark}

\begin{proposition}[Every PBE is a BNE]
\label{prop:pbe-bne}
Every PBE strategy profile is a Bayesian Nash equilibrium.
\end{proposition}

\begin{proof}
BNE requires ex ante optimality; PBE requires optimality at every
information set (a stronger condition). Optimality at each information
set implies optimality of the ex ante weighted average.
\end{proof}

%% SECTION 4: SIGNALING GAMES

\section{Signaling Games}
\label{sec:signaling}

\begin{definition}[Signaling Game]
\label{def:signaling-game}
A \emph{signaling game} has a Sender with private type
$\theta \in \Theta$ drawn from prior $p(\theta)$, who chooses message
$m \in M$; and a Receiver who observes $m$, forms belief
$\mu(\theta|m)$, and chooses action $a \in A$. Payoffs
$u_S(\theta,m,a)$ and $u_R(\theta,m,a)$ depend on all three.
\end{definition}

\begin{definition}[Equilibrium Types]
\label{def:eq-types}
A PBE is \emph{separating} if different types send different messages;
\emph{pooling} if all types send the same message; and
\emph{semi-separating} if some types pool and others separate.
\end{definition}

\begin{example}[Spence's Job Market Signaling]
\label{ex:spence}
Worker ability $\theta \in \{L, H\}$ with $\Prob(H) = \pi$. Worker
chooses education $e \geq 0$; employer offers wage $w(e)$. Payoffs:
$u_S = w - c(\theta, e)$ and $u_R = \theta - w$, where the
\emph{single-crossing property} holds: $c(H,e) < c(L,e)$ for $e > 0$.

\textbf{Separating PBE.} Type $H$ chooses $e_H > 0$, type $L$ chooses
$e_L = 0$. Bayes' rule: $\mu(H|e_H) = 1$, $\mu(H|0) = 0$, so
$w(e_H) = H$, $w(0) = L$. Incentive compatibility requires:
\begin{align*}
    H - c(H, e_H) &\geq L \quad\text{(type $H$ prefers $e_H$)}, \\
    L &\geq H - c(L, e_H) \quad\text{(type $L$ prefers $e_L = 0$)}.
\end{align*}
The single-crossing property ($c(L,e) > c(H,e)$) ensures a range of
valid $e_H$ exists. Off-path beliefs $\mu(H|e) = 0$ for
$e \notin \{0, e_H\}$ support the equilibrium.

\textbf{Pooling PBE.} Both types choose $e^* = 0$:
$\mu(H|0) = \pi$, $w(0) = \pi H + (1-\pi)L$. Off-path:
$\mu(H|e) = 0$ for $e > 0$, giving $w(e) = L$. No type deviates
since $L - c(\theta,e) < L \leq w(0)$.
\end{example}

\begin{proposition}[PBE Existence in Finite Signaling Games]
\label{prop:pbe-existence}
Every finite signaling game has a PBE.
\end{proposition}

\begin{proof}
Form the agent normal form: one agent per sender-type (choosing $m$) and
one per message (receiver choosing $a$). By Nash's theorem, this finite
game has a mixed NE. Translating back gives $\sigma_S$ and $\sigma_R$;
Bayes' rule determines on-path beliefs, and off-path beliefs are chosen
freely. Sequential rationality holds because each agent's strategy is a
best response.
\end{proof}

%% SECTION 5: POTENTIAL GAMES

\section{Potential Games}
\label{sec:potential-games}

\begin{definition}[Exact Potential Game]
\label{def:exact-potential}
A finite game $G = (N, (A_i)_{i \in N}, (u_i)_{i \in N})$ is an
\emph{exact potential game} if there exists a function
$\Phi : A \to \R$ (the \emph{potential function}) such that for every
player $i \in N$, every $a_{-i} \in A_{-i}$, and every
$a_i, a_i' \in A_i$:
\[
    u_i(a_i, a_{-i}) - u_i(a_i', a_{-i})
    = \Phi(a_i, a_{-i}) - \Phi(a_i', a_{-i}).
\]
That is, the change in any player's payoff from a unilateral deviation
equals the change in the potential.
\end{definition}

\begin{definition}[Ordinal Potential Game]
\label{def:ordinal-potential}
A finite game $G$ is an \emph{ordinal potential game} if there exists
$\Phi : A \to \R$ such that for every $i$, $a_{-i}$, and $a_i, a_i'$:
\[
    u_i(a_i, a_{-i}) > u_i(a_i', a_{-i})
    \iff \Phi(a_i, a_{-i}) > \Phi(a_i', a_{-i}).
\]
Every exact potential game is ordinal (equal differences have equal
signs); the converse is false.
\end{definition}

\begin{example}[Coordination Game as Potential Game]
\label{ex:coordination}
The coordination game with payoffs $(T,L) = (2,2)$, $(T,R) = (0,0)$,
$(B,L) = (0,0)$, $(B,R) = (1,1)$ is an exact potential game with
$\Phi(T,L) = 2$, $\Phi(T,R) = 0$, $\Phi(B,L) = 0$, $\Phi(B,R) = 1$.
Verification: $u_1(T,L) - u_1(B,L) = 2 = \Phi(T,L) - \Phi(B,L)$;
$u_2(T,L) - u_2(T,R) = 2 = \Phi(T,L) - \Phi(T,R)$; and similarly for
all other unilateral deviations.
\end{example}

\begin{keyresult}[title={Pure NE in Potential Games}]
\begin{theorem}[Potential Games Have Pure NE]
\label{thm:potential-pure-ne}
Every finite exact (or ordinal) potential game has a pure Nash
equilibrium.
\end{theorem}
\end{keyresult}

\begin{proof}
Since $A$ is finite, $\Phi$ attains its maximum at some $a^*$. For any
player $i$ and deviation $a_i'$:
$\Phi(a_i^*, a_{-i}^*) \geq \Phi(a_i', a_{-i}^*)$ by maximality.
The potential property gives
$u_i(a_i^*, a_{-i}^*) - u_i(a_i', a_{-i}^*) \geq 0$, so $a^*$ is a
NE. For ordinal potentials: if
$u_i(a_i', a_{-i}^*) > u_i(a_i^*, a_{-i}^*)$ held, the ordinal
property would give
$\Phi(a_i', a_{-i}^*) > \Phi(a_i^*, a_{-i}^*)$, contradicting
maximality.
\end{proof}

%% SECTION 6: CONVERGENCE OF DYNAMICS

\section{Convergence of Learning Dynamics}
\label{sec:convergence}

\begin{definition}[Best-Response Dynamics]
\label{def:br-dynamics}
In \emph{best-response dynamics}, at each step one player $i$ switches
to $a_i^{t+1} \in \argmax_{a_i}u_i(a_i, a_{-i}^t)$; others keep
$a_j^{t+1} = a_j^t$. In \emph{better-response dynamics}, the switch
need only strictly improve $i$'s payoff.
\end{definition}

\begin{keyresult}[title={Finite Improvement Property}]
\begin{theorem}[Finite Improvement Property]
\label{thm:fip}
In every finite exact potential game, any sequence of strict unilateral
improvements terminates in finitely many steps at a pure NE.
\end{theorem}
\end{keyresult}

\begin{proof}
If player $i$ switches from $a_i^t$ to $a_i^{t+1}$ with
$u_i(a_i^{t+1}, a_{-i}^t) > u_i(a_i^t, a_{-i}^t)$, the potential
property gives $\Phi(a^{t+1}) > \Phi(a^t)$. Since $A$ is finite, a
strictly increasing sequence of $\Phi$-values must terminate. At
termination, no player can improve, so the profile is a NE.
\end{proof}

\begin{corollary}[Better-Response Convergence]
\label{cor:better-response}
Better-response dynamics also converge to a pure NE in finitely many
steps, since the proof uses only the strict improvement property.
\end{corollary}

%% SECTION 7: CONGESTION GAMES

\section{Congestion Games and Rosenthal's Theorem}
\label{sec:congestion}

\begin{definition}[Congestion Game]
\label{def:congestion-game}
A \emph{congestion game} is $(N, E, (A_i), (d_e))$ where $N$ is the
player set, $E$ is a finite set of resources, $A_i \subseteq 2^E$ is
$i$'s action set (subsets of resources to use), and
$d_e : \{1,\ldots,n\} \to \R$ is the delay for resource $e$ depending
on congestion $n_e(a) = |\{j : e \in a_j\}|$. Player $i$'s cost is
$C_i(a) = \sum_{e \in a_i}d_e(n_e(a))$, with payoff $u_i = -C_i$.
\end{definition}

\begin{keyresult}[title={Rosenthal's Theorem}]
\begin{theorem}[Congestion Games are Exact Potential Games]
\label{thm:rosenthal}
Every congestion game is an exact potential game with
$\Phi(a) = -\sum_{e \in E}\sum_{k=1}^{n_e(a)}d_e(k)$.
\end{theorem}
\end{keyresult}

\begin{proof}
Fix player $i$, $a_{-i}$, and two actions $a_i, a_i'$. Let
$n_e = n_e(a_i, a_{-i})$ and $n_e' = n_e(a_i', a_{-i})$. We check
three cases for each resource $e$:

\textbf{(i) $e \in a_i \cap a_i'$:} $n_e = n_e'$; contributions
cancel in both differences.

\textbf{(ii) $e \in a_i \setminus a_i'$:} $n_e = n_e' + 1$.
Contribution to $u_i - u_i'$: $-d_e(n_e'+1)$.
Contribution to $\Phi - \Phi'$:
$-\sum_{k=1}^{n_e'+1}d_e(k) + \sum_{k=1}^{n_e'}d_e(k)
= -d_e(n_e'+1)$. Equal.

\textbf{(iii) $e \in a_i' \setminus a_i$:} $n_e' = n_e + 1$.
Contribution to $u_i - u_i'$: $d_e(n_e+1)$.
Contribution to $\Phi - \Phi'$: $d_e(n_e+1)$. Equal.

Summing over all resources gives
$u_i(a_i, a_{-i}) - u_i(a_i', a_{-i}) = \Phi(a_i, a_{-i})
- \Phi(a_i', a_{-i})$.
\end{proof}

\begin{corollary}[Congestion Games Have Pure NE]
\label{cor:congestion-ne}
Every finite congestion game has at least one pure Nash equilibrium, and
best-response dynamics converge to one in finitely many steps.
\end{corollary}

\begin{proof}
Immediate from Theorems~\ref{thm:rosenthal},
\ref{thm:potential-pure-ne}, and~\ref{thm:fip}.
\end{proof}

\begin{remark}
The converse of Rosenthal's theorem also holds:
\citet{monderer1996} showed that every finite exact potential game is
isomorphic (in best-response structure) to a congestion game. Thus the
classes of exact potential games and congestion games essentially
coincide.
\end{remark}

\begin{example}[Network Routing]
\label{ex:routing}
Source $s$ to destination $d$ via two parallel links: $d_A(k) = k$,
$d_B(k) = 2 + k/2$, with $n = 4$ users. Each user chooses one link.
Profile $(3,1)$: delays $d_A(3) = 3$, $d_B(1) = 2.5$. No user
improves by switching, so this is a NE. The Rosenthal potential selects
among multiple NE based on which the dynamics reach.
\end{example}

\begin{example}[Resource Allocation]
\label{ex:resource-alloc}
$n$ agents choose among $m$ machines with linear congestion
$d_e(k) = k\ell_e$. The potential is
$\Phi(a) = -\sum_e\ell_e\,n_e(n_e+1)/2$. Pure NE correspond to
load-balanced configurations.
\end{example}

%% SECTION 8: RL CONNECTIONS

\section{Connections to Multi-Agent RL}
\label{sec:rl-connections}

\begin{rlconnection}[title={Independent Learners in Potential Games}]
Potential games guarantee convergence of independent learners:
best-response dynamics converge (Theorem~\ref{thm:fip}); log-linear
learning ($\Prob(a_i) \propto \exp(\beta\,u_i)$) concentrates on
potential maximizers as $\beta \to \infty$; stochastic better-response
with exploration converges to the global $\Phi$-maximizer.
\end{rlconnection}

\begin{rlconnection}[title={Signaling and Emergent Communication}]
Signaling game theory predicts three outcomes for MARL communication:
separating equilibria (meaningful messages), pooling (babbling---a common
failure mode), and semi-separating (partial information). The
single-crossing property predicts when separating equilibria emerge.
\end{rlconnection}

\begin{rlconnection}[title={Potential-Based Reward Shaping}]
The difference reward $D_i(a) = R(a) - R(a_{-i}, c_i)$ satisfies
$D_i(a_i, a_{-i}) - D_i(a_i', a_{-i}) = R(a_i, a_{-i})
- R(a_i', a_{-i})$, making $R$ an exact potential for individual payoffs
$D_i$. This turns cooperative MARL into a potential game, ensuring
independent learners converge while maximizing the team objective.
\end{rlconnection}

\begin{rlconnection}[title={Opponent Modeling via PBE}]
PBE formalizes Bayesian opponent modeling: maintain prior
$p(\theta_{-i})$, update via Bayes' rule after observing actions, act
optimally under the posterior. The off-path belief freedom warns that
learned opponent models may be fragile to surprising actions.
\end{rlconnection}

\section{Exercises}
\label{sec:exercises}

\begin{exercise}[Simple Signaling PBE (*)]
\label{ex:simple-signaling}
Consider a signaling game with $\Theta = \{G,B\}$,
$\Prob(\theta = G) = 2/3$, $M = \{m_1, m_2\}$, $A = \{a_1, a_2\}$.
Receiver payoffs: $u_R(\theta, m, a_1) = 1$ if $\theta = G$,
$-1$ if $\theta = B$; $u_R(\theta, m, a_2) = 0$. Sender payoffs:
\begin{align*}
    u_S(G, m_1, a_1) &= 3, \; u_S(G, m_1, a_2) = 1, \;
    u_S(G, m_2, a_1) = 2, \; u_S(G, m_2, a_2) = 0, \\
    u_S(B, m_1, a_1) &= 1, \; u_S(B, m_1, a_2) = 2, \;
    u_S(B, m_2, a_1) = 3, \; u_S(B, m_2, a_2) = 0.
\end{align*}
\begin{enumerate}[label=(\alph*)]
    \item Does a separating PBE exist where $G$ sends $m_1$ and $B$
    sends $m_2$?
    \item Does a pooling PBE exist where both types send $m_1$?
    \item Does a pooling PBE exist where both types send $m_2$?
\end{enumerate}
\end{exercise}

\begin{exercise}[Verifying a Potential Function (**)]
\label{ex:verify-potential}
Consider the game with $A_1 = A_2 = \{X, Y\}$ and payoffs:
\[
\begin{array}{c|cc}
  & X & Y \\ \hline
X & 3,2 & 1,1 \\
Y & 2,1 & 4,3
\end{array}
\]
\begin{enumerate}[label=(\alph*)]
    \item Show this is an exact potential game by constructing $\Phi$.
    \item Find all pure Nash equilibria using $\Phi$.
    \item Trace best-response dynamics from every initial profile and
    verify convergence.
\end{enumerate}
\end{exercise}

\begin{exercise}[Three-Player Congestion Game (**)]
\label{ex:congestion-three}
Three players share two resources $\{A, B\}$. Each player chooses
exactly one resource. Delay functions: $d_A(k) = k$, $d_B(k) = 2k - 1$.
\begin{enumerate}[label=(\alph*)]
    \item Write the Rosenthal potential $\Phi$.
    \item Enumerate all action profiles (up to symmetry) and compute
    $\Phi$ for each.
    \item Find all pure NE by identifying $\Phi$-maximizers.
    \item Verify your NE by checking no player wants to deviate.
\end{enumerate}
\end{exercise}

\begin{exercise}[Spence Signaling with Specific Costs (**)]
\label{ex:spence-specific}
In Example~\ref{ex:spence}, let $H = 2$, $L = 1$, $\pi = 1/2$,
$c(H,e) = e/2$, and $c(L,e) = e$.
\begin{enumerate}[label=(\alph*)]
    \item Find the range of $e_H$ for which a separating PBE exists
    with $e_L = 0$.
    \item Identify the least-cost separating equilibrium.
    \item Verify the pooling PBE with $e^* = 0$ and compute the pooling
    wage.
    \item Compare total surplus (worker + firm payoff) in the least-cost
    separating PBE versus the pooling PBE.
\end{enumerate}
\end{exercise}

\begin{exercise}[Not Every Game is a Potential Game (**)]
\label{ex:not-potential}
Consider Matching Pennies:
\[
\begin{array}{c|cc}
  & H & T \\ \hline
H & 1,-1 & -1,1 \\
T & -1,1 & 1,-1
\end{array}
\]
\begin{enumerate}[label=(\alph*)]
    \item Set up the four equations that an exact potential $\Phi$ must
    satisfy and show they are inconsistent.
    \item Does this game have a pure NE? Relate to
    Theorem~\ref{thm:potential-pure-ne}.
    \item Show that no ordinal potential exists either.
\end{enumerate}
\end{exercise}

\begin{exercise}[Network Routing with Three Links (***)]
\label{ex:routing-three}
Three parallel links with delays $d_1(k) = k$, $d_2(k) = k + 1$,
$d_3(k) = 2k$. There are $n = 6$ users, each choosing one link.
\begin{enumerate}[label=(\alph*)]
    \item Write the Rosenthal potential function.
    \item Find a pure NE by applying best-response dynamics from the
    initial profile $(6, 0, 0)$.
    \item Is the NE you found the global maximizer of $\Phi$?
    \item Compute the price of anarchy: the ratio of total delay at the
    worst NE to the socially optimal total delay.
\end{enumerate}
\end{exercise}

\begin{exercise}[PBE with Three Types (***)]
\label{ex:pbe-three-types}
Sender has type $\theta \in \{1,2,3\}$ with equal probability $1/3$.
Messages: $M = \{m_1, m_2\}$. Receiver actions: $A = \{a_1, a_2\}$.
Payoffs: $u_S(\theta, m, a_1) = \theta$, $u_S(\theta, m, a_2) = 0$;
$u_R(\theta, m, a_1) = \theta - 2$, $u_R(\theta, m, a_2) = 0$.
\begin{enumerate}[label=(\alph*)]
    \item Show that no fully separating PBE exists (three types, two
    messages).
    \item Find a semi-separating PBE where type~3 sends $m_1$ and
    types~1,2 send $m_2$.
    \item Find a pooling PBE where all types send $m_1$.
    \item Which equilibrium is most informative? Discuss.
\end{enumerate}
\end{exercise}

%% SECTION 10: SUMMARY

\section{Summary}
\label{sec:summary}

\begin{itemize}
    \item \textbf{PBE} combines sequential rationality with Bayesian
    belief consistency; every PBE is a BNE
    (Proposition~\ref{prop:pbe-bne}).
    \item \textbf{Signaling games:} sender-receiver structure with
    separating, pooling, and semi-separating equilibria. Spence's model
    illustrates the single-crossing property. Every finite signaling
    game has a PBE (Proposition~\ref{prop:pbe-existence}).
    \item \textbf{Exact potential games:} unilateral payoff changes
    equal potential changes. Pure NE exist at $\Phi$-maximizers
    (Theorem~\ref{thm:potential-pure-ne}).
    \item \textbf{Finite improvement property:} best/better-response
    dynamics converge in finite steps (Theorem~\ref{thm:fip}).
    \item \textbf{Rosenthal's theorem:} congestion games are exact
    potential games (Theorem~\ref{thm:rosenthal}), hence have pure NE.
    \item For MARL: potential games guarantee independent learner
    convergence; difference rewards create potential structure;
    signaling theory predicts emergent communication outcomes.
\end{itemize}
