%% SECTION 1: MOTIVATION AND OVERVIEW

\section{Motivation and Overview}
\label{sec:motivation}

The Nash equilibrium concept tells us what stable play \emph{looks like},
but says nothing about how players might \emph{arrive} at it. This lesson
develops two classical learning dynamics---fictitious play and
multiplicative weights update---and establishes their convergence in
zero-sum games, then introduces Bayesian games modeling strategic
interaction under incomplete information.

Both parts share a common theme: \emph{uncertainty about the opponent}.
In learning dynamics the uncertainty concerns the opponent's strategy; in
Bayesian games it concerns the opponent's type.

\begin{rlconnection}[title={From Game Learning to Multi-Agent RL}]
Fictitious play is the prototype for self-play algorithms (NFSP, PSRO),
while MWU is a precursor to policy gradient methods in competitive
settings. The Bayesian game framework underpins opponent modeling, where
an agent maintains beliefs about opponent types and updates via Bayes'
rule. These classical results provide the theoretical foundation for
modern MARL algorithms.
\end{rlconnection}

%% SECTION 3: FICTITIOUS PLAY

\section{Fictitious Play}
\label{sec:fictitious-play}

\begin{definition}[Fictitious Play]
\label{def:fictitious-play}
Consider a finite two-player game $G = (\{1,2\}, A_1, A_2, g_1, g_2)$.
\emph{Fictitious play} (FP) is a learning dynamic where each player
best-responds to the empirical frequency of the opponent's past actions.
The empirical frequency of player $j$ after $T$ rounds is
\[
    \hat{\alpha}_j^T(a_j) = \frac{1}{T} \sum_{t=1}^T
    \indicator\{a_j^t = a_j\}, \qquad a_j \in A_j.
\]
At round $T+1$, player $i$ plays
$a_i^{T+1} \in \BR_i(\hat{\alpha}_j^T)$.
\end{definition}

\begin{remark}
Fictitious play requires no knowledge of the opponent's payoff function---only
observation of the opponent's past actions. Each player acts \emph{as if}
the opponent plays a fixed mixed strategy equal to the empirical frequency,
and best-responds to this belief. The ``fictitious'' qualifier refers to
this assumption: the opponent is not actually stationary, but is
simultaneously updating.
\end{remark}

We work with two-player zero-sum games where $g_1 = g = -g_2$ and
$v^* = \max_{\alpha_1}\min_{\alpha_2}\E[g(\alpha_1,\alpha_2)]$ is the
value (Lesson~34, minimax theorem).

\begin{theorem}[FP Convergence in $2 \times 2$ Zero-Sum Games]
\label{thm:fp-2x2}
In any $2 \times 2$ zero-sum game, the empirical frequencies under
fictitious play converge to a Nash equilibrium as $T \to \infty$.
\end{theorem}

\begin{proof}
Let $R_k^T = \sum_{t=1}^T g(k, a_2^t)$ for rows $k \in \{1,2\}$ and
$D^T = R_1^T - R_2^T$. Player~1 plays row~1 when $D^T > 0$ and row~2
when $D^T < 0$. Player~2's best response to the heavier-weighted row
tends to decrease $|D^T|$. Each step changes $D^T$ by at most
$C_0 = \max_{a_2}|g(1,a_2) - g(2,a_2)|$, and the dynamics push $D$
back toward zero, so $|D^T|$ remains bounded.

Since $|D^T|$ is bounded, $D^T/T \to 0$. Define
$V_1^T = \frac{1}{T}\max_{a_1}\sum_{t=1}^T g(a_1, a_2^t)
= \frac{1}{T}\max\{R_1^T, R_2^T\}$. We have $V_1^T \geq v^*$ and
$V_1^T = (R_1^T + R_2^T)/(2T) + |D^T|/(2T) \to \bar{r}$. The
analogous quantity $V_2^T$ for player~2 satisfies
$V_1^T + V_2^T \to 0$, giving $V_1^T \to v^*$.

This forces $\hat{\alpha}_2^T$ to converge to a distribution making
player~1 indifferent (the NE condition), and symmetrically for
$\hat{\alpha}_1^T$.
\end{proof}

\begin{keyresult}[title={Robinson's Theorem}]
\begin{theorem}[Robinson's Theorem]
\label{thm:robinson}
In any finite two-player zero-sum game, the empirical frequencies under
fictitious play converge to the set of Nash equilibria.
\end{theorem}
\end{keyresult}

\begin{proof}
Define $\Phi_1^T = \max_{a_1}\sum_{t=1}^T g(a_1, a_2^t)$ and
$\Phi_2^T = \max_{a_2}\sum_{t=1}^T(-g(a_1^t, a_2))$.
Then $\Phi_1^T/T \geq v^*$ and $\Phi_2^T/T \geq -v^*$.

\textbf{Step 1.} We show $(\Phi_1^T + \Phi_2^T)/T \to 0$.
Let $S^T = \sum_{t=1}^T g(a_1^t, a_2^t)$. Then
$\Phi_1^T \geq S^T$ and $\Phi_2^T \geq -S^T$, so
$\Psi^T := \Phi_1^T + \Phi_2^T \geq 0$. For the upper bound, the
best-response property gives
$g(a_1^{T+1}, \hat{\alpha}_2^T) \geq \Phi_1^T/T$ and
$-g(\hat{\alpha}_1^T, a_2^{T+1}) \geq \Phi_2^T/T$. Tracking the
increments $\Psi^{T+1} - \Psi^T$ carefully (Robinson's original
argument), one shows $\Psi^T = O(\sqrt{T})$, hence $\Psi^T/T \to 0$.

\textbf{Step 2.} Since $\Phi_1^T/T \geq v^*$, $\Phi_2^T/T \geq -v^*$,
and their sum divided by $T$ tends to zero:
$\Phi_1^T/T \to v^*$ and $\Phi_2^T/T \to -v^*$.
Any limit point $(\alpha_1^*, \alpha_2^*)$ of the empirical frequencies
(which exists by compactness) satisfies
$\max_{a_1}g(a_1, \alpha_2^*) = v^* = \min_{a_2}g(\alpha_1^*, a_2)$,
the NE condition.
\end{proof}

\begin{rlconnection}[title={Fictitious Play and PSRO}]
Fictitious play is the ancestor of \emph{Policy-Space Response Oracles}
(PSRO) in modern MARL: each player maintains a population of policies
and computes a best response to the meta-mixture, exactly as in FP but
with neural-network policies. Robinson's theorem guarantees convergence
to NE in zero-sum games.
\end{rlconnection}

%% SECTION 4: NO-REGRET LEARNING

\section{No-Regret Learning and Multiplicative Weights}
\label{sec:no-regret}

\begin{definition}[External Regret]
\label{def:external-regret}
Player $i$'s \emph{external regret} after $T$ rounds is
\[
    R_i^T = \max_{a_i \in A_i} \sum_{t=1}^T g_i(a_i, a_{-i}^t)
    - \sum_{t=1}^T g_i(\alpha_i^t, a_{-i}^t),
\]
comparing cumulative payoff to the best fixed action in hindsight.
A learning algorithm is \emph{no-regret} (Hannan consistent) if
$R_i^T/T \to 0$ regardless of the opponent's sequence.
\end{definition}

\begin{definition}[Multiplicative Weights Update]
\label{def:mwu}
Let $K = |A_i|$ and $\eta > 0$ be a learning rate. The
\emph{multiplicative weights update} (MWU) maintains weights
$w^t(a_i)$ for each action $a_i \in A_i$:
\begin{align*}
    w^1(a_i) &= 1 \quad \text{for all } a_i \in A_i, \\
    w^{t+1}(a_i) &= w^t(a_i) \cdot
    \exp\bigl(\eta \cdot g_i(a_i, a_{-i}^t)\bigr).
\end{align*}
The mixed strategy at time $t$ is
$\alpha_i^t(a_i) = w^t(a_i)/W^t$, where
$W^t = \sum_{a_i' \in A_i} w^t(a_i')$.
\end{definition}

\begin{remark}
MWU increases the weight of actions that performed well and decreases the
weight of those that performed poorly. The exponential update ensures
weights remain positive. When $\eta$ is small, updates are conservative;
when $\eta$ is large, the algorithm tracks the best action aggressively
but may overshoot.
\end{remark}

\begin{keyresult}[title={MWU Regret Bound}]
\begin{theorem}[MWU Regret Bound]
\label{thm:mwu-regret}
If $|g_i(a)| \leq 1$ for all $a$, then with
$\eta = \sqrt{(\ln K)/T}$, MWU achieves $R_i^T \leq 2\sqrt{T\ln K}$.
\end{theorem}
\end{keyresult}

\begin{proof}
We use a potential function argument with $W^t = \sum_{a_i}w^t(a_i)$,
so $W^1 = K$.

\textbf{Upper bound on $\ln(W^{T+1}/W^1)$.}
\begin{align*}
    \frac{W^{t+1}}{W^t}
    &= \sum_{a_i}\alpha_i^t(a_i)
    \exp\bigl(\eta\, g_i(a_i, a_{-i}^t)\bigr).
\end{align*}
Since $|g_i| \leq 1$ and $\eta \leq 1$ (for $T \geq \ln K$), we have
$|\eta g_i| \leq 1$. Using $e^x \leq 1 + x + x^2$ for $|x| \leq 1$:
\[
    \frac{W^{t+1}}{W^t}
    \leq 1 + \eta\,g_i(\alpha_i^t, a_{-i}^t) + \eta^2.
\]
Since $1 + x \leq e^x$, taking logs and telescoping over $t = 1,\ldots,T$:
\[
    \ln \frac{W^{T+1}}{W^1}
    \leq \eta\sum_{t=1}^T g_i(\alpha_i^t, a_{-i}^t) + \eta^2 T.
\]

\textbf{Lower bound on $\ln(W^{T+1}/W^1)$.}
For any fixed action $a_i^* \in A_i$:
$W^{T+1} \geq w^{T+1}(a_i^*)
= \exp\bigl(\eta\sum_{t=1}^T g_i(a_i^*, a_{-i}^t)\bigr)$,
so $\ln(W^{T+1}/W^1) \geq \eta\sum_t g_i(a_i^*, a_{-i}^t) - \ln K$.

\textbf{Combining.} Rearranging:
$\sum_t g_i(a_i^*, a_{-i}^t) - \sum_t g_i(\alpha_i^t, a_{-i}^t)
\leq (\ln K)/\eta + \eta T$.
Taking the max over $a_i^*$: $R_i^T \leq (\ln K)/\eta + \eta T$.
Substituting $\eta = \sqrt{(\ln K)/T}$:
$R_i^T \leq 2\sqrt{T\ln K}$.
\end{proof}

\begin{rlconnection}[title={MWU and Policy Gradient}]
The MWU update $\alpha_i^{t+1} \propto \alpha_i^t\exp(\eta\,g_i(\cdot,
a_{-i}^t))$ is precisely the REINFORCE policy gradient update with
softmax parameterization. The $O(\sqrt{T})$ regret bound provides
convergence rate guarantees for policy gradient in competitive games.
\end{rlconnection}

%% SECTION 5: NO-REGRET IMPLIES NE

\section{No-Regret Learning Implies NE in Zero-Sum Games}
\label{sec:no-regret-ne}

\begin{keyresult}[title={No-Regret to Nash Equilibrium}]
\begin{theorem}[No-Regret Implies NE in Zero-Sum Games]
\label{thm:no-regret-ne}
In a two-player zero-sum game, if both players use no-regret algorithms,
the time-averaged strategies $\bar{\alpha}_i^T = \frac{1}{T}\sum_t
\alpha_i^t$ converge to the set of Nash equilibria.
\end{theorem}
\end{keyresult}

\begin{proof}
By no-regret for player~1 and linearity:
$\max_{a_1}g(a_1, \bar{\alpha}_2^T)
\leq \frac{1}{T}\sum_t g(\alpha_1^t, \alpha_2^t) + \varepsilon_1^T$
where $\varepsilon_1^T = R_1^T/T \to 0$.
Similarly for player~2:
$\min_{a_2}g(\bar{\alpha}_1^T, a_2)
\geq \frac{1}{T}\sum_t g(\alpha_1^t, \alpha_2^t) - \varepsilon_2^T$.

Combining: $\min_{a_2}g(\bar{\alpha}_1^T, a_2) \geq
\max_{a_1}g(a_1, \bar{\alpha}_2^T) - \varepsilon_1^T - \varepsilon_2^T$.
The minimax inequality gives the reverse direction, so the duality gap
$\max_{a_1}g(a_1, \bar{\alpha}_2^T) - \min_{a_2}g(\bar{\alpha}_1^T, a_2)
\leq \varepsilon_1^T + \varepsilon_2^T \to 0$.
Any limit point satisfies the NE condition.
\end{proof}

\begin{corollary}
\label{cor:mwu-rate}
With both players using MWU, the duality gap is at most
$4\sqrt{(\ln K)/T}$, giving $O(1/\sqrt{T})$ convergence to NE.
\end{corollary}

%% SECTION 6: BAYESIAN GAMES

\section{Bayesian Games}
\label{sec:bayesian-games}

\begin{definition}[Bayesian Game]
\label{def:bayesian-game}
A \emph{Bayesian game} (or game of incomplete information) is a tuple
$\Gamma = (N, (A_i)_{i \in N}, (\Theta_i)_{i \in N}, p,
(u_i)_{i \in N})$ where:
\begin{enumerate}[label=(\alph*)]
    \item $N = \{1, \ldots, n\}$ is the player set.
    \item $A_i$ is the finite action set of player $i$.
    \item $\Theta_i$ is the finite \emph{type space} of player $i$.
    \item $p \in \Delta(\Theta)$ is the \emph{common prior} over the type
    profile $\theta = (\theta_1, \ldots, \theta_n) \in
    \Theta = \prod_i \Theta_i$.
    \item $u_i : A \times \Theta \to \R$ is player $i$'s payoff.
\end{enumerate}
The timing is: nature draws $\theta \sim p$; each player $i$ observes
only $\theta_i$; players simultaneously choose actions.
\end{definition}

\begin{remark}
The \emph{common prior assumption} means all players agree on $p$.
Player $i$'s posterior belief is
$p(\theta_{-i}|\theta_i) = p(\theta_i, \theta_{-i})/
\sum_{\theta_{-i}'}p(\theta_i, \theta_{-i}')$.
\citet{harsanyi1967} showed any incomplete-information game can be
modeled this way.
\end{remark}

\begin{definition}[Bayesian Nash Equilibrium]
\label{def:bne}
A \emph{(pure) strategy} in a Bayesian game is a function
$s_i : \Theta_i \to A_i$ mapping types to actions. A strategy profile
$s^* = (s_1^*, \ldots, s_n^*)$ is a \emph{Bayesian Nash equilibrium}
(BNE) if, for every player $i$ and every type
$\theta_i \in \Theta_i$ with $p(\theta_i) > 0$:
\[
    s_i^*(\theta_i) \in \argmax_{a_i \in A_i}
    \E_{\theta_{-i} \sim p(\cdot|\theta_i)}\bigl[
    u_i(a_i, s_{-i}^*(\theta_{-i}), \theta_i, \theta_{-i})\bigr].
\]
A \emph{mixed} BNE allows $s_i(\theta_i) \in \Delta(A_i)$.
\end{definition}

\begin{remark}
The BNE condition says: each type of each player best-responds to the
\emph{expected} play of opponents, where the expectation is over
opponents' types (determining their actions through $s_{-i}^*$) using
the posterior $p(\theta_{-i}|\theta_i)$.
\end{remark}

\begin{keyresult}[title={BNE Existence}]
\begin{theorem}[Existence of BNE]
\label{thm:bne-existence}
Every finite Bayesian game has a BNE in mixed strategies.
\end{theorem}
\end{keyresult}

\begin{proof}
For each pair $(i, \theta_i)$, create an ``agent'' choosing from $A_i$
with payoff $\tilde{u}_{i,\theta_i} = \sum_{\theta_{-i}}
p(\theta_{-i}|\theta_i)\,u_i(a_{i,\theta_i}, a_{-i,\theta_{-i}},
\theta_i, \theta_{-i})$. This \emph{agent normal form} is a finite game,
so Nash's theorem (Lesson~34) guarantees a mixed NE. Translating back:
$s_i^*(\theta_i) = \sigma_{i,\theta_i}^*$ is a BNE, since the NE
condition for agent $(i,\theta_i)$ is exactly the BNE condition.
\end{proof}

%% SECTION 7: EXAMPLES

\section{Examples}
\label{sec:bayesian-examples}

\begin{example}[Cournot Duopoly with Unknown Costs]
\label{ex:cournot}
Inverse demand $P(Q) = a - q_1 - q_2$. Firm~1 has known cost $c_1$;
firm~2 has private cost $c_2 \in \{c_L, c_H\}$ with
$\Prob(c_2 = c_L) = \pi$. Profit:
$u_i = (a - q_1 - q_2 - c_i)q_i$. The BNE first-order conditions are:
\begin{align}
    q_1 &= \tfrac{1}{2}[a - c_1 - \pi\,s_2(c_L) - (1-\pi)s_2(c_H)],
    \label{eq:foc1}\\
    s_2(c) &= \tfrac{1}{2}(a - c - q_1) \quad
    \text{for } c \in \{c_L, c_H\}. \label{eq:foc2}
\end{align}
Substituting \eqref{eq:foc2} into \eqref{eq:foc1} with
$\bar{c} = \pi c_L + (1-\pi)c_H$ gives
$q_1 = (a - 2c_1 + \bar{c})/3$. Firm~1's quantity depends on the
\emph{expected} cost; firm~2 conditions on its actual type.
\end{example}

\begin{example}[First-Price Sealed-Bid Auction]
\label{ex:auction}
Two bidders with $v_i \sim \mathrm{Uniform}[0,1]$ independently.
Highest bidder wins and pays her bid:
$u_i = (v_i - b_i)\indicator\{b_i > b_j\}$. Seeking symmetric BNE
$b_i = \beta(v_i)$ with $\beta$ increasing, the FOC yields the ODE
$(v\beta)' = v$ with $\beta(0) = 0$, giving $\beta(v) = v/2$: each
bidder bids half her value.

\textbf{Verification.} Against $\beta(v_j) = v_j/2$, bidder $i$ solves
$\max_b(v_i - b)\cdot 2b$. The FOC gives $2v_i - 4b = 0$, hence
$b = v_i/2$, confirming the BNE.

The equilibrium bidding function $\beta(v) = v/2$ shows that each bidder
``shades'' her bid below her true value. The extent of shading depends
on the number of competitors and the value distribution.
\end{example}

\begin{rlconnection}[title={Opponent Modeling as Bayesian Inference}]
The Bayesian framework formalizes opponent modeling in MARL: type
$\theta_j$ encodes the opponent's policy parameters; the prior
$p(\theta_{-i})$ represents initial beliefs; Bayes' rule updates beliefs
as actions are observed; the BNE condition requires maximizing expected
return under the current posterior. Algorithms like LOLA and Bayesian
policy reuse implement this framework.
\end{rlconnection}

%% SECTION 8: EXERCISES

\section{Exercises}
\label{sec:exercises}

\begin{exercise}[Fictitious Play in Matching Pennies (*)]
\label{ex:fp-matching-pennies}
Consider the zero-sum game Matching Pennies:
\[
\begin{array}{c|cc}
  & H & T \\ \hline
H & 1,-1 & -1,1 \\
T & -1,1 & 1,-1
\end{array}
\]
\begin{enumerate}[label=(\alph*)]
    \item Compute the unique Nash equilibrium and the value of the game.
    \item Simulate fictitious play for 10 rounds starting with both
    players choosing $H$. Record the empirical frequencies at each step.
    \item Verify that the empirical frequencies approach the NE.
\end{enumerate}
\end{exercise}

\begin{exercise}[MWU Weight Computation (**)]
\label{ex:mwu-comp}
Player~1 has $A_1 = \{A,B,C\}$. The payoff sequence for rounds
$t = 1,2,3$ is $(1,0,-1)$, $(-1,1,0)$, $(0,-1,1)$.
\begin{enumerate}[label=(\alph*)]
    \item Starting with $w^1 = (1,1,1)$ and $\eta = 0.5$, compute
    weights and mixed strategies for $t = 1, 2, 3, 4$.
    \item Compute the external regret $R^3$.
    \item Compare with the theoretical bound $2\sqrt{3\ln 3}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[No-Regret in General-Sum Games (**)]
\label{ex:no-regret-general}
Consider the Prisoner's Dilemma:
\[
\begin{array}{c|cc}
  & L & R \\ \hline
T & 3,3 & 0,4 \\
B & 4,0 & 1,1
\end{array}
\]
\begin{enumerate}[label=(\alph*)]
    \item Find all Nash equilibria of this game.
    \item Suppose both players use no-regret learning. Show that the
    time-averaged play can converge to a correlated distribution
    placing high weight on $(T,L)$, which is not a NE.
    \item Explain why Theorem~\ref{thm:no-regret-ne} fails for
    general-sum games.
\end{enumerate}
\end{exercise}

\begin{exercise}[BNE Computation (**)]
\label{ex:bne-2x2}
Player~1 has type $\theta_1 \in \{H,L\}$ with $\Prob(H) = 1/2$.
Player~2 has no private information. Actions: $A_1 = \{U,D\}$,
$A_2 = \{L,R\}$. Payoffs:
\[
\text{Type } H: \;
\begin{array}{c|cc} & L & R \\ \hline U & 2,1 & 0,0 \\ D & 0,0 & 1,2
\end{array}
\quad
\text{Type } L: \;
\begin{array}{c|cc} & L & R \\ \hline U & 0,0 & 1,2 \\ D & 2,1 & 0,0
\end{array}
\]
\begin{enumerate}[label=(\alph*)]
    \item Formulate this as a Bayesian game and identify the agent
    normal form.
    \item Find all pure-strategy BNE.
    \item Find a mixed-strategy BNE where player~2 randomizes.
\end{enumerate}
\end{exercise}

\begin{exercise}[Second-Price Auction (*)]
\label{ex:second-price}
In a second-price sealed-bid auction with two bidders having private
values $v_i \sim \mathrm{Uniform}[0,1]$, the highest bidder wins and
pays the second-highest bid.
\begin{enumerate}[label=(\alph*)]
    \item Show that bidding $b_i = v_i$ (truthful bidding) is a weakly
    dominant strategy for each bidder.
    \item Conclude that truthful bidding is a BNE.
    \item Compare the expected seller revenue with the first-price
    auction of Example~\ref{ex:auction}.
\end{enumerate}
\end{exercise}

\begin{exercise}[MWU Convergence Rate (**)]
\label{ex:mwu-rate}
In Rock-Paper-Scissors (payoffs $0, \pm 1$):
\begin{enumerate}[label=(\alph*)]
    \item Compute the unique NE and the value of the game.
    \item If both players use MWU with $\eta = \sqrt{(\ln 3)/T}$, bound
    the duality gap after $T = 1000$ rounds.
    \item How many rounds $T$ are needed for the duality gap to drop
    below $0.01$?
\end{enumerate}
\end{exercise}

\begin{exercise}[Cournot Duopoly BNE (***)]
\label{ex:cournot-bne}
In Example~\ref{ex:cournot}, let $a = 10$, $c_1 = 2$, $c_L = 1$,
$c_H = 3$, and $\pi = 1/2$.
\begin{enumerate}[label=(\alph*)]
    \item Solve the system \eqref{eq:foc1}--\eqref{eq:foc2} for $q_1$,
    $s_2(c_L)$, and $s_2(c_H)$.
    \item Compute each firm's expected profit in the BNE.
    \item Compare with the complete-information outcomes: (i) $c_2 = c_L$
    known, (ii) $c_2 = c_H$ known.
    \item Does incomplete information help or hurt firm~1? Explain.
\end{enumerate}
\end{exercise}

\begin{exercise}[FP Non-Convergence in General-Sum Games (***)]
\label{ex:fp-shapley}
Shapley (1964) showed FP can fail to converge. Consider:
\[
\begin{array}{c|ccc}
  & L & C & R \\ \hline
T & 0,0 & 1,0 & 0,1 \\
M & 0,1 & 0,0 & 1,0 \\
B & 1,0 & 0,1 & 0,0
\end{array}
\]
\begin{enumerate}[label=(\alph*)]
    \item Verify the unique NE is $(1/3,1/3,1/3)$ for each player.
    \item Simulate FP for 20 rounds from $(T,L)$.
    \item Show the empirical frequencies cycle without converging.
    \item Discuss implications for using FP in general MARL settings.
\end{enumerate}
\end{exercise}

%% SECTION 9: SUMMARY

\section{Summary}
\label{sec:summary}

\begin{itemize}
    \item \textbf{Fictitious play:} best-respond to empirical
    frequencies. Converges to NE in $2 \times 2$ zero-sum games
    (Theorem~\ref{thm:fp-2x2}) and all finite zero-sum games
    (Robinson, Theorem~\ref{thm:robinson}).
    \item \textbf{MWU} achieves regret
    $R^T \leq 2\sqrt{T\ln K}$ (Theorem~\ref{thm:mwu-regret}).
    \item \textbf{No-regret implies NE} in zero-sum games at rate
    $O(1/\sqrt{T})$ (Theorem~\ref{thm:no-regret-ne}).
    \item \textbf{Bayesian games} model incomplete information via types;
    \textbf{BNE} requires each type to best-respond given posterior
    beliefs. Existence in finite games via agent normal form
    (Theorem~\ref{thm:bne-existence}).
    \item Examples: Cournot duopoly with unknown costs, first-price
    auctions with bid shading $\beta(v) = v/2$.
\end{itemize}
