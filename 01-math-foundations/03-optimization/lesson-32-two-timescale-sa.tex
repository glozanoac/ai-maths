\section{Motivation and Overview}
\label{sec:motivation}

Lesson 31 established \emph{that} stochastic approximation algorithms
converge. This lesson addresses three natural follow-up questions:
(1)~\emph{how fast} do they converge? (2)~what happens when two coupled
SA iterations run at \emph{different rates}? and (3)~how does the
general theory map to specific RL algorithms?

The convergence rate analysis yields a central limit theorem (CLT) for
SA and the Polyak--Ruppert averaging trick, which achieves the optimal
rate without knowledge of the problem parameters. Two-timescale SA,
developed by \citet{borkar1997}, provides the convergence framework for
actor-critic algorithms. The linear SA theory gives the sharpest
results for TD methods with function approximation.

\begin{rlconnection}[title={Why This Lesson Matters for RL}]
\begin{itemize}[nosep]
  \item The \textbf{CLT for SA} characterizes the asymptotic variance
    of RL estimators, explaining why REINFORCE has high variance and
    motivating baselines.
  \item \textbf{Polyak--Ruppert averaging} underlies target networks
    (DQN), exponential moving averages (DDPG/TD3), and iterate
    averaging in policy optimization.
  \item \textbf{Two-timescale SA} is the convergence engine for
    actor-critic methods: the critic (fast) estimates the value
    function while the actor (slow) improves the policy.
  \item \textbf{Linear SA} provides the convergence guarantee for
    TD(0) with linear function approximation and explains the
    \emph{deadly triad} (divergence under off-policy + FA +
    bootstrapping).
\end{itemize}
\end{rlconnection}

\medskip
\noindent\textbf{Prerequisites.} Lesson 31 (Robbins--Monro iteration,
ODE method, Borkar--Meyn theorem). The martingale CLT, basic matrix
analysis (Hurwitz matrices, Lyapunov equations), and familiarity with
RL update rules (TD(0), Q-learning, REINFORCE, actor-critic).

\section{Convergence Rate Analysis}
\label{sec:clt}

Having established convergence (Lesson 31, Theorem 3.6), we now ask:
\emph{how fast} does $x_n \to x^*$?

\subsection{CLT Assumptions}

\begin{definition}[CLT Assumptions for SA]
\label{def:clt-assumptions}
Assume the ODE method assumptions (B1)--(B5) from Lesson 31 hold,
and additionally:
\begin{enumerate}[label=(D\arabic*)]
  \item \label{D1} $h$ is differentiable at $x^*$ with Jacobian
    $A = Dh(x^*) \in \R^{d \times d}$.
  \item \label{D2} $A$ is Hurwitz: all eigenvalues of $A$ have
    strictly negative real parts.
  \item \label{D3} The step size is $a_n = a/n$ for some constant
    $a > 0$ such that $aA + \frac{1}{2}I$ is Hurwitz (i.e., all
    eigenvalues of $aA$ have real part less than $-1/2$).
  \item \label{D4} The conditional covariance converges:
    $\E[M_{n+1} M_{n+1}^T \mid \mathcal{F}_n] \to \Sigma$ in
    probability as $x_n \to x^*$, for some positive definite
    $\Sigma$.
\end{enumerate}
\end{definition}

\begin{remark}
Condition \ref{D3} requires that the step-size constant $a$ be
``large enough'' relative to the spectral radius of $A$. If $A$ has
eigenvalues $\{\lambda_i\}$, the condition becomes
$\mathrm{Re}(a\lambda_i) < -1/2$ for all $i$. When $d = 1$ and
$A = -\lambda$ ($\lambda > 0$), this gives $a > 1/(2\lambda)$.
\end{remark}

\subsection{The CLT for Stochastic Approximation}

\begin{theorem}[CLT for Stochastic Approximation]
\label{thm:sa-clt}
Under the assumptions of Definition~\ref{def:clt-assumptions},
\[
  \sqrt{n}(x_n - x^*) \cdist \mathcal{N}(0, V),
\]
where $V$ is the unique solution of the \emph{Lyapunov equation}
\begin{equation}
\label{eq:lyapunov}
  (aA + \tfrac{1}{2}I) V + V(aA + \tfrac{1}{2}I)^T
    + a^2 \Sigma = 0.
\end{equation}
\end{theorem}

\begin{proof}
Define $e_n = x_n - x^*$ and $u_n = \sqrt{n}\,e_n$. Near $x^*$,
$h(x_n) = Ae_n + o(\|e_n\|)$. With $a_n = a/n$ and
$\sqrt{(n+1)/n} = 1 + 1/(2n) + O(1/n^2)$, substituting
$e_n = u_n/\sqrt{n}$:
\[
  u_{n+1} = u_n + \frac{1}{n}\Bigl(aA + \frac{1}{2}I\Bigr)u_n
    + \frac{a}{\sqrt{n}}M_{n+1} + o(1/n).
\]
This is a linear stochastic recursion with drift matrix
$aA + \frac{1}{2}I$ (Hurwitz by \ref{D3}) and noise covariance
$a^2\Sigma$. The covariance $V_n = \E[u_nu_n^T]$ satisfies
$V_{n+1} \approx V_n + \frac{1}{n}[(aA+\frac{1}{2}I)V_n +
V_n(aA+\frac{1}{2}I)^T + a^2\Sigma]$.
At the fixed point, the Lyapunov equation~\eqref{eq:lyapunov}
determines the unique $V \succ 0$. The martingale CLT applied to
the linearized recursion gives $u_n \cdist \mathcal{N}(0,V)$.
\end{proof}

\begin{remark}[Optimal Step Size]
\label{rem:optimal-step}
When $d = 1$ and $A = -\lambda$ ($\lambda > 0$), the Lyapunov
equation gives $V = a^2\sigma^2/(2a\lambda - 1)$. Minimizing over
$a > 1/(2\lambda)$ yields $a_{\mathrm{opt}} = 1/\lambda$ and
$V_{\mathrm{opt}} = \sigma^2/\lambda^2$. In higher dimensions,
the optimal $a$ depends on the unknown Jacobian $A$.
\end{remark}

\subsection{Polyak--Ruppert Averaging}

The optimal step size depends on the Jacobian $A$ at $x^*$, which
is typically unknown. \citet{polyak1992} proposed averaging the
iterates to achieve the optimal rate \emph{without} knowing $A$.

\begin{theorem}[Polyak--Ruppert Averaging]
\label{thm:polyak-ruppert}
Let $\{x_n\}$ be the SA iterates with step size
$a_n = a/n^\alpha$ for $\alpha \in (1/2, 1)$, and define
\[
  \bar{x}_n = \frac{1}{n}\sum_{k=1}^{n} x_k.
\]
Under assumptions (B1)--(B5) and (D1)--(D4) (with the modified step
size), we have
\[
  \sqrt{n}(\bar{x}_n - x^*) \cdist
    \mathcal{N}(0,\, A^{-1}\Sigma A^{-T}).
\]
This is the optimal asymptotic covariance (matching the
Cramer--Rao bound for the root-finding problem).
\end{theorem}

\begin{proof}
Since $x_n \to x^*$ and $h(x^*) = 0$, write
$x_{n+1} - x_n = a_n[A(x_n - x^*) + r_n + M_{n+1}]$ where
$r_n = o(\|x_n - x^*\|)$. Summing from $k=1$ to $n$:
\[
  x_{n+1} - x_1 = A\sum_{k=1}^n a_k(x_k - x^*)
    + \sum_{k=1}^n a_k r_k + \sum_{k=1}^n a_k M_{k+1}.
\]
An Abel summation argument (using $a_k^{-1} \sim (k/a)^\alpha$
grows polynomially while $a_k(x_k-x^*) \to 0$) shows
\[
  \sqrt{n}(\bar{x}_n - x^*)
    = -A^{-1}\cdot\frac{1}{\sqrt{n}}\sum_{k=1}^n M_{k+1} + o_P(1).
\]
The remainders vanish: $x_{n+1} - x_1 = o(\sqrt{n})$ and
$\sum a_k r_k$ is negligible since $\alpha < 1$. By the martingale
CLT, $\frac{1}{\sqrt{n}}\sum M_{k+1} \cdist \mathcal{N}(0,\Sigma)$,
so $\sqrt{n}(\bar{x}_n - x^*) \cdist \mathcal{N}(0, A^{-1}\Sigma A^{-T})$.
\end{proof}

\begin{keyresult}[title={Polyak--Ruppert Averaging}]
By averaging the iterates of a slowly decaying SA scheme
($a_n = O(n^{-\alpha})$, $\alpha \in (1/2,1)$), the averaged
iterate $\bar{x}_n$ achieves the optimal convergence rate
$O(1/\sqrt{n})$ with the minimal asymptotic covariance
$A^{-1}\Sigma A^{-T}$, \emph{without} knowledge of the Jacobian
$A = Dh(x^*)$.
\end{keyresult}

\section{Two-Timescale Stochastic Approximation}
\label{sec:two-timescale}

\subsection{Motivation}

Many RL algorithms involve two coupled iterative updates running
simultaneously at different rates. The canonical example is
actor-critic: the critic updates the value function (fast timescale)
while the actor updates the policy (slow timescale).

\subsection{Formulation}

\begin{definition}[Two-Timescale SA]
\label{def:two-timescale}
Consider the coupled iterations
\begin{align}
  x_{n+1} &= x_n + a_n[f(x_n, y_n) + M_{n+1}^{(1)}],
    \label{eq:slow} \\
  y_{n+1} &= y_n + b_n[g(x_n, y_n) + M_{n+1}^{(2)}],
    \label{eq:fast}
\end{align}
where $x_n \in \R^{d_1}$, $y_n \in \R^{d_2}$, and the step-size
sequences satisfy:
\begin{enumerate}[label=(\roman*)]
  \item $\sum a_n = \sum b_n = \infty$ and $\sum a_n^2 < \infty$,
    $\sum b_n^2 < \infty$.
  \item $a_n/b_n \to 0$ as $n \to \infty$
    \quad (\emph{timescale separation}).
\end{enumerate}
The typical choice is $a_n = 1/n$ and $b_n = 1/n^\beta$ with
$\beta \in (1/2, 1)$.
\end{definition}

\begin{remark}[Timescale Separation]
\label{rem:timescale}
The condition $a_n/b_n \to 0$ means $y$ uses \emph{larger} step
sizes than $x$, so $y$ moves faster. From the $y$-perspective,
$x$ changes so slowly it appears quasi-static. From the
$x$-perspective, $y$ equilibrates so quickly it appears to have
converged.
\end{remark}

\subsection{Two-Timescale Assumptions}

\begin{definition}[Two-Timescale Assumptions]
\label{def:two-ts-assumptions}
\begin{enumerate}[label=(E\arabic*)]
  \item \label{E1} \textbf{Step sizes:} As in
    Definition~\ref{def:two-timescale}.
  \item \label{E2} \textbf{Lipschitz drifts:}
    $f : \R^{d_1} \times \R^{d_2} \to \R^{d_1}$ and
    $g : \R^{d_1} \times \R^{d_2} \to \R^{d_2}$ are globally
    Lipschitz.
  \item \label{E3} \textbf{Martingale difference noise:}
    $\E[M_{n+1}^{(i)} \mid \mathcal{F}_n] = 0$ and
    $\E[\|M_{n+1}^{(i)}\|^2 \mid \mathcal{F}_n] \leq
    K(1 + \|x_n\|^2 + \|y_n\|^2)$ for $i = 1,2$.
  \item \label{E4} \textbf{Stability:}
    $\sup_n(\|x_n\| + \|y_n\|) < \infty$ a.s.
  \item \label{E5} \textbf{Fast timescale equilibrium:} For each
    fixed $x \in \R^{d_1}$, the ODE $\dot{y} = g(x,y)$ has a
    globally asymptotically stable equilibrium $\lambda(x)$, and
    $\lambda : \R^{d_1} \to \R^{d_2}$ is Lipschitz.
  \item \label{E6} \textbf{Slow timescale stability:} The ODE
    $\dot{x} = f(x, \lambda(x))$ has a globally asymptotically
    stable equilibrium $x^*$.
\end{enumerate}
\end{definition}

\subsection{Convergence Theorem}

\begin{theorem}[Two-Timescale Convergence (Borkar)]
\label{thm:two-timescale}
Under assumptions \ref{E1}--\ref{E6}, the two-timescale SA iterates
satisfy
\[
  x_n \cas x^*, \quad y_n \cas \lambda(x^*) \eqqcolon y^*.
\]
\end{theorem}

\begin{proof}
\noindent\textbf{Phase 1 (fast timescale).}
Since $a_n/b_n \to 0$, on any window of fast-timescale length $T$
the total change in $x$ satisfies $\|x_{m_n} - x_n\| \leq
C\sum_{k=n}^{m_n-1}a_k \leq CT\sup_{k \geq n}(a_k/b_k) \to 0$.
So $x$ is quasi-static on the fast timescale. The $y$-iteration
behaves like SA with frozen $x$, converging to $\lambda(x)$ by the
ODE method. Thus $y_n$ tracks $\lambda(x_n)$: for any $\varepsilon > 0$,
$\|y_n - \lambda(x_n)\| < \varepsilon$ for all large $n$ a.s.

\medskip\noindent\textbf{Phase 2 (slow timescale).}
Since $y_n \approx \lambda(x_n)$, the Lipschitz property of $f$
gives $f(x_n,y_n) = f(x_n,\lambda(x_n)) + o(1)$. The $x$-iteration
behaves like SA with drift $f(x,\lambda(x))$, whose ODE has $x^*$
as globally stable equilibrium by \ref{E6}. The ODE method gives
$x_n \cas x^*$, and continuity of $\lambda$ yields
$y_n \to \lambda(x^*) = y^*$.
\end{proof}

\begin{keyresult}[title={Two-Timescale Convergence}]
In a two-timescale SA system with $a_n/b_n \to 0$:
\begin{enumerate}[nosep]
  \item The fast iterate $y_n$ tracks the equilibrium map
    $\lambda(x_n)$ where $g(x, \lambda(x)) = 0$.
  \item The slow iterate $x_n$ converges as if $y_n$ were always
    at equilibrium, driven by $\dot{x} = f(x, \lambda(x))$.
  \item The overall system converges to
    $(x^*, y^*) = (x^*, \lambda(x^*))$.
\end{enumerate}
\end{keyresult}

\section{Linear Stochastic Approximation}
\label{sec:linear-sa}

The most important special case for RL is \emph{linear} SA.

\begin{definition}[Linear SA]
\label{def:linear-sa}
A \emph{linear stochastic approximation} iteration is
\begin{equation}
\label{eq:linear-sa}
  x_{n+1} = x_n + a_n[Ax_n + b + M_{n+1}],
\end{equation}
where $A \in \R^{d \times d}$, $b \in \R^d$, and $h(x) = Ax + b$.
The associated ODE is $\dot{x} = Ax + b$ with equilibrium
$x^* = -A^{-1}b$ (when $A$ is invertible).
\end{definition}

\begin{theorem}[Convergence of Linear SA]
\label{thm:linear-sa}
Suppose $A$ is Hurwitz, the step sizes satisfy the Robbins--Monro
conditions, the noise satisfies (B3) from Lesson 31, and the iterates
are bounded a.s. Then $x_n \cas x^* = -A^{-1}b$.
\end{theorem}

\begin{proof}
Since $A$ is Hurwitz, there exists $P \succ 0$ with $A^TP + PA = -I$.
The Lyapunov function $V(x) = (x-x^*)^TP(x-x^*)$ satisfies
$\dot{V} = (x-x^*)^T(A^TP+PA)(x-x^*) = -\|x-x^*\|^2$ along ODE
trajectories (using $Ax+b = A(x-x^*)$). So $x^*$ is globally
asymptotically stable, and the ODE method gives $x_n \cas x^*$.
\end{proof}

\begin{corollary}[Automatic Stability for Linear SA]
\label{cor:linear-stability}
For linear SA~\eqref{eq:linear-sa} with $A$ Hurwitz, the boundedness
assumption $\sup_n \|x_n\| < \infty$ a.s.\ holds automatically.
\end{corollary}

\begin{proof}
Apply the Borkar--Meyn theorem (Lesson 31, Theorem 5.2). For
$h(x) = Ax + b$, the scaled limit is
$h_\infty(x) = \lim_{r \to \infty}(A(rx)+b)/r = Ax$. Since $A$ is
Hurwitz, the ODE $\dot{x} = Ax$ has the origin as globally
asymptotically stable equilibrium, so (C2) from Lesson 31 is
satisfied.
\end{proof}

\begin{theorem}[Complete Convergence for Linear SA]
\label{thm:linear-sa-complete}
For the linear SA iteration~\eqref{eq:linear-sa} with $A$ Hurwitz,
Robbins--Monro step sizes, and noise satisfying (B3) from Lesson 31,
we have $x_n \cas -A^{-1}b$ without any additional stability
assumptions.
\end{theorem}

\begin{proof}
Combine Corollary~\ref{cor:linear-stability} (boundedness) with
Theorem~\ref{thm:linear-sa} (convergence given boundedness).
\end{proof}

\begin{keyresult}[title={Linear SA: The Hurwitz Condition}]
For linear SA with drift $h(x) = Ax + b$, convergence to
$x^* = -A^{-1}b$ is equivalent to $A$ being Hurwitz (all
eigenvalues have negative real parts). The Hurwitz condition is
both necessary and sufficient: if $A$ has any eigenvalue with
positive real part, the iteration diverges.
\end{keyresult}

\section{Lock-In to Attractors}
\label{sec:lock-in}

When the ODE has multiple locally stable equilibria, the SA iteration
converges to one of them. The following result characterizes which
equilibria are reachable.

\begin{proposition}[Lock-In to Attractors]
\label{prop:lock-in}
Suppose the ODE $\dot{x} = h(x)$ has finitely many equilibria, each
either asymptotically stable or unstable. Under the ODE method
assumptions (B1)--(B4) from Lesson 31, the SA iterates converge
a.s.\ to one of the \emph{stable} equilibria. The probability of
converging to an unstable equilibrium is zero.
\end{proposition}

\begin{proof}
By the ODE method, the limit set of $\{x_n\}$ is a compact connected
ODE-flow-invariant set. By the avoidance of unstable equilibria
result (Pemantle, 1990; Brandiere, 1998): with nondegenerate noise
($\E[M_{n+1}M_{n+1}^T \mid \mathcal{F}_n]$ positive definite), the
probability of converging to an unstable equilibrium is zero. The
noise ``kicks'' the iterate off the unstable manifold, and the ODE
dynamics carry it toward a stable equilibrium.
\end{proof}

\begin{warning}[title={Multiple Stable Equilibria in RL}]
When the mean-field ODE has multiple stable equilibria, the RL
algorithm may converge to a suboptimal one. This is a fundamental
challenge in policy gradient methods where $J(\theta)$ is generally
non-concave. Two-timescale SA (actor-critic) inherits this issue.
Entropy regularization (as in SAC) and exploration bonuses are
practical techniques to mitigate convergence to poor local optima.
\end{warning}

\section{RL Applications}
\label{sec:rl-apps}

We now map the general SA theory to specific RL algorithms.

\subsection{TD(0) as Stochastic Approximation}

\begin{rlconnection}[title={TD(0) -- Complete SA Analysis}]
\textbf{Update.}
$V_{n+1}(s_n) = V_n(s_n) + \alpha_n(s_n)
  [R_{n+1} + \gamma V_n(s_{n+1}) - V_n(s_n)]$.

\textbf{SA form.} Linear SA with $A = \gamma P^\pi - I$,
$b = r^\pi$. Since eigenvalues of $P^\pi$ satisfy $|\lambda_i| \leq 1$
and $\gamma < 1$, all eigenvalues of $A$ have real part at most
$\gamma - 1 < 0$, so $A$ is Hurwitz. By
Theorem~\ref{thm:linear-sa-complete}, TD(0) converges a.s.\ to
$V^\pi = (I - \gamma P^\pi)^{-1}r^\pi$.

\textbf{With linear FA:} $V(s) = \phi(s)^T w$, $A = \Phi^T D(\gamma
P^\pi - I)\Phi$ where $D = \diag(d^\pi)$. On-policy: $A$ is negative
definite, TD converges. Off-policy: $A$ may not be Hurwitz --- this is
the deadly triad.
\end{rlconnection}

\subsection{Q-Learning as Asynchronous SA}

\begin{rlconnection}[title={Q-Learning -- Asynchronous SA}]
\textbf{Update.}
$Q_{n+1}(s_n,a_n) = Q_n(s_n,a_n) + \alpha_n(s_n,a_n)
  [R_{n+1} + \gamma\max_{a'} Q_n(s_{n+1},a') - Q_n(s_n,a_n)]$.

\textbf{SA form.} Nonlinear SA with drift
$h(Q)(s,a) = r(s,a) + \gamma\sum_{s'}P(s'|s,a)\max_{a'}Q(s',a') -
Q(s,a)$. The ODE $\dot{Q} = \mathcal{T}^*Q - Q$ uses the Bellman
optimality operator $\mathcal{T}^*$, a $\gamma$-contraction. The
Lyapunov function $V(Q) = \|Q - Q^*\|_\infty$ gives
$\dot{V} \leq -(1-\gamma)V$, so $Q^*$ is globally stable. Under
Robbins--Monro step sizes per $(s,a)$ and sufficient exploration
(each pair updated infinitely often), $Q_n \cas Q^*$.
\end{rlconnection}

\subsection{REINFORCE as SA}

\begin{rlconnection}[title={REINFORCE -- SA on the Gradient ODE}]
\textbf{Update.}
$\theta_{n+1} = \theta_n + \alpha_n\widehat{\nabla J(\theta_n)}$.

\textbf{SA form.} This is Robbins--Monro with
$h(\theta) = \nabla J(\theta)$ and noise
$M_{n+1} = \widehat{\nabla J(\theta_n)} - \nabla J(\theta_n)$.

\textbf{Associated ODE.} $\dot{\theta} = \nabla J(\theta)$. Any
local maximum $\theta^*$ with negative definite Hessian is a stable
equilibrium. By Proposition~\ref{prop:lock-in}, the iterates converge
to a local maximum, avoiding saddle points and minima.

\textbf{Convergence rate.} The CLT (Theorem~\ref{thm:sa-clt}) gives
$\sqrt{n}(\theta_n - \theta^*) \cdist \mathcal{N}(0,V)$ near each
local maximum. The high variance of REINFORCE ($\Sigma$ is large)
motivates baselines: subtracting a baseline $b(s)$ from the return
does not change $h(\theta)$ but reduces $\Sigma$, shrinking the
asymptotic covariance $V$.
\end{rlconnection}

\subsection{Actor-Critic as Two-Timescale SA}

\begin{rlconnection}[title={Actor-Critic -- Two-Timescale Analysis}]
Critic $w$ (fast, step $b_n$), actor $\theta$ (slow, step $a_n$),
$a_n/b_n \to 0$:
\begin{align*}
  w_{n+1} &= w_n + b_n[\delta_n\phi(s_n)], \quad
    \delta_n = R_{n+1} + \gamma\phi(s_{n+1})^Tw_n - \phi(s_n)^Tw_n,\\
  \theta_{n+1} &= \theta_n
    + a_n[\delta_n\nabla_\theta\log\pi_\theta(a_n|s_n)].
\end{align*}
Map: $y_n = w_n$ (fast), $x_n = \theta_n$ (slow). For fixed
$\theta$, the critic ODE is linear SA for TD under $\pi_\theta$ with
equilibrium $\lambda(\theta) = w^*(\theta)$ (\ref{E5}). The slow ODE
becomes $\dot{\theta} \approx \nabla J(\theta)$ (\ref{E6}). By
Theorem~\ref{thm:two-timescale}, critic converges to $w^*(\theta^*)$
and actor converges to a local optimum $\theta^*$.
\end{rlconnection}

\section{Exercises}
\label{sec:exercises}

\begin{exercise}[Polyak--Ruppert for Mean Estimation (*)]
\label{ex:polyak-ruppert}
Let $X_1, X_2, \ldots$ be i.i.d.\ with $\E[X_i] = \mu$ and
$\Var(X_i) = \sigma^2$. Consider the SA iteration
$\theta_{n+1} = \theta_n + a_n(X_{n+1} - \theta_n)$ with
$a_n = 1/n^{2/3}$.
\begin{enumerate}[label=(\alph*)]
  \item Show this is SA with $h(\theta) = \mu - \theta$ and identify
    $A$ and $\Sigma$.
  \item Define $\bar{\theta}_n = \frac{1}{n}\sum_{k=1}^n \theta_k$.
    Using Theorem~\ref{thm:polyak-ruppert}, find the asymptotic
    distribution of $\sqrt{n}(\bar{\theta}_n - \mu)$.
  \item Compare the asymptotic variance with that of the sample mean
    $\bar{X}_n$. Are they the same?
\end{enumerate}
\end{exercise}

\begin{exercise}[Two-Timescale Separation (**)]
\label{ex:two-timescale-sep}
Consider the system
\begin{align*}
  x_{n+1} &= x_n + a_n(-x_n + y_n + M_{n+1}^{(1)}), \\
  y_{n+1} &= y_n + b_n(-2y_n + x_n + M_{n+1}^{(2)}),
\end{align*}
with $a_n = 1/n$, $b_n = 1/n^{2/3}$, and noises i.i.d.\ with zero
mean and unit variance.
\begin{enumerate}[label=(\alph*)]
  \item Verify $a_n/b_n \to 0$.
  \item Fix $x$ and find $\lambda(x)$ such that
    $-2\lambda(x) + x = 0$.
  \item Substitute $y = \lambda(x)$ into the $x$-ODE to get
    $\dot{x} = -x/2$. Find $x^*$.
  \item Compute $(x^*, y^*)$ and verify Hurwitz conditions for both
    fast and slow ODEs.
\end{enumerate}
\end{exercise}

\begin{exercise}[CLT for Scalar SA (**)]
\label{ex:clt-scalar}
For the scalar SA $x_{n+1} = x_n + (a/n)(-\lambda x_n + M_{n+1})$
with $\lambda > 0$, $\E[M_{n+1}^2] = \sigma^2$:
\begin{enumerate}[label=(\alph*)]
  \item Verify that $x^* = 0$ and $A = -\lambda$.
  \item Find the condition on $a$ for (D3) to hold.
  \item Solve the Lyapunov equation~\eqref{eq:lyapunov} for $V$.
  \item Find the optimal $a$ minimizing $V$ and compute
    $V_{\mathrm{opt}}$.
  \item Compare with Polyak--Ruppert averaging applied to this
    problem.
\end{enumerate}
\end{exercise}

\begin{exercise}[Hurwitz Verification for TD (**)]
\label{ex:hurwitz-td}
Three-state MDP with transition matrix
\[
  P^\pi = \begin{pmatrix}
    0.2 & 0.5 & 0.3 \\ 0.1 & 0.6 & 0.3 \\ 0.4 & 0.3 & 0.3
  \end{pmatrix},
\]
$\gamma = 0.95$, stationary distribution
$d^\pi = (0.2, 0.45, 0.35)$, and feature matrix
$\Phi = \begin{pmatrix} 1 & 0 \\ 0 & 1 \\ 1 & 1 \end{pmatrix}$.
\begin{enumerate}[label=(\alph*)]
  \item Compute $A = \Phi^T D(\gamma P^\pi - I)\Phi$ where
    $D = \diag(d^\pi)$.
  \item Verify $A$ is negative definite.
  \item Conclude TD(0) converges with this function approximation.
\end{enumerate}
\end{exercise}

\begin{exercise}[The Deadly Triad (***)]
\label{ex:deadly-triad}
Construct a two-state MDP with off-policy sampling where TD(0)
with linear FA diverges:
\begin{enumerate}[label=(\alph*)]
  \item Take $\mathcal{S} = \{1,2\}$, $\gamma = 0.99$, behavior
    distribution $\mu(1) = 0.01$, $\mu(2) = 0.99$.
  \item Target policy transitions: $P^\pi(1|1) = 0.9$,
    $P^\pi(2|1) = 0.1$, $P^\pi(1|2) = 0.1$,
    $P^\pi(2|2) = 0.9$.
  \item Features: $\phi(1) = 1$, $\phi(2) = 2$ (scalar).
  \item Compute $A = \E_\mu[\phi(s)(\gamma\phi(s') - \phi(s))]$
    and show it has a positive eigenvalue.
  \item Explain why the SA iteration diverges and relate to the
    Borkar--Meyn stability criterion.
\end{enumerate}
\end{exercise}

\begin{exercise}[GTD2 as Two-Timescale SA (***)]
\label{ex:gtd2}
The GTD2 algorithm uses the updates:
\begin{align*}
  w_{n+1} &= w_n + b_n[\phi_n(\phi_n - \gamma\phi_n')^T\theta_n
    - \phi_n\phi_n^Tw_n], \\
  \theta_{n+1} &= \theta_n
    + a_n[\delta_n\phi_n - \gamma\phi_n'(\phi_n^Tw_n)],
\end{align*}
where $\phi_n = \phi(s_n)$, $\phi_n' = \phi(s_{n+1})$,
$\delta_n = R_{n+1} + \gamma\phi_n'^T\theta_n - \phi_n^T\theta_n$.
\begin{enumerate}[label=(\alph*)]
  \item Identify fast and slow iterates and verify timescale
    separation.
  \item Compute the fast-timescale equilibrium $\lambda(\theta)$.
  \item Substitute into the slow ODE and show it is gradient descent
    on the MSPBE (mean squared projected Bellman error).
  \item Explain why GTD2 avoids the deadly triad.
\end{enumerate}
\end{exercise}

\begin{exercise}[Linear SA with Non-Hurwitz Matrix (*)]
\label{ex:non-hurwitz}
Consider the scalar SA $x_{n+1} = x_n + a_n(\lambda x_n + M_{n+1})$
with $\lambda > 0$ and $a_n = 1/n$.
\begin{enumerate}[label=(\alph*)]
  \item Show that $A = \lambda$ is not Hurwitz and the ODE
    $\dot{x} = \lambda x$ has no stable equilibrium.
  \item Argue that $x_n \to \pm\infty$ a.s.
  \item Relate this to off-policy TD(0) where the SA matrix has a
    positive eigenvalue.
\end{enumerate}
\end{exercise}

\section{Summary}
\label{sec:summary}

\begin{itemize}[leftmargin=2em]
  \item \textbf{CLT for SA:} $\sqrt{n}(x_n - x^*) \cdist
    \mathcal{N}(0,V)$ with $V$ from the Lyapunov equation. Rate:
    $O(1/\sqrt{n})$.
  \item \textbf{Polyak--Ruppert averaging:} $\bar{x}_n$ achieves
    optimal covariance $A^{-1}\Sigma A^{-T}$ without knowing $A$.
  \item \textbf{Two-timescale SA:} Fast iterate tracks
    $\lambda(x_n)$; slow iterate converges via
    $\dot{x} = f(x,\lambda(x))$.
  \item \textbf{Linear SA:} Convergence iff $A$ Hurwitz. Boundedness
    automatic.
  \item \textbf{Lock-in:} SA avoids unstable equilibria.
  \item \textbf{RL:} TD(0) = linear SA (Hurwitz on-policy);
    Q-learning = nonlinear SA (contraction); REINFORCE = gradient
    ascent SA (lock-in); actor-critic = two-timescale SA.
\end{itemize}
