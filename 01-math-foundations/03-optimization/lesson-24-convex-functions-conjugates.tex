\section{Motivation}

This lesson studies convex \emph{functions}---the maps whose local minima are
always global.  We establish first- and second-order characterizations, prove
Jensen's inequality, develop composition rules, and introduce the Fenchel
conjugate.

\begin{rlconnection}
Convex functions arise throughout RL and machine learning:
\begin{itemize}
  \item \textbf{Loss functions.}  Squared error, cross-entropy, and hinge
        loss are convex in the parameters for linear models.
  \item \textbf{First-order condition.}  $\nabla f(x^*) = 0$ implies global
        optimality for convex $f$, which is why convex surrogate losses
        (e.g., the linearized objective in TRPO) can be minimized greedily.
  \item \textbf{Jensen's inequality.}  Foundation of the ELBO in variational
        inference and of importance-sampling variance analysis.
  \item \textbf{Fenchel duality.}  The conjugate of negative entropy is the
        log-sum-exp, underlying soft Q-learning and SAC.
\end{itemize}
\end{rlconnection}

\textbf{Prerequisites.}  Lesson~23 (convex sets, separation theorems).
\textbf{References.}  Boyd and Vandenberghe~\cite{boyd2004}, Ch.~3;
Rockafellar~\cite{rockafellar1970}.

\section{Convex Functions}

\begin{definition}[Convex Function]
\label{def:convex-function}
A function $f : \R^n \to \R \cup \{+\infty\}$ is \emph{convex} if $\dom f$ is
a convex set and for all $x, y \in \dom f$ and $\theta \in [0,1]$:
\[
  f(\theta x + (1-\theta) y) \le \theta f(x) + (1-\theta) f(y).
\]
It is \emph{strictly convex} if strict inequality holds whenever $x \ne y$ and
$\theta \in (0,1)$.  It is \emph{concave} if $-f$ is convex.
\end{definition}

\begin{definition}[Epigraph]
\label{def:epigraph}
The \emph{epigraph} of $f$ is
$\epi f = \{(x, t) \in \R^{n+1} : x \in \dom f,\; f(x) \le t\}$.
\end{definition}

\begin{proposition}[Epigraph Characterization]
\label{prop:epigraph}
$f$ is convex if and only if $\epi f$ is a convex set.
\end{proposition}

\begin{proof}
($\Rightarrow$)  Let $(x_1, t_1), (x_2, t_2) \in \epi f$ and
$\theta \in [0,1]$.  Then $f(x_i) \le t_i$ and
$f(\theta x_1 + (1-\theta) x_2) \le \theta f(x_1) + (1-\theta) f(x_2)
\le \theta t_1 + (1-\theta) t_2$, so the convex combination lies in $\epi f$.

($\Leftarrow$)  Points $(x_i, f(x_i)) \in \epi f$, so their convex
combination $(\theta x_1 + (1-\theta) x_2,\; \theta f(x_1) +
(1-\theta) f(x_2)) \in \epi f$, giving the convexity inequality.
\end{proof}

\begin{proposition}[Line Restriction]
\label{prop:line-restriction}
$f : \R^n \to \R \cup \{+\infty\}$ is convex if and only if for every
$x \in \dom f$ and $v \in \R^n$, the function $g(t) = f(x + tv)$ is convex.
\end{proposition}

\begin{proof}
($\Rightarrow$) $g(\theta t_1 + (1-\theta) t_2) =
f(\theta(x + t_1 v) + (1-\theta)(x + t_2 v)) \le
\theta g(t_1) + (1-\theta) g(t_2)$.

($\Leftarrow$)  Set $v = x_2 - x_1$, $g(t) = f(x_1 + t v)$.  Then
$g(\theta) \le \theta g(1) + (1-\theta) g(0) =
\theta f(x_2) + (1-\theta) f(x_1)$.
\end{proof}

\section{Differentiable Convex Functions}

\begin{theorem}[First-Order Condition for Convexity]
\label{thm:first-order}
Let $f : \R^n \to \R$ be differentiable with convex, open domain.  Then $f$
is convex if and only if for all $x, y \in \dom f$:
\begin{equation}
  \label{eq:first-order}
  f(y) \ge f(x) + \nabla f(x)^T (y - x).
\end{equation}
\end{theorem}

\begin{proof}
($\Rightarrow$)  For $t \in (0,1]$, convexity gives
$f(x + t(y-x)) \le (1-t) f(x) + t f(y)$, so
$f(y) \ge f(x) + \frac{f(x + t(y-x)) - f(x)}{t}$.
Taking $t \to 0^+$ yields $f(y) \ge f(x) + \nabla f(x)^T(y-x)$.

($\Leftarrow$)  Let $z = \theta x_1 + (1-\theta) x_2$.  Apply
\eqref{eq:first-order} with $x = z$:
\begin{align*}
  f(x_1) &\ge f(z) + \nabla f(z)^T(x_1 - z), \\
  f(x_2) &\ge f(z) + \nabla f(z)^T(x_2 - z).
\end{align*}
Multiply by $\theta$ and $(1-\theta)$ respectively and add:
$\theta f(x_1) + (1-\theta) f(x_2) \ge f(z)$, since
$\theta(x_1 - z) + (1-\theta)(x_2 - z) = 0$.
\end{proof}

\begin{keyresult}
The first-order condition says the tangent hyperplane is a \emph{global
underestimator}.  Consequence: if $\nabla f(x^*) = 0$ then $f(y) \ge f(x^*)$
for all $y$, so every stationary point is a global minimizer.
\end{keyresult}

\begin{theorem}[Second-Order Condition for Convexity]
\label{thm:second-order}
Let $f : \R^n \to \R$ be twice continuously differentiable with open convex
domain.  Then $f$ is convex if and only if $\nabla^2 f(x) \succeq 0$ for all
$x \in \dom f$.  If $\nabla^2 f(x) \succ 0$ everywhere, then $f$ is strictly
convex (the converse fails: $f(x) = x^4$ is strictly convex but
$f''(0) = 0$).
\end{theorem}

\begin{proof}
By line restriction, it suffices to prove the univariate case: $g$ twice
differentiable is convex iff $g''(t) \ge 0$ for all $t$.

($\Rightarrow$)  Suppose $g''(t_0) < 0$.  The first-order condition gives
$g(t_0 + h) \ge g(t_0) + g'(t_0) h$.  But Taylor's theorem gives
$g(t_0 + h) = g(t_0) + g'(t_0) h + \frac{1}{2} g''(\xi) h^2$ with
$g''(\xi) < 0$ for small $|h|$ by continuity---contradiction.

($\Leftarrow$)  If $g'' \ge 0$, then Taylor's theorem gives
$g(y) = g(x) + g'(x)(y-x) + \frac{1}{2} g''(\xi)(y-x)^2 \ge
g(x) + g'(x)(y-x)$, which is the first-order condition.

\textbf{Multivariate case.}  $g(t) = f(x + tv)$ has
$g''(t) = v^T \nabla^2 f(x+tv)\, v$.  So $g'' \ge 0$ for all $x, v, t$ iff
$\nabla^2 f \succeq 0$ everywhere.
\end{proof}

\section{Jensen's Inequality}

\begin{theorem}[Jensen's Inequality]
\label{thm:jensen}
Let $f : \R^n \to \R$ be convex and $X$ a random vector with
$\E[\|X\|] < \infty$ and $\E[X] \in \dom f$.  Then
$f(\E[X]) \le \E[f(X)]$.
If $f$ is concave, the inequality reverses.
\end{theorem}

\begin{proof}
\textbf{Differentiable case.}  By the first-order condition at
$\bar{x} = \E[X]$: $f(X) \ge f(\bar{x}) + \nabla f(\bar{x})^T(X - \bar{x})$.
Taking expectations: $\E[f(X)] \ge f(\bar{x}) +
\nabla f(\bar{x})^T(\E[X] - \bar{x}) = f(\E[X])$.

\textbf{General case.}  At any interior point $\bar{x}$ of $\dom f$, a convex
function admits a \emph{subgradient} $g$ satisfying $f(y) \ge f(\bar{x}) +
g^T(y - \bar{x})$ for all $y$ (by the supporting hyperplane theorem applied
to $\epi f$).  The same calculation gives $\E[f(X)] \ge f(\bar{x})$.
\end{proof}

\begin{keyresult}[title={Jensen's Inequality -- Finite Form}]
For convex $f$, points $x_1, \ldots, x_k$, and weights $\theta_i \ge 0$ with
$\sum \theta_i = 1$:
$f\bigl(\sum_{i=1}^k \theta_i x_i\bigr) \le \sum_{i=1}^k \theta_i f(x_i)$.
\end{keyresult}

\section{Operations Preserving Convexity}

\begin{proposition}[Nonnegative Weighted Sum]
\label{prop:nonneg-sum}
If $f_1, \ldots, f_m$ are convex and $w_i \ge 0$, then
$f = \sum_i w_i f_i$ is convex.
\end{proposition}

\begin{proof}
$f(\theta x + (1-\theta)y) = \sum_i w_i f_i(\theta x + (1-\theta)y) \le
\sum_i w_i[\theta f_i(x) + (1-\theta) f_i(y)] =
\theta f(x) + (1-\theta) f(y)$.
\end{proof}

\begin{proposition}[Affine Composition]
\label{prop:affine-comp}
If $f$ is convex, then $g(x) = f(Ax + b)$ is convex.
\end{proposition}

\begin{proof}
$g(\theta x + (1-\theta)y) = f(\theta(Ax+b) + (1-\theta)(Ay+b)) \le
\theta f(Ax+b) + (1-\theta) f(Ay+b) = \theta g(x) + (1-\theta) g(y)$.
\end{proof}

\begin{proposition}[Pointwise Maximum and Supremum]
\label{prop:pointwise-sup}
If $f_1, \ldots, f_m$ are convex, then $f(x) = \max_i f_i(x)$ is convex.
More generally, if $f(x,y)$ is convex in $x$ for each $y \in \mathcal{A}$,
then $g(x) = \sup_{y \in \mathcal{A}} f(x,y)$ is convex.
\end{proposition}

\begin{proof}
For the finite case: $\max_i f_i(\theta x + (1-\theta)y) \le
\max_i[\theta f_i(x) + (1-\theta) f_i(y)] \le
\theta \max_i f_i(x) + (1-\theta) \max_i f_i(y)$.

For the general case: $\epi g = \bigcap_{y} \epi f(\cdot, y)$ is an
intersection of convex sets, hence convex.  So $g$ is convex by the epigraph
characterization.
\end{proof}

\begin{proposition}[Scalar Composition]
\label{prop:scalar-composition}
Let $f = h \circ g$.  Then:
(a) $f$ is convex if $h$ is convex nondecreasing and $g$ is convex;
(b) $f$ is convex if $h$ is convex nonincreasing and $g$ is concave;
(c) $f$ is concave if $h$ is concave nondecreasing and $g$ is concave.
\end{proposition}

\begin{proof}
We prove (a).  Since $g$ is convex:
$g(\theta x + (1-\theta)y) \le \theta g(x) + (1-\theta) g(y)$.
Since $h$ is nondecreasing:
$h(g(\theta x + (1-\theta)y)) \le h(\theta g(x) + (1-\theta) g(y))$.
Since $h$ is convex:
$h(\theta g(x) + (1-\theta) g(y)) \le \theta h(g(x)) + (1-\theta) h(g(y))$.
Combining gives $f(\theta x + (1-\theta)y) \le \theta f(x) + (1-\theta)f(y)$.
Parts (b) and (c) follow by analogous chains with appropriate monotonicity.
\end{proof}

\begin{proposition}[Partial Minimization]
\label{prop:partial-min}
If $f(x,y)$ is jointly convex and $C$ is convex, then
$g(x) = \inf_{y \in C} f(x,y)$ is convex (when $g > -\infty$).
\end{proposition}

\begin{proof}
For $\varepsilon > 0$, pick $y_i \in C$ with $f(x_i, y_i) \le g(x_i) +
\varepsilon$.  Then $g(\theta x_1 + (1-\theta) x_2) \le f(\theta x_1 +
(1-\theta) x_2,\; \theta y_1 + (1-\theta) y_2) \le \theta f(x_1, y_1) +
(1-\theta) f(x_2, y_2) \le \theta g(x_1) + (1-\theta) g(x_2) + \varepsilon$.
\end{proof}

\begin{proposition}[Perspective Transform]
\label{prop:perspective-function}
If $f : \R^n \to \R$ is convex, then $g(x, t) = t\, f(x/t)$ (for $t > 0$)
is convex.
\end{proposition}

\begin{proof}
$(x, t, s) \in \epi g$ iff $(x/t, s/t) \in \epi f$ and $t > 0$, so $\epi g$
is the preimage of $\epi f$ under the perspective map, which preserves
convexity (Lesson~23).
\end{proof}

\section{Conjugate Functions}

\begin{definition}[Fenchel Conjugate]
\label{def:conjugate}
The \emph{Fenchel conjugate} of $f : \R^n \to \R \cup \{+\infty\}$ is
\[
  f^*(y) = \sup_{x \in \dom f} \bigl(y^T x - f(x)\bigr).
\]
\end{definition}

\begin{proposition}[Conjugate is Always Convex]
\label{prop:conjugate-convex}
For any $f$ (not necessarily convex), $f^*$ is closed and convex.
\end{proposition}

\begin{proof}
$f^*(y) = \sup_x (y^T x - f(x))$ is a pointwise supremum of affine
functions, hence convex.  Its epigraph $\epi f^* = \bigcap_x \{(y,t) :
y^T x - f(x) \le t\}$ is an intersection of closed halfspaces, hence closed.
\end{proof}

\begin{example}[Standard Conjugates]
\label{ex:conjugates}
\begin{enumerate}[label=(\alph*)]
  \item \textbf{Affine:} $f(x) = a^T x + b$ gives $f^*(y) = -b$ if $y = a$,
        $+\infty$ otherwise.
  \item \textbf{Quadratic:} $f(x) = \frac{1}{2} x^T Q x$ ($Q \succ 0$)
        gives $f^*(y) = \frac{1}{2} y^T Q^{-1} y$
        (set $\nabla_x(y^T x - \frac{1}{2} x^T Q x) = 0$).
  \item \textbf{Negative entropy:} $f(x) = x \log x$ ($x > 0$) gives
        $f^*(y) = e^{y-1}$ (set $y - \log x - 1 = 0$).
  \item \textbf{Indicator:} $\delta_C(x)$ gives $\delta_C^*(y) =
        \sup_{x \in C} y^T x$, the support function.
\end{enumerate}
\end{example}

\begin{theorem}[Fenchel's Inequality]
\label{thm:fenchel-inequality}
For any $f$ and any $x, y \in \R^n$: $f(x) + f^*(y) \ge y^T x$.
\end{theorem}

\begin{proof}
$f^*(y) = \sup_z(y^T z - f(z)) \ge y^T x - f(x)$.  Rearranging gives the
result.
\end{proof}

\begin{keyresult}[title={Fenchel's Inequality}]
The Young--Fenchel inequality $f(x) + f^*(y) \ge \langle y, x \rangle$
is the pointwise analog of strong duality.  Equality holds iff
$y \in \partial f(x)$ when $f$ is proper closed convex.
\end{keyresult}

\begin{theorem}[Biconjugate Theorem]
\label{thm:biconjugate}
If $f : \R^n \to \R \cup \{+\infty\}$ is proper, closed, and convex, then
$f^{**} = f$.
\end{theorem}

\begin{proof}
\textbf{Step 1: $f^{**} \le f$.}  For each $y$,
$f^*(y) \ge y^T x - f(x)$, so $y^T x - f^*(y) \le f(x)$.  Taking the
supremum over $y$: $f^{**}(x) = \sup_y(x^T y - f^*(y)) \le f(x)$.

\textbf{Step 2: $f^{**} \ge f$.}  Suppose $f^{**}(x_0) < f(x_0)$ for some
$x_0$.  Then $(x_0, f^{**}(x_0)) \notin \epi f$, which is closed convex.
By the separating hyperplane theorem, there exist $(a, b) \ne 0$ and
$\gamma$ with
$a^T x + bt \ge \gamma$ for all $(x,t) \in \epi f$ and
$a^T x_0 + b\, f^{**}(x_0) < \gamma$.

Since $t$ can be arbitrarily large in $\epi f$, we need $b \ge 0$.  If
$b = 0$, then $a^T x \ge \gamma$ for all $x \in \dom f$ while
$a^T x_0 < \gamma$, contradicting $x_0 \in \dom f$.  So $b > 0$.

Setting $y = -a/b$ and $c = \gamma/b$: $f(x) - y^T x \ge c$ for all
$x \in \dom f$, so $f^*(y) \le -c$.  Also
$f^{**}(x_0) < y^T x_0 - c \le y^T x_0 + f^*(y)$.  But
$f^{**}(x_0) \ge x_0^T y - f^*(y)$ by definition---contradiction.
\end{proof}

\begin{rlconnection}
Fenchel conjugates appear in \textbf{regularized MDPs}: if the policy is
entropy-regularized with $\Omega(\pi) = \sum_a \pi(a)\log\pi(a)$, the
regularized Bellman operator involves $\Omega^*$, which is the log-sum-exp
(soft maximum).  This is the foundation of soft Q-learning and SAC.
\end{rlconnection}

\section{Quasiconvexity and Log-Convexity}

\begin{definition}[Sublevel Set]
\label{def:sublevel}
The $\alpha$-\emph{sublevel set} of $f$ is
$C_\alpha = \{x \in \dom f : f(x) \le \alpha\}$.
\end{definition}

\begin{proposition}[Sublevel Sets of Convex Functions]
\label{prop:sublevel}
If $f$ is convex, then every sublevel set $C_\alpha$ is convex.
\end{proposition}

\begin{proof}
For $x, y \in C_\alpha$ and $\theta \in [0,1]$:
$f(\theta x + (1-\theta)y) \le \theta f(x) + (1-\theta) f(y) \le \alpha$.
\end{proof}

\begin{warning}
The converse is \textbf{false}: $f(x) = -e^x$ has convex sublevel sets but is
concave.  Functions with convex sublevel sets are called \emph{quasiconvex}.
\end{warning}

\begin{definition}[Quasiconvex Function]
\label{def:quasiconvex}
$f$ is \emph{quasiconvex} if $\dom f$ is convex and all sublevel sets are
convex.  Equivalently, $f(\theta x + (1-\theta)y) \le \max\{f(x), f(y)\}$
for all $x, y$ and $\theta \in [0,1]$.  $f$ is \emph{quasiconcave} if $-f$ is
quasiconvex.
\end{definition}

\begin{proposition}
\label{prop:quasi-equiv}
$f$ has convex sublevel sets iff
$f(\theta x + (1-\theta)y) \le \max\{f(x), f(y)\}$ for all $x, y, \theta$.
\end{proposition}

\begin{proof}
($\Rightarrow$) Set $\alpha = \max\{f(x), f(y)\}$; then $x, y \in C_\alpha$
(convex), so $\theta x + (1-\theta)y \in C_\alpha$.
($\Leftarrow$) For $x, y \in C_\alpha$: $\max\{f(x), f(y)\} \le \alpha$,
so $f(\theta x + (1-\theta)y) \le \alpha$.
\end{proof}

\begin{definition}[Log-Concave and Log-Convex]
\label{def:log-concave}
A positive function $f$ is \emph{log-concave} if $\log f$ is concave:
$f(\theta x + (1-\theta)y) \ge f(x)^\theta f(y)^{1-\theta}$.
It is \emph{log-convex} if $\log f$ is convex.
\end{definition}

\begin{proposition}[Hierarchy]
\label{prop:hierarchy}
For positive $f$:
$\text{log-convex} \Rightarrow \text{convex} \Rightarrow \text{quasiconvex}$
and
$\text{concave} \Rightarrow \text{log-concave} \Rightarrow
\text{quasiconcave}$.
\end{proposition}

\begin{proof}
\textbf{Log-convex $\Rightarrow$ convex.}  $f = \exp(\log f)$ is the
composition of convex nondecreasing $\exp$ with convex $\log f$, hence convex
by Proposition~\ref{prop:scalar-composition}(a).

\textbf{Convex $\Rightarrow$ quasiconvex.}  Immediate from
Proposition~\ref{prop:sublevel}.

\textbf{Concave $\Rightarrow$ log-concave.}  By weighted AM--GM:
$f(\theta x + (1-\theta)y) \ge \theta f(x) + (1-\theta) f(y) \ge
f(x)^\theta f(y)^{1-\theta}$.

\textbf{Log-concave $\Rightarrow$ quasiconcave.}  Superlevel sets of $f$
equal superlevel sets of the concave function $\log f$, hence are convex.
\end{proof}

\begin{example}[Log-Concave Densities]
\label{ex:log-concave-densities}
Gaussian, exponential, uniform, and Laplace densities are log-concave.
Products of log-concave functions are log-concave ($\log(fg) = \log f +
\log g$).  By Prekopa's theorem, marginals of log-concave functions are
log-concave---important for Bayesian posterior analysis.
\end{example}

\section{Exercises}

\begin{exercise}[Epigraph and Convexity]
\label{exer:epigraph}
Show that $f(x) = \max\{|x_1|, \ldots, |x_n|\}$ (the $\ell_\infty$-norm) is
convex by expressing $\epi f$ as an intersection of halfspaces.
\end{exercise}

\begin{exercise}[First-Order Condition Consequences]
\label{exer:first-order}
Let $f : \R^n \to \R$ be convex and differentiable.
\begin{enumerate}[label=(\alph*)]
  \item Show that $\nabla f(x^*) = 0$ implies $x^*$ is a global minimizer.
  \item Show that if $f$ is strictly convex, the minimizer is unique.
\end{enumerate}
\end{exercise}

\begin{exercise}[Jensen Applications]
\label{exer:jensen}
Using Jensen's inequality, prove:
\begin{enumerate}[label=(\alph*)]
  \item $\E[X^2] \ge (\E[X])^2$ for any $X$ with finite second moment.
  \item $D_{\mathrm{KL}}(p \| q) \ge 0$ for distributions $p, q$ on a finite
        set.
  \item Weighted AM--GM: for $w_i > 0$, $\sum w_i = 1$, $a_i > 0$:
        $\sum w_i a_i \ge \prod a_i^{w_i}$.
\end{enumerate}
\end{exercise}

\begin{exercise}[Conjugate Computations]
\label{exer:conjugates}
Compute the Fenchel conjugate of:
\begin{enumerate}[label=(\alph*)]
  \item $f(x) = \|x\|_1$.
  \item $f(x) = -\log x$ for $x > 0$.
  \item $f(x) = \frac{1}{p}\|x\|_p^p$ for $p > 1$
        (answer: $f^*(y) = \frac{1}{q}\|y\|_q^q$, $\frac{1}{p}+\frac{1}{q}=1$).
\end{enumerate}
\end{exercise}

\begin{exercise}[Biconjugate Verification]
\label{exer:biconjugate}
For $f(x) = |x|$ on $\R$: (a) compute $f^*(y)$; (b) compute $f^{**}(x)$ and
verify $f^{**} = f$.
\end{exercise}

\begin{exercise}[Quasiconvex but Not Convex]
\label{exer:quasiconvex}
Show $f(x) = \sqrt{|x|}$ is quasiconvex but not convex on $\R$.
\end{exercise}

\begin{exercise}[Perspective and KL Divergence]
\label{exer:perspective-kl}
(a) Show the perspective of $f(u) = u\log u$ is $g(u,t) = u\log(u/t)$.
(b) Conclude $D_{\mathrm{KL}}(p\|q) = \sum_i p_i \log(p_i/q_i)$ is jointly
convex in $(p,q)$.
\end{exercise}

\begin{exercise}[Log-Sum-Exp as Conjugate]
\label{exer:lse-conjugate}
Show $\log(\sum_i e^{y_i}) = \sup_{x \in \Delta_n}(y^T x +
\sum_i x_i \log x_i)$ where $\Delta_n$ is the probability simplex.
\textit{Hint:} Use Lagrange multipliers.
\end{exercise}
