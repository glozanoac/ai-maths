\section{Motivation and Overview}
\label{sec:motivation}

Reinforcement learning algorithms optimize under uncertainty: an agent
seeks an optimal policy when the environment dynamics are unknown and the
only information arrives through noisy, sequential interactions. The
mathematical framework that captures this setting is \emph{stochastic
approximation} (SA) --- finding the root (or optimum) of a function
that can only be observed through noisy samples.

The field was born with the seminal 1951 paper of \citet{robbins1951},
who proposed an iterative scheme for finding the root of a regression
function from noisy observations. The theoretical foundations were
deepened by Kushner and Clark, Ljung, and \citet{borkar2008}, who
established the powerful \emph{ODE method}: under appropriate conditions,
a stochastic approximation algorithm converges to the same limit as the
solution of an associated ordinary differential equation.

\begin{rlconnection}[title={SA as the Engine of RL}]
Every major RL update rule is a stochastic approximation iteration:
\begin{itemize}[nosep]
  \item \textbf{TD(0):}
    $V(s_n) \leftarrow V(s_n) + \alpha_n[R_{n+1} + \gamma V(s_{n+1}) - V(s_n)]$
  \item \textbf{Q-learning:}
    $Q(s_n,a_n) \leftarrow Q(s_n,a_n) + \alpha_n[R_{n+1} + \gamma \max_{a'} Q(s_{n+1},a') - Q(s_n,a_n)]$
  \item \textbf{REINFORCE:}
    $\theta_{n+1} = \theta_n + \alpha_n \widehat{\nabla J(\theta_n)}$
  \item \textbf{Actor-Critic:} critic and actor update at
    \emph{different rates} --- two-timescale SA (Lesson 32).
\end{itemize}
The convergence theorems in this lesson provide the theoretical
foundation for understanding \emph{when} and \emph{why} these
algorithms converge.
\end{rlconnection}

\medskip
\noindent\textbf{Prerequisites.} Probability spaces, conditional
expectation, martingale difference sequences (Lesson 8), the law of
large numbers, and basic ODE theory (existence, uniqueness, Lyapunov
stability).

\section{The Robbins--Monro Algorithm}
\label{sec:robbins-monro}

\subsection{Problem Formulation}

Suppose we wish to find a root of a function $h : \R^d \to \R^d$,
i.e., a point $x^* \in \R^d$ such that $h(x^*) = 0$. The difficulty
is that we cannot evaluate $h$ directly. Instead, for each query point
$x$, we observe a noisy sample
\[
  H(x, Y) \quad \text{where} \quad h(x) = \E[H(x, Y)]
\]
and $Y$ is a random element representing the noise source.

\begin{definition}[Step-Size Sequence (Robbins--Monro Conditions)]
\label{def:step-size}
A \emph{step-size sequence} (or \emph{learning rate schedule}) is a
deterministic sequence $\{a_n\}_{n \geq 0}$ of positive real numbers
satisfying the \emph{Robbins--Monro conditions}:
\begin{enumerate}[label=(\roman*)]
  \item $\displaystyle\sum_{n=0}^{\infty} a_n = \infty$
    \quad (steps are large enough to reach any target),
  \item $\displaystyle\sum_{n=0}^{\infty} a_n^2 < \infty$
    \quad (steps are small enough that noise averages out).
\end{enumerate}
\end{definition}

\begin{remark}
Condition (i) ensures the iterates can travel an unbounded total
distance. Condition (ii) controls accumulated noise variance (proportional
to $\sum a_n^2 \Var(\text{noise})$). The canonical choice is
$a_n = c/n$; more generally, $a_n = c/n^\alpha$ works for
$\alpha \in (1/2, 1]$.
\end{remark}

\subsection{The Iteration}

\begin{definition}[Robbins--Monro Iteration]
\label{def:rm-iteration}
Given an initial point $x_0 \in \R^d$, the \emph{Robbins--Monro
iteration} is
\begin{equation}
\label{eq:rm}
  x_{n+1} = x_n + a_n H(x_n, Y_n), \quad n \geq 0,
\end{equation}
where $\{a_n\}$ satisfies Definition~\ref{def:step-size} and
$Y_0, Y_1, \ldots$ are noise samples.
\end{definition}

We decompose the update into ``signal'' and ``noise.'' Define the
\emph{martingale difference noise}
\begin{equation}
\label{eq:mds-noise}
  M_{n+1} \coloneqq H(x_n, Y_n) - h(x_n).
\end{equation}
Then the iteration becomes
\begin{equation}
\label{eq:rm-decomposed}
  x_{n+1} = x_n + a_n\bigl[h(x_n) + M_{n+1}\bigr].
\end{equation}
This is the canonical form we use throughout: the iterate moves in the
direction $h(x_n)$ (the ``drift'') plus additive noise $M_{n+1}$.

\subsection{Standing Assumptions}

\begin{definition}[Standing Assumptions for Robbins--Monro]
\label{def:rm-assumptions}
Let $\mathcal{F}_n = \sigma(x_0, Y_0, \ldots, Y_{n-1})$ be the
natural filtration. We assume:
\begin{enumerate}[label=(A\arabic*)]
  \item \label{A1} \textbf{Step sizes:} $\{a_n\}$ satisfies
    Definition~\ref{def:step-size}.
  \item \label{A2} \textbf{Martingale difference noise:}
    $\E[M_{n+1} \mid \mathcal{F}_n] = 0$ a.s.\ for all $n$.
  \item \label{A3} \textbf{Bounded noise variance:} There exists
    $K > 0$ such that
    $\E[\|M_{n+1}\|^2 \mid \mathcal{F}_n] \leq K(1 + \|x_n\|^2)$ a.s.
  \item \label{A4} \textbf{Stability (monotonicity):} There exists a
    unique root $x^* \in \R^d$ with $h(x^*) = 0$, and
    $\langle x - x^*, h(x) \rangle < 0$ for all $x \neq x^*$.
  \item \label{A5} \textbf{Linear growth:}
    $\|h(x)\| \leq K(1 + \|x\|)$ for all $x$.
\end{enumerate}
\end{definition}

\begin{remark}
Assumption \ref{A4} requires that $h$ points ``inward'' toward $x^*$.
Geometrically, the angle between $x - x^*$ and $h(x)$ is always
obtuse. This is satisfied when $h = -\nabla f$ for a strictly convex
$f$ with minimum at $x^*$. For non-convex problems (e.g., policy
gradient in RL), this condition fails, motivating the ODE method of
Section~\ref{sec:ode-method}.
\end{remark}

\subsection{Robbins--Siegmund Theorem}

The convergence proof relies on a classical supermartingale result.

\begin{theorem}[Robbins--Siegmund]
\label{thm:robbins-siegmund}
Let $\{V_n\}$, $\{\alpha_n\}$, $\{\beta_n\}$, $\{u_n\}$ be
nonnegative $\mathcal{F}_n$-adapted sequences satisfying
\[
  \E[V_{n+1} \mid \mathcal{F}_n]
    \leq (1 + \alpha_n) V_n - \beta_n + u_n
\]
with $\sum \alpha_n < \infty$ and $\sum u_n < \infty$ a.s. Then:
\begin{enumerate}[label=(\roman*)]
  \item $V_n$ converges a.s.\ to a finite limit.
  \item $\sum \beta_n < \infty$ a.s.
\end{enumerate}
\end{theorem}

\begin{proof}
Define $W_n = V_n \prod_{k=0}^{n-1}(1+\alpha_k)^{-1}$. Since
$\sum \alpha_k < \infty$, the product $P_n = \prod_{k=0}^{n-1}
(1+\alpha_k)$ converges to a finite limit, so $1/P_n$ is bounded
away from zero. Dividing the hypothesis by $P_{n+1}$ and
rearranging shows that $Z_n = W_n + \sum_{k=n}^{\infty}
u_k/P_k$ is a nonnegative supermartingale. By the supermartingale
convergence theorem, $Z_n$ converges a.s., hence $W_n$ and $V_n$
converge a.s. Summing the inequality yields
$\sum \beta_n/P_n < \infty$, and since $1/P_n$ is bounded below,
$\sum \beta_n < \infty$ a.s.
\end{proof}

\subsection{Convergence Theorem}

\begin{theorem}[Robbins--Monro Convergence]
\label{thm:rm-convergence}
Under assumptions \ref{A1}--\ref{A5}, the Robbins--Monro
iteration~\eqref{eq:rm} satisfies $x_n \cas x^*$.
\end{theorem}

\begin{proof}
Define $e_n = x_n - x^*$. Expanding $\|e_{n+1}\|^2$ and taking
conditional expectation (using \ref{A2}):
\[
  \E[\|e_{n+1}\|^2 \mid \mathcal{F}_n]
    = \|e_n\|^2 + 2a_n\langle e_n, h(x_n)\rangle
      + a_n^2\E[\|h(x_n) + M_{n+1}\|^2 \mid \mathcal{F}_n].
\]
Using $\|a+b\|^2 \leq 2\|a\|^2 + 2\|b\|^2$, \ref{A3}, \ref{A5},
and $\|x_n\|^2 \leq 2\|e_n\|^2 + 2\|x^*\|^2$, we get
\[
  \E[\|e_{n+1}\|^2 \mid \mathcal{F}_n]
    \leq (1 + Ca_n^2)\|e_n\|^2 + 2a_n\langle e_n, h(x_n)\rangle + Ca_n^2
\]
for some $C > 0$. Set $V_n = \|e_n\|^2$, $\alpha_n = Ca_n^2$,
$\beta_n = -2a_n\langle e_n, h(x_n)\rangle$, $u_n = Ca_n^2$. By
\ref{A4}, $\beta_n > 0$ when $x_n \neq x^*$. Since $\sum a_n^2 < \infty$,
the Robbins--Siegmund theorem gives: (i)~$V_n$ converges a.s.\ to
some $L \geq 0$, and (ii)~$\sum \beta_n < \infty$ a.s.

If $L > 0$ with positive probability, then $\|e_n\| \to \sqrt{L} > 0$,
so by \ref{A4} there exists $\delta > 0$ with $\beta_n \geq
2\delta a_n$ eventually, giving $\sum \beta_n \geq 2\delta\sum a_n
= \infty$, contradicting (ii). Hence $L = 0$ a.s.
\end{proof}

\begin{keyresult}[title={Robbins--Monro Convergence}]
Under a monotonicity condition on $h$ (the drift points toward the
root $x^*$), martingale difference noise with bounded conditional
variance, and the step-size conditions $\sum a_n = \infty$,
$\sum a_n^2 < \infty$, the iteration
$x_{n+1} = x_n + a_n[h(x_n) + M_{n+1}]$ converges almost surely
to $x^*$.
\end{keyresult}

\begin{rlconnection}[title={SGD as Robbins--Monro}]
Stochastic gradient descent for minimizing
$f(\theta) = \E_Z[\ell(\theta, Z)]$ is the Robbins--Monro iteration
with $h(\theta) = -\nabla f(\theta)$ and
$H(\theta, Z) = -\nabla_\theta \ell(\theta, Z)$. Under strict
convexity of $f$, the monotonicity condition \ref{A4} holds for
$-\nabla f$, so Theorem~\ref{thm:rm-convergence} guarantees
$\theta_n \cas \theta^*$.
\end{rlconnection}

\section{The ODE Method}
\label{sec:ode-method}

The Robbins--Monro analysis requires the strong monotonicity condition
\ref{A4}, which fails for many RL algorithms (e.g., policy gradient
on non-concave objectives). The \emph{ODE method}, developed by Ljung
(1977), Kushner and Clark (1978), and systematized by
\citet{borkar2008}, replaces monotonicity with a more natural
condition: convergence of the SA iteration follows from the stability
of an associated ordinary differential equation.

\subsection{The Associated ODE}

\begin{definition}[Associated ODE]
\label{def:associated-ode}
Given the SA iteration~\eqref{eq:rm-decomposed}, the \emph{associated
ODE} is
\begin{equation}
\label{eq:ode}
  \dot{x}(t) = h(x(t)), \quad x(0) = x_0.
\end{equation}
\end{definition}

The key insight is that for small step sizes, the SA iteration
$x_{n+1} \approx x_n + a_n h(x_n)$ behaves like an Euler
discretization of the ODE~\eqref{eq:ode} with time step $a_n$,
plus noise that averages out.

\subsection{Interpolated Trajectory}

\begin{definition}[Interpolated Trajectory]
\label{def:interpolated-traj}
Define the time instants $t_0 = 0$ and
$t_n = \sum_{k=0}^{n-1} a_k$ for $n \geq 1$. Note that $t_n \to
\infty$ by condition (i) of Definition~\ref{def:step-size}. The
\emph{interpolated trajectory} $\bar{x} : [0,\infty) \to \R^d$ is
the piecewise constant function
\[
  \bar{x}(t) = x_n \quad \text{for } t \in [t_n, t_{n+1}).
\]
\end{definition}

\subsection{ODE Method Assumptions}

\begin{definition}[ODE Method Assumptions]
\label{def:ode-assumptions}
We modify the standing assumptions as follows:
\begin{enumerate}[label=(B\arabic*)]
  \item \label{B1} \textbf{Step sizes:} As in \ref{A1}.
  \item \label{B2} \textbf{Lipschitz drift:} $h : \R^d \to \R^d$ is
    Lipschitz continuous:
    $\|h(x) - h(y)\| \leq L\|x-y\|$ for all $x,y$ and some $L > 0$.
  \item \label{B3} \textbf{Martingale difference noise:}
    $\E[M_{n+1} \mid \mathcal{F}_n] = 0$ a.s.\ and
    $\E[\|M_{n+1}\|^2 \mid \mathcal{F}_n] \leq K(1+\|x_n\|^2)$ a.s.
  \item \label{B4} \textbf{Stability (boundedness):}
    $\sup_n \|x_n\| < \infty$ a.s.
  \item \label{B5} \textbf{Asymptotic stability of the ODE:} The
    ODE~\eqref{eq:ode} has a globally asymptotically stable
    equilibrium $x^*$ (i.e., $h(x^*) = 0$ and $x(t) \to x^*$ as
    $t \to \infty$ for every initial condition).
\end{enumerate}
\end{definition}

\begin{remark}[The Boundedness Assumption \ref{B4}]
\label{rem:stability}
Assumption \ref{B4} is the most delicate. In practice, one either
(a) proves it from properties of $h$ via the Borkar--Meyn theorem
(Section~\ref{sec:borkar-meyn}), (b) uses projection to a compact
set, or (c) verifies it empirically. For many RL algorithms,
boundedness follows from the structure of the MDP (e.g., bounded
rewards and $\gamma < 1$).
\end{remark}

\subsection{The Noise Vanishes Asymptotically}

\begin{lemma}[Noise Term Vanishes]
\label{lem:noise-vanishes}
Under \ref{B1}, \ref{B3}, and \ref{B4}, the martingale
$Z_n = \sum_{k=0}^{n-1} a_k M_{k+1}$ converges a.s. In particular,
for any $T > 0$,
\[
  \sup_{t_n \leq s \leq t_n + T}
    \Bigl\|\sum_{k=n}^{m(s)-1} a_k M_{k+1}\Bigr\|
    \to 0 \quad \text{a.s.\ as } n \to \infty,
\]
where $m(s) = \min\{j : t_j > s\}$.
\end{lemma}

\begin{proof}
$Z_n$ is a martingale with
$\E[\|Z_n\|^2] = \sum_{k=0}^{n-1} a_k^2 \E[\|M_{k+1}\|^2]
\leq K(1+C)\sum a_k^2 < \infty$
(using \ref{B4} to bound $\E[\|x_k\|^2] \leq C$). By the martingale
convergence theorem, $Z_n$ converges a.s. The tail sum
$Z_{m(s)} - Z_n \to 0$ uniformly over $s \in [t_n, t_n+T]$ since
convergent series have vanishing tails.
\end{proof}

\subsection{Convergence via the ODE Method}

\begin{theorem}[Convergence via the ODE Method]
\label{thm:ode-convergence}
Under assumptions \ref{B1}--\ref{B5}, the SA iteration
$x_{n+1} = x_n + a_n[h(x_n) + M_{n+1}]$ satisfies
$x_n \cas x^*$.
\end{theorem}

\begin{proof}
The proof proceeds in three steps.

\medskip
\noindent\textbf{Step 1: Shifted interpolated trajectories.}
For each $n \geq 0$, define the shifted interpolated trajectory
$x^{(n)} : [0,\infty) \to \R^d$ by
\[
  x^{(n)}(0) = x_n, \quad
  x^{(n)}(t) = x_{n+k} \;\text{ for }\;
    t \in [t_{n+k} - t_n,\, t_{n+k+1} - t_n),
  \quad k \geq 0.
\]

\medskip
\noindent\textbf{Step 2: Convergence to ODE solutions.}
By \ref{B4} and \ref{B2}, both $\|x_n\|$ and $\|h(x_n)\|$ are
bounded a.s. The trajectory increments satisfy
$\|x^{(n)}(t) - x^{(n)}(s)\| \leq C|t-s| + \varepsilon_n$ where
$\varepsilon_n \to 0$ by Lemma~\ref{lem:noise-vanishes}, giving
equicontinuity on $[0,T]$. By the Arzela--Ascoli theorem, every
subsequence has a uniformly convergent further subsequence with
limit $x^{(\infty)}$. The SA iteration gives
\[
  x^{(n)}(t) = x^{(n)}(0)
    + \int_0^t h(x^{(n)}(s))\,ds + \varepsilon_n(t),
\]
where $\varepsilon_n(t) \to 0$ (discretization error vanishes since
$a_n \to 0$; noise vanishes by Lemma~\ref{lem:noise-vanishes}).
Passing to the limit via Lipschitz continuity of $h$ shows
$x^{(\infty)}$ solves the ODE.

\medskip
\noindent\textbf{Step 3: Convergence to the equilibrium.}
By \ref{B5}, every ODE solution converges to $x^*$. The set of limit
points of $\{x_n\}$ is nonempty (by \ref{B4}), compact, connected,
and ODE-flow-invariant (by Step 2). The only such set attracted to
$x^*$ under global asymptotic stability is $\{x^*\}$ itself: if
$\bar{x} \neq x^*$ were a limit point, the ODE trajectory from
$\bar{x}$ would enter any neighborhood of $x^*$, and Lyapunov
stability would prevent escape, making $x^*$ the unique limit point.
\end{proof}

\begin{keyresult}[title={The ODE Method}]
A stochastic approximation iteration
$x_{n+1} = x_n + a_n[h(x_n) + M_{n+1}]$ with decreasing step
sizes converges to the set of asymptotically stable equilibria of
the ODE $\dot{x} = h(x)$, provided the iterates remain bounded
and the noise is a martingale difference sequence. This is the
primary tool for analyzing RL algorithm convergence.
\end{keyresult}

\begin{rlconnection}[title={TD(0) via the ODE Method}]
For TD(0) in tabular form, the mean field is
$h(V) = r^\pi + \gamma P^\pi V - V = (\gamma P^\pi - I)V + r^\pi$.
The associated ODE $\dot{V} = (\gamma P^\pi - I)V + r^\pi$ is a
linear system. The matrix $A = \gamma P^\pi - I$ has all eigenvalues
with real part at most $\gamma - 1 < 0$ (since $P^\pi$ has spectral
radius 1), so $A$ is Hurwitz. The unique equilibrium is
$V^\pi = (I - \gamma P^\pi)^{-1} r^\pi$, the true value function,
and it is globally asymptotically stable. By
Theorem~\ref{thm:ode-convergence}, TD(0) converges a.s.\ to
$V^\pi$.
\end{rlconnection}

\section{The Borkar--Meyn Stability Theorem}
\label{sec:borkar-meyn}

Assumption \ref{B4} (boundedness of iterates) is often the hardest
to verify. The Borkar--Meyn theorem \citep{borkar2000} provides a
systematic way to establish it by examining the behavior of $h$ at
infinity.

\begin{definition}[Borkar--Meyn Assumptions]
\label{def:borkar-meyn-assumptions}
In addition to \ref{B1}--\ref{B3}, assume:
\begin{enumerate}[label=(C\arabic*)]
  \item \label{C1} $h$ is Lipschitz and the \emph{scaled limit}
    $h_\infty(x) \coloneqq \lim_{r \to \infty} h(rx)/r$ exists
    uniformly on compact sets.
  \item \label{C2} The ODE $\dot{x} = h_\infty(x)$ has the origin
    as its unique globally asymptotically stable equilibrium.
\end{enumerate}
\end{definition}

\begin{remark}
For an affine function $h(x) = Ax + b$, the scaled limit is
$h_\infty(x) = \lim_{r \to \infty}(A(rx) + b)/r = Ax$. So \ref{C2}
reduces to requiring that $A$ is Hurwitz. For nonlinear functions
that behave linearly at infinity, $h_\infty$ captures the dominant
asymptotic behavior.
\end{remark}

\begin{theorem}[Borkar--Meyn Stability]
\label{thm:borkar-meyn}
Under \ref{B1}--\ref{B3} and \ref{C1}--\ref{C2}, the iterates
$\{x_n\}$ are bounded almost surely:
$\sup_n \|x_n\| < \infty$ a.s.
\end{theorem}

\begin{proof}
Suppose for contradiction $\|x_n\| \to \infty$ along a subsequence
with positive probability. Let $r_n = \max(\|x_0\|, \ldots, \|x_n\|)$
and $\hat{x}_n = x_n/r_n$. Dividing the SA iteration by $r_{n+1}$
and using \ref{C1} ($h(r_n\hat{x}_n)/r_n \to h_\infty(\hat{x}_n)$),
the rescaled trajectory converges along subsequences to solutions of
$\dot{x} = h_\infty(x)$ (by the same interpolation argument as in
Theorem~\ref{thm:ode-convergence}). But $\|\hat{x}_n\| \leq 1$ and
$\|\hat{x}_n\| = 1$ infinitely often, so the limiting trajectory
is confined to the unit ball and visits its boundary. By \ref{C2},
every solution of $\dot{x} = h_\infty(x)$ converges to the origin,
giving a contradiction.
\end{proof}

\begin{keyresult}[title={Borkar--Meyn Stability Criterion}]
To verify iterate boundedness for SA, examine the ``behavior of $h$
at infinity'': compute $h_\infty(x) = \lim_{r \to \infty} h(rx)/r$
and check that $\dot{x} = h_\infty(x)$ has a stable origin. For
linear SA ($h(x) = Ax + b$), this reduces to checking that $A$ is
Hurwitz.
\end{keyresult}

\begin{rlconnection}[title={Step-Size Schedules in Practice}]
Common RL step-size strategies and their SA interpretation:
\begin{itemize}[nosep]
  \item \textbf{Polynomial decay:} $a_n = c/(n+d)^\alpha$,
    $\alpha \in (1/2,1]$. Satisfies both conditions.
  \item \textbf{Constant step size:} $a_n = \alpha$. Violates
    $\sum a_n^2 < \infty$; iterates fluctuate in an $O(\alpha)$
    neighborhood of $x^*$. Preferred for non-stationary environments.
  \item \textbf{Warmup + decay:} Used in PPO, SAC. No exact
    convergence guarantee but good empirical performance.
\end{itemize}
The Borkar--Meyn theorem explains why constant-step-size RL remains
bounded: if $h_\infty$ is stable, iterates cannot escape to infinity.
\end{rlconnection}

\section{Stochastic Root-Finding}
\label{sec:root-finding}

\begin{example}[Mean Estimation as SA]
\label{ex:mean-estimation}
Let $X_1, X_2, \ldots$ be i.i.d.\ with $\E[X_i] = \mu$ and
$\Var(X_i) = \sigma^2$. The iteration
$\theta_{n+1} = \theta_n + a_n(X_{n+1} - \theta_n)$ has
$h(\theta) = \mu - \theta$, $x^* = \mu$, and noise
$M_{n+1} = X_{n+1} - \mu$. Monotonicity \ref{A4} holds:
$(\theta - \mu)(\mu - \theta) = -(\theta-\mu)^2 < 0$. The
associated ODE $\dot{\theta} = \mu - \theta$ has solution
$\theta(t) = \mu + (\theta_0 - \mu)e^{-t}$ and Jacobian $A = -1$
(Hurwitz).
\end{example}

\begin{example}[Quantile Estimation]
\label{ex:quantile}
Let $X_1, X_2, \ldots$ be i.i.d.\ with continuous strictly
increasing CDF $F$. To find the $p$-quantile $q_p = F^{-1}(p)$:
\[
  \theta_{n+1} = \theta_n + a_n(p - \indicator(X_{n+1} \leq \theta_n)).
\]
Here $h(\theta) = p - F(\theta)$, $x^* = q_p$, and the noise
$M_{n+1} = F(\theta_n) - \indicator(X_{n+1} \leq \theta_n)$ has
conditional mean zero. The monotonicity condition \ref{A4} holds
because $F$ is increasing: if $\theta > q_p$ then $F(\theta) > p$
so $(\theta - q_p)(p - F(\theta)) < 0$, and vice versa.
\end{example}

\begin{example}[Q-learning Update as SA]
\label{ex:q-learning-sa}
In Q-learning, $h(Q)(s,a) = r(s,a) + \gamma\sum_{s'}P(s'|s,a)
\max_{a'}Q(s',a') - Q(s,a)$. This is \emph{nonlinear} SA. The
associated ODE $\dot{Q} = \mathcal{T}^*Q - Q$ involves the Bellman
optimality operator, a $\gamma$-contraction in the sup-norm. The
Lyapunov function $V(Q) = \|Q - Q^*\|_\infty$ satisfies
$\dot{V} \leq -(1-\gamma)V$, so $Q^*$ is globally asymptotically
stable. By the ODE method, Q-learning converges a.s.\ to $Q^*$.
\end{example}

\section{Exercises}
\label{sec:exercises}

\begin{exercise}[Step-Size Conditions (*)]
\label{ex:step-sizes}
For each of the following, determine whether both Robbins--Monro
conditions hold:
\begin{enumerate}[label=(\alph*)]
  \item $a_n = 1/n$
  \item $a_n = 1/n^2$
  \item $a_n = 1/\sqrt{n}$
  \item $a_n = c/(n+d)$ for constants $c,d > 0$
  \item $a_n = 1/(n \log n)$ for $n \geq 2$
  \item $a_n = 1/n^{3/4}$
\end{enumerate}
For each case that fails, explain which condition fails and the
practical consequence for an SA algorithm.
\end{exercise}

\begin{exercise}[Scalar Robbins--Monro (**)]
\label{ex:scalar-rm}
Consider $x_{n+1} = x_n + a_n(c - x_n + \xi_{n+1})$ where
$c \in \R$ and $\{\xi_n\}$ are i.i.d.\ with $\E[\xi_n] = 0$,
$\E[\xi_n^2] = \sigma^2$.
\begin{enumerate}[label=(\alph*)]
  \item Identify $h(x)$ and $x^*$.
  \item Verify the monotonicity condition \ref{A4}.
  \item Compute the Jacobian $A = h'(x^*)$ and verify $A$ is
    Hurwitz.
  \item Write the associated ODE and solve it explicitly.
  \item Show that $\|x_n - x^*\|^2$ satisfies a recursion compatible
    with the Robbins--Siegmund theorem.
\end{enumerate}
\end{exercise}

\begin{exercise}[ODE for Q-Learning (**)]
\label{ex:ode-qlearning}
Consider a two-state, two-action MDP with
$\mathcal{S} = \{1,2\}$, $\mathcal{A} = \{L,R\}$, $\gamma = 0.9$,
deterministic transitions $P(2|1,L) = 1$, $P(1|1,R) = 1$,
$P(1|2,L) = 1$, $P(2|2,R) = 1$, and rewards $r(1,L) = 1$,
$r(1,R) = 0$, $r(2,L) = 0$, $r(2,R) = 2$.
\begin{enumerate}[label=(\alph*)]
  \item Compute $Q^*(s,a)$ for all $(s,a)$ by solving the Bellman
    optimality equation.
  \item Write the associated ODE $\dot{Q} = h(Q)$ as a
    4-dimensional system.
  \item Verify that $Q^*$ is the unique equilibrium.
\end{enumerate}
\end{exercise}

\begin{exercise}[Lyapunov Function for Contraction (**)]
\label{ex:lyapunov-contraction}
Let $T : \R^d \to \R^d$ be a $\gamma$-contraction in the sup-norm
with fixed point $x^*$. Consider the ODE $\dot{x} = Tx - x$.
\begin{enumerate}[label=(\alph*)]
  \item Show that $V(x) = \|x - x^*\|_\infty$ satisfies
    $\dot{V} \leq -(1-\gamma)V$.
  \item Conclude that $x^*$ is globally asymptotically stable.
  \item Apply this to $T = \mathcal{T}^*$ (Bellman optimality
    operator) to verify the ODE analysis for Q-learning.
\end{enumerate}
\end{exercise}

\begin{exercise}[Borkar--Meyn for Linear SA (**)]
\label{ex:borkar-meyn-linear}
For linear SA with $h(x) = Ax + b$:
\begin{enumerate}[label=(\alph*)]
  \item Compute $h_\infty(x)$ and state the Borkar--Meyn condition
    \ref{C2} in terms of the eigenvalues of $A$.
  \item Verify that the condition is equivalent to $A$ being Hurwitz.
  \item Let
    $A = \begin{pmatrix} -2 & 1 \\ 0 & -3 \end{pmatrix}$
    and $b = \begin{pmatrix} 1 \\ 2 \end{pmatrix}$.
    Find the equilibrium $x^* = -A^{-1}b$ and verify boundedness.
\end{enumerate}
\end{exercise}

\begin{exercise}[SA with Markovian Noise (***)]
\label{ex:markov-noise}
In RL, noise is Markovian (from a Markov chain), not i.i.d. Suppose
$\{Y_n\}$ is ergodic with stationary distribution $\mu$,
$h(x) = \E_\mu[H(x,Y)]$, and the Poisson equation
$g(x,y) - Pg(x,y) = H(x,y) - h(x)$ has a solution $g$.
Show how to decompose $H(x_n,Y_n) - h(x_n)$ into a martingale
difference plus a telescoping term using $g$, recovering the ODE
method assumptions.
\end{exercise}

\section{Summary}
\label{sec:summary}

\begin{itemize}[leftmargin=2em]
  \item \textbf{Robbins--Monro conditions:} $\sum a_n = \infty$ and
    $\sum a_n^2 < \infty$. Canonical choice: $a_n = c/n$.
  \item \textbf{Robbins--Monro convergence:} Under monotonicity of
    $h$, bounded noise variance, and RM step sizes, $x_n \cas x^*$.
  \item \textbf{ODE method:} SA converges to stable equilibria of
    $\dot{x} = h(x)$, given bounded iterates and Lipschitz $h$.
    The primary tool for RL convergence proofs.
  \item \textbf{Borkar--Meyn:} Boundedness holds if
    $h_\infty(x) = \lim_{r \to \infty} h(rx)/r$ has a stable origin.
    For linear SA, reduces to $A$ Hurwitz.
  \item \textbf{RL:} TD(0) is linear SA with Hurwitz $A = \gamma P^\pi - I$.
    Q-learning is nonlinear SA with contraction-based stability.
\end{itemize}
