\section{Introduction}\label{sec:intro}

Duality theory transforms a constrained optimization problem into a related
problem---the \emph{dual}---whose optimal value provides a bound on the original
(primal) problem. When the bound is tight (strong duality), the dual offers both
computational advantages and deep structural insight: it reveals the
sensitivity of the optimal value to perturbations in the constraints and leads
directly to the Karush--Kuhn--Tucker (KKT) conditions, which are the central
tool for characterizing optimal solutions of convex programs.

This lesson develops Lagrangian duality for the convex optimization
problem in standard form:
\begin{equation}\label{eq:std-form}
\begin{aligned}
  \min_{x \in \R^n} \quad & f_0(x) \\
  \text{subject to} \quad & f_i(x) \le 0, \quad i = 1,\dots,m, \\
                          & a_j^\top x = b_j, \quad j = 1,\dots,p,
\end{aligned}
\end{equation}
where $f_0, f_1, \dots, f_m$ are convex and the equality constraints are
affine. We denote the feasible set by $\mathcal{F}$ and the optimal value
by $p^*$.

Section~\ref{sec:lagrangian} introduces the Lagrangian and the dual function.
Section~\ref{sec:weak-duality} proves weak duality.
Section~\ref{sec:strong-duality} presents Slater's condition and proves strong
duality. Section~\ref{sec:kkt} derives the KKT conditions and proves both
their necessity (under strong duality) and sufficiency (for convex problems).
Section~\ref{sec:examples} works through detailed examples, and
Section~\ref{sec:exercises} provides exercises.

\section{The Lagrangian and Dual Function}\label{sec:lagrangian}

\begin{definition}[Lagrangian]\label{def:lagrangian}
The \emph{Lagrangian} $L : \R^n \times \R^m \times \R^p \to \R$ associated
with~\eqref{eq:std-form} is
\begin{equation}\label{eq:lagrangian}
  L(x, \lambda, \nu) = f_0(x) + \sum_{i=1}^m \lambda_i f_i(x)
    + \sum_{j=1}^p \nu_j (a_j^\top x - b_j),
\end{equation}
where $\lambda = (\lambda_1, \dots, \lambda_m) \in \R^m$ are the
\emph{dual variables} (or \emph{Lagrange multipliers}) for the inequality
constraints and $\nu = (\nu_1, \dots, \nu_p) \in \R^p$ are the dual
variables for the equality constraints.
\end{definition}

\begin{remark}\label{rem:lagrangian-interpretation}
The Lagrangian can be viewed as a ``relaxation'' of the primal problem.
Instead of enforcing the constraints exactly, we penalize violations
through the multipliers: a large $\lambda_i$ imposes a heavy cost for
violating $f_i(x) \le 0$, while $\nu_j$ penalizes deviations from the
equality $a_j^\top x = b_j$ in either direction.
\end{remark}

\begin{definition}[Lagrange dual function]\label{def:dual-fn}
The \emph{Lagrange dual function} $g : \R^m \times \R^p \to
\R \cup \{-\infty\}$ is defined by
\begin{equation}\label{eq:dual-fn}
  g(\lambda, \nu) = \inf_{x \in \R^n} L(x, \lambda, \nu)
  = \inf_{x \in \R^n} \biggl\{f_0(x) + \sum_{i=1}^m \lambda_i f_i(x)
    + \sum_{j=1}^p \nu_j (a_j^\top x - b_j) \biggr\}.
\end{equation}
\end{definition}

\begin{proposition}[Dual function is concave]\label{prop:dual-concave}
For any optimization problem (not necessarily convex), the Lagrange dual
function $g$ is concave.
\end{proposition}

\begin{proof}
For each fixed $x \in \R^n$, the function
$(\lambda, \nu) \mapsto L(x, \lambda, \nu)$ is affine in $(\lambda, \nu)$,
because $L$ is a sum of terms each linear in $\lambda_i$ or $\nu_j$. An
affine function is both convex and concave. The pointwise infimum of a family
of concave functions is concave:

Let $\alpha \in [0,1]$ and $(\lambda, \nu)$, $(\lambda', \nu')$ be given.
For each $x$:
\begin{align*}
  L(x, \alpha\lambda + (1{-}\alpha)\lambda', \alpha\nu + (1{-}\alpha)\nu')
  &= \alpha L(x, \lambda, \nu) + (1{-}\alpha) L(x, \lambda', \nu').
\end{align*}
Taking the infimum over $x$ on both sides:
\begin{align*}
  g(\alpha\lambda {+} (1{-}\alpha)\lambda', \alpha\nu {+} (1{-}\alpha)\nu')
  &= \inf_x \bigl[\alpha L(x,\lambda,\nu) + (1{-}\alpha) L(x,\lambda',\nu')\bigr] \\
  &\ge \alpha \inf_x L(x,\lambda,\nu) + (1{-}\alpha) \inf_x L(x,\lambda',\nu') \\
  &= \alpha\, g(\lambda,\nu) + (1{-}\alpha)\, g(\lambda',\nu'),
\end{align*}
where the inequality uses $\inf(f + g) \ge \inf f + \inf g$. This establishes
concavity.
\end{proof}

\begin{example}[Dual of a linear program]\label{ex:lp-dual}
Consider the LP: $\min\, c^\top x$ s.t. $Ax \le b$, with $A \in \R^{m \times n}$.
The Lagrangian is $L(x, \lambda) = c^\top x + \lambda^\top(Ax - b)
= (c + A^\top \lambda)^\top x - \lambda^\top b$.
The infimum over $x$ is $-\infty$ unless $c + A^\top \lambda = 0$, so
\[
  g(\lambda) = \begin{cases}
    -\lambda^\top b & \text{if } A^\top \lambda + c = 0, \\
    -\infty & \text{otherwise.}
  \end{cases}
\]
The dual problem is $\max_{\lambda \ge 0} \{-\lambda^\top b : A^\top \lambda + c = 0\}$,
which is the standard LP dual.
\end{example}

\begin{example}[Dual of a QP]\label{ex:qp-dual}
Consider $\min\, \frac{1}{2}x^\top P x + q^\top x$ s.t. $Ax \le b$ with
$P \succ 0$. The Lagrangian is
$L(x,\lambda) = \frac{1}{2}x^\top P x + (q + A^\top\lambda)^\top x - \lambda^\top b$.
Setting $\nabla_x L = Px + q + A^\top\lambda = 0$ gives
$x^*(\lambda) = -P^{-1}(q + A^\top\lambda)$. Substituting:
\[
  g(\lambda) = -\frac{1}{2}(q + A^\top\lambda)^\top P^{-1}(q + A^\top\lambda)
  - \lambda^\top b.
\]
This is a concave quadratic in $\lambda$, so the dual is also a QP.
\end{example}

\section{Weak Duality}\label{sec:weak-duality}

\begin{definition}[Dual problem]\label{def:dual-problem}
The \emph{Lagrange dual problem} associated with~\eqref{eq:std-form} is
\begin{equation}\label{eq:dual-problem}
  \max_{\lambda \ge 0,\, \nu} \; g(\lambda, \nu).
\end{equation}
Its optimal value is denoted $d^*$. A pair $(\lambda, \nu)$ with
$\lambda \ge 0$ is called \emph{dual feasible}. The quantity $p^* - d^*$
is the \emph{duality gap}.
\end{definition}

\begin{theorem}[Weak duality]\label{thm:weak-duality}
For any optimization problem (not necessarily convex),
\begin{equation}\label{eq:weak-duality}
  d^* \le p^*.
\end{equation}
\end{theorem}

\begin{proof}
Let $\tilde{x}$ be any primal feasible point: $f_i(\tilde{x}) \le 0$ for
$i = 1,\dots,m$ and $a_j^\top \tilde{x} = b_j$ for $j = 1,\dots,p$.
Let $(\lambda, \nu)$ be any dual feasible pair: $\lambda \ge 0$.

We compute:
\begin{align}
  L(\tilde{x}, \lambda, \nu)
  &= f_0(\tilde{x}) + \sum_{i=1}^m \underbrace{\lambda_i}_{\ge\, 0}
    \underbrace{f_i(\tilde{x})}_{\le\, 0}
    + \sum_{j=1}^p \nu_j \underbrace{(a_j^\top \tilde{x} - b_j)}_{=\, 0}
    \notag \\
  &\le f_0(\tilde{x}). \label{eq:wd-step1}
\end{align}
Taking the infimum over $x \in \R^n$:
\begin{equation}\label{eq:wd-step2}
  g(\lambda, \nu) = \inf_{x} L(x, \lambda, \nu) \le L(\tilde{x}, \lambda, \nu)
  \le f_0(\tilde{x}).
\end{equation}
Since this holds for every primal feasible $\tilde{x}$:
\begin{equation}\label{eq:wd-step3}
  g(\lambda, \nu) \le \inf_{\tilde{x} \in \mathcal{F}} f_0(\tilde{x}) = p^*.
\end{equation}
Since this holds for every dual feasible $(\lambda, \nu)$:
\begin{equation}\label{eq:wd-step4}
  d^* = \sup_{\lambda \ge 0, \nu} g(\lambda, \nu) \le p^*. \qedhere
\end{equation}
\end{proof}

\begin{keyresult}[title={Weak Duality}]
For any optimization problem, $d^* \le p^*$. The dual function $g(\lambda, \nu)$
evaluated at any dual-feasible point provides a \emph{lower bound} on the primal
optimal value. This holds regardless of convexity.
\end{keyresult}

\begin{corollary}[Certificate of optimality]\label{cor:certificate}
If a primal feasible $x$ and dual feasible $(\lambda, \nu)$ satisfy
$f_0(x) = g(\lambda, \nu)$, then $x$ is primal optimal and
$(\lambda, \nu)$ is dual optimal.
\end{corollary}

\begin{proof}
By weak duality, $g(\lambda, \nu) \le d^* \le p^* \le f_0(x)$. If
$f_0(x) = g(\lambda, \nu)$, all inequalities are equalities.
\end{proof}

\section{Strong Duality and Slater's Condition}\label{sec:strong-duality}

\begin{definition}[Strong duality]\label{def:strong-duality}
\emph{Strong duality} holds for an optimization problem when $d^* = p^*$,
i.e., the duality gap is zero.
\end{definition}

\begin{definition}[Slater's condition]\label{def:slater}
The convex program~\eqref{eq:std-form} satisfies \emph{Slater's condition}
(or \emph{Slater's constraint qualification}) if there exists a point
$\hat{x} \in \text{relint}(\dom f_0)$ such that
\begin{equation}\label{eq:slater}
  f_i(\hat{x}) < 0, \quad i = 1,\dots,m, \qquad
  a_j^\top \hat{x} = b_j, \quad j = 1,\dots,p.
\end{equation}
Such a point $\hat{x}$ is called a \emph{strictly feasible point}. Note
that strict inequality is required for the inequality constraints, but the
equality constraints need only be satisfied exactly.
\end{definition}

\begin{remark}\label{rem:slater-affine}
If some inequality constraints are affine (say $f_i$ is affine for
$i \in \mathcal{A}$), Slater's condition can be relaxed: the affine
inequalities need only be satisfied with $\le$ (not strictly $<$) at
the Slater point. This is sometimes called the \emph{refined Slater
condition}.
\end{remark}

\begin{theorem}[Slater implies strong duality]\label{thm:strong-duality}
If the convex program~\eqref{eq:std-form} satisfies Slater's condition and
$p^*$ is finite, then strong duality holds: $d^* = p^*$. Moreover, the dual
optimal value is attained: there exist $\lambda^* \ge 0$, $\nu^*$ with
$g(\lambda^*, \nu^*) = d^* = p^*$.
\end{theorem}

\begin{proof}
We use a supporting hyperplane argument on the set of achievable constraint
slack and objective values.

\textbf{Step 1: Define the achievable set.}
Let
\begin{equation}\label{eq:set-G}
  \mathcal{G} = \bigl\{(u, v, t) \in \R^m \times \R^p \times \R :
    \exists\, x \;\text{with}\; f_i(x) \le u_i, \;\;
    a_j^\top x - b_j = v_j, \;\; f_0(x) \le t \bigr\}.
\end{equation}
This set is convex. To see this, let $(u, v, t)$ and $(u', v', t')$ lie in
$\mathcal{G}$, witnessed by $x$ and $x'$ respectively. For
$\alpha \in [0,1]$, set $\bar{x} = \alpha x + (1{-}\alpha) x'$. By
convexity of $f_i$:
$f_i(\bar{x}) \le \alpha f_i(x) + (1{-}\alpha) f_i(x') \le \alpha u_i
+ (1{-}\alpha) u'_i$. Similarly for the affine equalities and the
objective. Hence $\alpha(u,v,t) + (1{-}\alpha)(u',v',t') \in \mathcal{G}$.

\textbf{Step 2: The point $(0, 0, p^*)$ is on the boundary of $\mathcal{G}$.}
By definition of $p^*$, the point $(0, 0, p^*)$ lies in $\cl(\mathcal{G})$
(it is achieved in the limit by a minimizing sequence). However,
$(0, 0, p^* - \epsilon) \notin \mathcal{G}$ for any $\epsilon > 0$, since
that would require a feasible $x$ with $f_0(x) \le p^* - \epsilon < p^*$,
contradicting the definition of $p^*$.

\textbf{Step 3: Supporting hyperplane.}
By the supporting hyperplane theorem for convex sets, there exists
$(\tilde{\lambda}, \tilde{\nu}, \mu) \neq (0, 0, 0)$ with
$\tilde{\lambda} \in \R^m$, $\tilde{\nu} \in \R^p$, $\mu \in \R$ such that
\begin{equation}\label{eq:supporting}
  \tilde{\lambda}^\top u + \tilde{\nu}^\top v + \mu\, t
  \ge \mu\, p^*
\end{equation}
for all $(u, v, t) \in \mathcal{G}$. Since $(u, v, t) \in \mathcal{G}$
allows $u_i \to +\infty$, we must have $\tilde{\lambda}_i \ge 0$ for each
$i$; and since $t \to +\infty$ is allowed, $\mu \ge 0$.

\textbf{Step 4: Show $\mu > 0$ using Slater's condition.}
Suppose for contradiction that $\mu = 0$. Then~\eqref{eq:supporting} becomes
$\tilde{\lambda}^\top u + \tilde{\nu}^\top v \ge 0$ for all
$(u, v, t) \in \mathcal{G}$. Evaluate at the Slater point $\hat{x}$: take
$u_i = f_i(\hat{x}) < 0$, $v_j = 0$, $t = f_0(\hat{x})$. This gives
\[
  \sum_{i=1}^m \tilde{\lambda}_i f_i(\hat{x}) \ge 0.
\]
Since $\tilde{\lambda}_i \ge 0$ and $f_i(\hat{x}) < 0$, each term is
non-positive. The sum being non-negative forces every term to be zero.
Since $f_i(\hat{x}) < 0$, we get $\tilde{\lambda}_i = 0$ for all $i$.

Now the inequality reduces to $\tilde{\nu}^\top v \ge 0$ for all achievable
$v$. Since the equality constraints are affine, $v = Ax - b$ for a matrix
$A$ with rows $a_j^\top$ and vector $b$ with entries $b_j$. As $x$ ranges
over $\R^n$, $v$ ranges over $\text{range}(A) - b \supseteq \{0\}$. For any
direction $d \in \text{range}(A)$, both $d$ and $-d$ are achievable
perturbations, so $\tilde{\nu}^\top d \ge 0$ and
$-\tilde{\nu}^\top d \ge 0$, giving $\tilde{\nu}^\top d = 0$ for all $d$
in the range. This requires $\tilde{\nu} = 0$ (assuming the equality
constraints are non-redundant; otherwise $\tilde{\nu}$ lies in the null
space of $A^\top$, but the supporting hyperplane must be nonzero). This
contradicts $(\tilde{\lambda}, \tilde{\nu}, \mu) \neq (0,0,0)$.

Therefore $\mu > 0$.

\textbf{Step 5: Conclude strong duality.}
Divide~\eqref{eq:supporting} by $\mu > 0$ and define
$\lambda^* = \tilde{\lambda}/\mu$, $\nu^* = \tilde{\nu}/\mu$. We have
$\lambda^* \ge 0$. For any $x$ that achieves $(u, v, t)$
with $u_i = f_i(x)$, $v_j = a_j^\top x - b_j$, $t = f_0(x)$:
\[
  \sum_{i=1}^m \lambda^*_i f_i(x) + \sum_{j=1}^p \nu^*_j(a_j^\top x - b_j)
  + f_0(x) \ge p^*.
\]
Hence $L(x, \lambda^*, \nu^*) \ge p^*$ for all $x$. Taking the infimum
over $x$:
\[
  g(\lambda^*, \nu^*) = \inf_x L(x, \lambda^*, \nu^*) \ge p^*.
\]
By weak duality (Theorem~\ref{thm:weak-duality}), $g(\lambda^*, \nu^*) \le p^*$.
Therefore $g(\lambda^*, \nu^*) = p^*$, so $d^* \ge g(\lambda^*, \nu^*) = p^*$.
Combined with $d^* \le p^*$ from weak duality, we conclude $d^* = p^*$, and
the dual optimum is attained at $(\lambda^*, \nu^*)$.
\end{proof}

\begin{keyresult}[title={Strong Duality via Slater's Condition}]
For a convex program, if a strictly feasible point exists (Slater's condition),
then $d^* = p^*$. The dual optimum is attained. This converts the constrained
primal problem into an equivalent (unconstrained concave) dual problem.
\end{keyresult}

\section{Karush--Kuhn--Tucker Conditions}\label{sec:kkt}

\begin{definition}[KKT conditions]\label{def:kkt}
For the convex program~\eqref{eq:std-form} with differentiable
$f_0, f_1, \dots, f_m$, the \emph{Karush--Kuhn--Tucker (KKT) conditions}
for a primal-dual triple $(x^*, \lambda^*, \nu^*)$ are:
\begin{enumerate}[label=\textbf{(KKT\arabic*)}]
  \item \textbf{Primal feasibility:}\label{kkt:primal}
    $f_i(x^*) \le 0$ for $i = 1,\dots,m$; \;
    $a_j^\top x^* = b_j$ for $j = 1,\dots,p$.
  \item \textbf{Dual feasibility:}\label{kkt:dual}
    $\lambda^*_i \ge 0$ for $i = 1,\dots,m$.
  \item \textbf{Complementary slackness:}\label{kkt:cs}
    $\lambda^*_i f_i(x^*) = 0$ for $i = 1,\dots,m$.
  \item \textbf{Stationarity:}\label{kkt:stat}
    $\displaystyle \nabla f_0(x^*) + \sum_{i=1}^m \lambda^*_i \nabla f_i(x^*)
     + \sum_{j=1}^p \nu^*_j a_j = 0$.
\end{enumerate}
\end{definition}

\begin{remark}[Interpretation of complementary slackness]\label{rem:cs}
The condition $\lambda^*_i f_i(x^*) = 0$ says: for each inequality constraint,
either the constraint is \emph{active} ($f_i(x^*) = 0$) or its multiplier is
zero ($\lambda^*_i = 0$), or both. A multiplier $\lambda^*_i > 0$ means the
constraint is ``binding''---removing it would improve the objective.
\end{remark}

\begin{theorem}[KKT necessary conditions]\label{thm:kkt-necessary}
Suppose strong duality holds for~\eqref{eq:std-form}, the functions
$f_0, f_1, \dots, f_m$ are differentiable, $x^*$ is primal optimal, and
$(\lambda^*, \nu^*)$ is dual optimal. Then the KKT conditions hold at
$(x^*, \lambda^*, \nu^*)$.
\end{theorem}

\begin{proof}
Since strong duality holds:
\begin{equation}\label{eq:chain-start}
  f_0(x^*) = p^* = d^* = g(\lambda^*, \nu^*)
  = \inf_{x} L(x, \lambda^*, \nu^*) \le L(x^*, \lambda^*, \nu^*).
\end{equation}

Expanding the Lagrangian at $(x^*, \lambda^*, \nu^*)$:
\begin{equation}\label{eq:expand-L}
  L(x^*, \lambda^*, \nu^*) = f_0(x^*)
    + \sum_{i=1}^m \lambda^*_i f_i(x^*)
    + \sum_{j=1}^p \nu^*_j \underbrace{(a_j^\top x^* - b_j)}_{=\,0}.
\end{equation}
The equality constraint terms vanish by primal feasibility. The inequality
constraint terms satisfy $\lambda^*_i f_i(x^*) \le 0$ (since
$\lambda^*_i \ge 0$ and $f_i(x^*) \le 0$). Therefore:
\begin{equation}\label{eq:upper-bound}
  L(x^*, \lambda^*, \nu^*) = f_0(x^*) + \sum_{i=1}^m \lambda^*_i f_i(x^*)
  \le f_0(x^*).
\end{equation}

Combining~\eqref{eq:chain-start} and~\eqref{eq:upper-bound}:
\[
  f_0(x^*) \le L(x^*, \lambda^*, \nu^*) \le f_0(x^*),
\]
so $L(x^*, \lambda^*, \nu^*) = f_0(x^*)$. This means
$\sum_{i=1}^m \lambda^*_i f_i(x^*) = 0$. Since each term
$\lambda^*_i f_i(x^*)$ is non-positive, each must be zero:
\begin{equation}\label{eq:cs-derived}
  \lambda^*_i f_i(x^*) = 0, \quad i = 1, \dots, m.
\end{equation}
This establishes \textbf{complementary slackness}.

From~\eqref{eq:chain-start}, $g(\lambda^*, \nu^*) = L(x^*, \lambda^*, \nu^*)$,
which means $x^*$ attains the infimum in the definition of
$g(\lambda^*, \nu^*)$. Since $x \mapsto L(x, \lambda^*, \nu^*)$ is a
differentiable convex function (it is a non-negatively weighted sum of convex
functions) and $x^*$ is its global minimizer over $\R^n$, the gradient must
vanish:
\begin{equation}\label{eq:stat-derived}
  \nabla_x L(x^*, \lambda^*, \nu^*)
  = \nabla f_0(x^*) + \sum_{i=1}^m \lambda^*_i \nabla f_i(x^*)
    + \sum_{j=1}^p \nu^*_j a_j = 0.
\end{equation}
This establishes \textbf{stationarity}. Primal and dual feasibility hold by
assumption.
\end{proof}

\begin{theorem}[KKT sufficient conditions for convex problems]\label{thm:kkt-sufficient}
For the convex program~\eqref{eq:std-form} with differentiable
$f_0, f_1, \dots, f_m$, if a triple $(x^*, \lambda^*, \nu^*)$ satisfies
the KKT conditions, then $x^*$ is primal optimal and $(\lambda^*, \nu^*)$
is dual optimal, with zero duality gap.
\end{theorem}

\begin{proof}
\textbf{Step 1:} By stationarity (KKT4), $\nabla_x L(x^*, \lambda^*, \nu^*) = 0$.
The function $x \mapsto L(x, \lambda^*, \nu^*)$ is convex in $x$ (since
$f_0, f_1, \dots, f_m$ are convex, $\lambda^* \ge 0$, and the remaining
terms are affine). A differentiable convex function with vanishing gradient
at $x^*$ attains its global minimum there. Hence:
\begin{equation}\label{eq:suff-step1}
  g(\lambda^*, \nu^*) = \inf_x L(x, \lambda^*, \nu^*)
  = L(x^*, \lambda^*, \nu^*).
\end{equation}

\textbf{Step 2:} By complementary slackness (KKT3) and primal feasibility
(KKT1):
\begin{equation}\label{eq:suff-step2}
  L(x^*, \lambda^*, \nu^*) = f_0(x^*)
    + \underbrace{\sum_{i=1}^m \lambda^*_i f_i(x^*)}_{=\,0}
    + \underbrace{\sum_{j=1}^p \nu^*_j(a_j^\top x^* - b_j)}_{=\,0}
  = f_0(x^*).
\end{equation}

\textbf{Step 3:} Combining:
$g(\lambda^*, \nu^*) = f_0(x^*)$.
By weak duality (Theorem~\ref{thm:weak-duality}):
\[
  d^* \ge g(\lambda^*, \nu^*) = f_0(x^*) \ge p^* \ge d^*.
\]
All inequalities are equalities. Hence $f_0(x^*) = p^*$ ($x^*$ is primal
optimal), $g(\lambda^*, \nu^*) = d^*$ ($(\lambda^*, \nu^*)$ is dual optimal),
and $p^* = d^*$ (zero duality gap).
\end{proof}

\begin{keyresult}[title={KKT Conditions for Convex Problems}]
For differentiable convex programs, the KKT conditions are:
\begin{itemize}
  \item \textbf{Necessary} for optimality, provided a constraint qualification
    (such as Slater's condition) holds.
  \item \textbf{Sufficient} for optimality, without any constraint qualification.
\end{itemize}
Together, the KKT conditions provide a complete characterization of primal-dual
optimal pairs.
\end{keyresult}

\begin{rlconnection}
\textbf{Constrained MDPs.} In safe reinforcement learning, the agent maximizes
expected return subject to constraints on expected cost:
$\max_\pi J(\pi)$ s.t. $C_i(\pi) \le d_i$. The Lagrangian relaxation is
$\max_\pi \min_{\lambda \ge 0} \{J(\pi) - \sum_i \lambda_i (C_i(\pi) - d_i)\}$.
Slater's condition guarantees strong duality when a strictly feasible policy
exists, justifying primal-dual algorithms such as Lagrangian PPO.

\textbf{Trust-region methods.} In TRPO, the policy update solves
$\max_\theta L(\theta)$ s.t. $\text{KL}(\pi_{\text{old}} \| \pi_\theta) \le \delta$.
The KKT multiplier $\lambda^*$ on the KL constraint determines the effective
step size. Complementary slackness says: either the KL constraint is active
(the update reaches the trust-region boundary) or $\lambda^* = 0$ (the
unconstrained maximum lies inside the trust region).
\end{rlconnection}

\section{Detailed Examples}\label{sec:examples}

\begin{example}[Water-filling]\label{ex:water-filling}
Consider the problem of allocating power across $n$ channels to maximize
total capacity:
\[
  \max_{x \ge 0} \; \sum_{i=1}^n \log(\alpha_i + x_i) \quad
  \text{s.t.} \quad \sum_{i=1}^n x_i = P,
\]
where $\alpha_i > 0$ are channel gains and $P > 0$ is total power.
Equivalently, minimize $-\sum_i \log(\alpha_i + x_i)$. The Lagrangian is
\[
  L(x, \lambda, \nu) = -\sum_i \log(\alpha_i + x_i)
  + \sum_i \lambda_i(-x_i) + \nu\Bigl(\sum_i x_i - P\Bigr).
\]
The KKT stationarity condition gives
$\frac{-1}{\alpha_i + x_i^*} - \lambda_i^* + \nu^* = 0$, hence
$x_i^* + \alpha_i = \frac{1}{\nu^* - \lambda_i^*}$. Complementary
slackness ($\lambda_i^* x_i^* = 0$) yields the ``water-filling'' solution:
\[
  x_i^* = \max\Bigl(\frac{1}{\nu^*} - \alpha_i, \; 0\Bigr),
\]
where $\nu^*$ (the ``water level'') is chosen so that $\sum_i x_i^* = P$.
\end{example}

\begin{example}[Equality-constrained QP via KKT]\label{ex:eq-qp}
Consider $\min\, \frac{1}{2}x^\top P x + q^\top x$ s.t. $Ax = b$ with
$P \succ 0$. There are no inequality constraints, so the KKT conditions
reduce to stationarity and primal feasibility:
\[
  Px^* + q + A^\top \nu^* = 0, \qquad Ax^* = b.
\]
In matrix form:
$\begin{pmatrix} P & A^\top \\ A & 0 \end{pmatrix}
\begin{pmatrix} x^* \\ \nu^* \end{pmatrix}
= \begin{pmatrix} -q \\ b \end{pmatrix}$.
This is the \emph{KKT system}, a linear system of equations.
\end{example}

\begin{example}[Sensitivity interpretation]\label{ex:sensitivity}
Consider $p^*(\epsilon) = \min\{f_0(x) : f_1(x) \le \epsilon\}$. If strong
duality holds and $\lambda^*$ is the optimal dual variable, then
$\frac{dp^*}{d\epsilon}\big|_{\epsilon=0} = -\lambda^*$. A large multiplier
means the constraint is ``expensive'': relaxing it slightly yields a large
improvement in the objective.
\end{example}

\section{Exercises}\label{sec:exercises}

\begin{exercise}[Weak duality bound]\label{ex:wd-bound}
Consider $\min\, x_1 + x_2$ s.t. $x_1^2 + x_2^2 \le 1$. Find the dual
function $g(\lambda)$, evaluate it at $\lambda = 1$, and verify that
$g(1) \le p^*$. Then find $d^*$ and verify strong duality.
\end{exercise}

\begin{exercise}[Slater verification]\label{ex:slater-check}
For the problem $\min\, e^{-x}$ s.t. $x^2/y \le 0$, $y > 0$, determine
whether Slater's condition holds. If not, find the duality gap.
\end{exercise}

\begin{exercise}[KKT for box constraints]\label{ex:kkt-box}
Solve $\min\, (x_1 - 3)^2 + (x_2 - 5)^2$ s.t. $0 \le x_1 \le 4$,
$0 \le x_2 \le 3$ by writing and solving the KKT conditions. Identify
which constraints are active and which multipliers are nonzero.
\end{exercise}

\begin{exercise}[Dual of norm minimization]\label{ex:dual-norm}
Derive the dual of the problem $\min\, \|x\|_2$ s.t. $Ax = b$.
Show that the dual is $\max\, b^\top \nu$ s.t. $\|A^\top \nu\|_2 \le 1$.
Verify that strong duality holds (under what condition?).
\end{exercise}

\begin{exercise}[Complementary slackness in LP]\label{ex:cs-lp}
Given the LP $\min\, c^\top x$ s.t. $Ax \le b$, $x \ge 0$ and its dual
$\max\, b^\top y$ s.t. $A^\top y + s = c$, $y, s \ge 0$, show that at
optimality $x_i^* s_i^* = 0$ and $y_j^*(b_j - A_j x^*) = 0$. Explain the
economic interpretation: each variable is either zero or its associated
constraint is binding.
\end{exercise}

\begin{exercise}[CMDP Lagrangian relaxation]\label{ex:cmdp}
Consider a constrained MDP: $\max_\pi \E[\sum_t \gamma^t r_t]$ s.t.
$\E[\sum_t \gamma^t c_t] \le d$. Write the Lagrangian. Show that for
fixed $\lambda \ge 0$, the inner maximization over $\pi$ reduces to an
unconstrained MDP with modified reward $r_t + \lambda(d - c_t)$. Explain
how to find the optimal $\lambda^*$ via a one-dimensional search.
\end{exercise}

\begin{exercise}[Non-convex duality gap]\label{ex:noncvx-gap}
Consider $\min\, x^2$ s.t. $(x-1)(x-2) \le 0$ (feasible set $[1,2]$).
Compute $p^*$, derive the dual function, compute $d^*$, and show that
there is a nonzero duality gap. Explain why Slater's condition is
not applicable.
\end{exercise}
