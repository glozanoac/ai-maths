\section{Motivation and Overview}
\label{sec:motivation}

This lesson develops the core probabilistic tools that underpin every
convergence proof in reinforcement learning: conditional probability and
Bayes' theorem, expectation defined rigorously via the Lebesgue integral,
linearity of expectation, the common discrete distributions (Bernoulli,
categorical, geometric), and joint/marginal distributions.

\section{Key Definitions (Review from Lesson 4)}
\label{sec:review}

We recall the core definitions from Lesson~4 to fix notation.
Throughout, $(\Omega, \mathcal{F}, \Prob)$ is a probability space
(i.e., a measure space with $\Prob(\Omega) = 1$), and a \emph{random
variable} is a measurable function
$X: (\Omega, \mathcal{F}) \to (\R, \Borel(\R))$ with CDF
$F_X(x) = \Prob(X \leq x)$.

The basic properties of probability (complement rule, monotonicity,
inclusion-exclusion, union bound, continuity from above/below) and the
CDF properties (non-decreasing, correct limits at $\pm\infty$,
right-continuity) were proved in Lesson~4. We use them freely below.

\section{Conditional Probability and Bayes' Theorem}
\label{sec:cond_prob}

\begin{definition}[Conditional Probability]
\label{def:cond_prob}
For events $A, B \in \mathcal{F}$ with $\Prob(B) > 0$, the
\emph{conditional probability of $A$ given $B$} is
\begin{equation}
  \Prob(A \mid B) = \frac{\Prob(A \cap B)}{\Prob(B)}.
  \label{eq:cond_prob}
\end{equation}
\end{definition}

\begin{proposition}[Conditional Probability is a Probability Measure]
\label{prop:cond_is_prob}
For fixed $B$ with $\Prob(B) > 0$, the function
$A \mapsto \Prob(A \mid B)$ is a probability measure on
$(\Omega, \mathcal{F})$.
\end{proposition}

\begin{proof}
We verify the three Kolmogorov axioms.

\textit{(K1)} $\Prob(A \mid B) = \Prob(A \cap B)/\Prob(B) \geq 0$.

\textit{(K2)} $\Prob(\Omega \mid B) = \Prob(\Omega \cap B)/\Prob(B)
= \Prob(B)/\Prob(B) = 1$.

\textit{(K3)} For pairwise disjoint $A_1, A_2, \ldots$:
\begin{align}
  \Prob\!\left(\bigcup_n A_n \;\middle|\; B\right)
  &= \frac{\Prob((\bigcup_n A_n) \cap B)}{\Prob(B)}
  = \frac{\Prob(\bigcup_n (A_n \cap B))}{\Prob(B)}.
\end{align}
The sets $A_n \cap B$ are pairwise disjoint (since the $A_n$ are), so by
countable additivity of $\Prob$:
\[
  = \frac{\sum_n \Prob(A_n \cap B)}{\Prob(B)}
  = \sum_n \frac{\Prob(A_n \cap B)}{\Prob(B)}
  = \sum_n \Prob(A_n \mid B). \qedhere
\]
\end{proof}

\begin{theorem}[Chain Rule of Probability]
\label{thm:chain_rule}
For events $A_1, \ldots, A_n$ with
$\Prob(A_1 \cap \cdots \cap A_{n-1}) > 0$:
\begin{equation}
  \Prob(A_1 \cap \cdots \cap A_n)
  = \Prob(A_1)\, \Prob(A_2 \mid A_1)\,
    \Prob(A_3 \mid A_1 \cap A_2)
    \cdots \Prob(A_n \mid A_1 \cap \cdots \cap A_{n-1}).
  \label{eq:chain_rule}
\end{equation}
\end{theorem}

\begin{proof}
We proceed by induction on $n$.

\textit{Base case ($n = 2$).} By definition of conditional probability,
$\Prob(A_1 \cap A_2) = \Prob(A_1) \cdot \Prob(A_2 \mid A_1)$.

\textit{Inductive step.} Assume the formula holds for $n - 1$ events.
Let $B = A_1 \cap \cdots \cap A_{n-1}$. By the base case applied to $B$
and $A_n$:
\[
  \Prob(B \cap A_n) = \Prob(B) \cdot \Prob(A_n \mid B).
\]
By the inductive hypothesis, $\Prob(B)$ expands as
\[
  \Prob(B) = \Prob(A_1)\, \Prob(A_2 \mid A_1) \cdots
  \Prob(A_{n-1} \mid A_1 \cap \cdots \cap A_{n-2}).
\]
Combining gives the formula for $n$ events.
\end{proof}

\begin{rlconnection}[title={Connection to RL}]
The chain rule decomposes trajectory probabilities:
\[
  \Prob_\pi(S_0, A_0, R_1, S_1, \ldots)
  = p_0(S_0) \prod_{t=0}^{T-1} \pi(A_t \mid S_t)\,
    p(S_{t+1}, R_{t+1} \mid S_t, A_t).
\]
This factorization is the starting point of importance sampling ratios
in off-policy methods (TRPO, PPO).
\end{rlconnection}

\begin{keyresult}[title={Law of Total Probability}]
\begin{theorem}[Law of Total Probability]
\label{thm:total_prob}
Let $B_1, B_2, \ldots, B_n$ be a partition of $\Omega$ (pairwise disjoint,
$\bigcup_i B_i = \Omega$) with $\Prob(B_i) > 0$ for all $i$. Then for any
event $A \in \mathcal{F}$:
\begin{equation}
  \Prob(A) = \sum_{i=1}^n \Prob(A \mid B_i)\, \Prob(B_i).
  \label{eq:total_prob}
\end{equation}
\end{theorem}
\end{keyresult}

\begin{proof}
Since $\{B_i\}$ partitions $\Omega$:
\[
  A = A \cap \Omega = A \cap \bigcup_{i=1}^n B_i
  = \bigcup_{i=1}^n (A \cap B_i).
\]
The sets $A \cap B_1, \ldots, A \cap B_n$ are pairwise disjoint (inherited
from the $B_i$). By finite additivity:
\[
  \Prob(A) = \sum_{i=1}^n \Prob(A \cap B_i)
  = \sum_{i=1}^n \Prob(A \mid B_i)\, \Prob(B_i),
\]
where the last step uses $\Prob(A \cap B_i) = \Prob(A \mid B_i) \Prob(B_i)$
from the definition of conditional probability.
\end{proof}

\begin{keyresult}[title={Bayes' Theorem}]
\begin{theorem}[Bayes' Theorem]
\label{thm:bayes}
Let $B_1, \ldots, B_n$ partition $\Omega$ with $\Prob(B_i) > 0$, and let
$A$ be an event with $\Prob(A) > 0$. Then for each $j$:
\begin{equation}
  \Prob(B_j \mid A)
  = \frac{\Prob(A \mid B_j)\, \Prob(B_j)}
         {\sum_{i=1}^n \Prob(A \mid B_i)\, \Prob(B_i)}.
  \label{eq:bayes}
\end{equation}
\end{theorem}
\end{keyresult}

\begin{proof}
By the definition of conditional probability:
\[
  \Prob(B_j \mid A) = \frac{\Prob(A \cap B_j)}{\Prob(A)}.
\]
The numerator satisfies $\Prob(A \cap B_j) = \Prob(A \mid B_j) \Prob(B_j)$
(definition of conditional probability). The denominator is expanded using
the law of total probability (Theorem~\ref{thm:total_prob}):
$\Prob(A) = \sum_{i=1}^n \Prob(A \mid B_i) \Prob(B_i)$.
Substituting gives the result.
\end{proof}

\section{Expectation via the Lebesgue Integral}
\label{sec:expectation}

\subsection{Integral of Simple Functions}
\label{subsec:integral_simple}

\begin{definition}[Integral of a Non-negative Simple Function]
\label{def:integral_simple}
Let $(\Omega, \mathcal{F}, \mu)$ be a measure space and let
$\varphi = \sum_{i=1}^n a_i \indicator_{A_i}$ be a non-negative simple
function in standard representation ($a_i \geq 0$ distinct,
$A_i = \varphi^{-1}(\{a_i\})$). The \emph{Lebesgue integral} of $\varphi$
is
\begin{equation}
  \int_\Omega \varphi\, d\mu = \sum_{i=1}^n a_i\, \mu(A_i).
  \label{eq:integral_simple}
\end{equation}
\end{definition}

\begin{proposition}[Properties of the Simple Integral]
\label{prop:simple_integral_props}
For non-negative simple functions $\varphi, \psi$ and $c \geq 0$:
\begin{enumerate}[nosep,label=(\alph*)]
  \item $\int c\varphi\, d\mu = c \int \varphi\, d\mu$.
  \item $\int (\varphi + \psi)\, d\mu
    = \int \varphi\, d\mu + \int \psi\, d\mu$.
  \item If $\varphi \leq \psi$ pointwise, then
    $\int \varphi\, d\mu \leq \int \psi\, d\mu$.
\end{enumerate}
\end{proposition}

\begin{proof}
\textbf{(a)} If $\varphi = \sum a_i \indicator_{A_i}$, then
$c\varphi = \sum (ca_i) \indicator_{A_i}$. Hence
$\int c\varphi\, d\mu = \sum ca_i\, \mu(A_i)
= c \sum a_i\, \mu(A_i) = c \int \varphi\, d\mu$.

\textbf{(b)} Let $\varphi = \sum_{i=1}^m a_i \indicator_{A_i}$ and
$\psi = \sum_{j=1}^n b_j \indicator_{B_j}$ in standard representations.
The sets $C_{ij} = A_i \cap B_j$ form a partition of $\Omega$ (since both
$\{A_i\}$ and $\{B_j\}$ are partitions). On $C_{ij}$,
$\varphi + \psi = a_i + b_j$. Therefore:
\begin{align}
  \int (\varphi + \psi)\, d\mu
  &= \sum_{i,j} (a_i + b_j)\, \mu(C_{ij}) \nonumber\\
  &= \sum_i a_i \sum_j \mu(C_{ij}) + \sum_j b_j \sum_i \mu(C_{ij})
   \nonumber\\
  &= \sum_i a_i\, \mu(A_i) + \sum_j b_j\, \mu(B_j)
  = \int \varphi\, d\mu + \int \psi\, d\mu. \label{eq:simple_linearity}
\end{align}
Here we used $\sum_j \mu(C_{ij}) = \mu(A_i)$ (since
$\bigsqcup_j C_{ij} = A_i$).

\textbf{(c)} $\psi - \varphi \geq 0$, so
$\int \psi\, d\mu = \int \varphi\, d\mu + \int (\psi - \varphi)\, d\mu
\geq \int \varphi\, d\mu$.
\end{proof}

\subsection{Non-negative and General Integrals}
\label{subsec:integral_general}

\begin{definition}[Lebesgue Integral -- Non-negative Case]
\label{def:integral_nonneg}
For a non-negative measurable $f: (\Omega, \mathcal{F}) \to [0, \infty]$:
\begin{equation}
  \int_\Omega f\, d\mu
  = \sup\left\{\int_\Omega \varphi\, d\mu :
    0 \leq \varphi \leq f, \; \varphi \text{ simple}\right\}.
  \label{eq:integral_nonneg}
\end{equation}
By the simple function approximation theorem (Lesson~3), simple
$\varphi_n \uparrow f$ exist, and the monotone convergence theorem gives
$\int f\, d\mu = \lim_n \int \varphi_n\, d\mu$.
\end{definition}

\begin{definition}[Lebesgue Integral -- General Case]
\label{def:integral_general}
For measurable $f: \Omega \to [-\infty, \infty]$, write $f = f^+ - f^-$
where $f^+ = \max(f,0)$ and $f^- = \max(-f,0)$. Define
$\int f\, d\mu = \int f^+\, d\mu - \int f^-\, d\mu$ whenever at least one
side is finite. If both are finite, $f$ is \emph{integrable}
($f \in L^1(\mu)$).
\end{definition}

\begin{definition}[Expectation]
\label{def:expectation}
Let $X$ be a random variable on $(\Omega, \mathcal{F}, \Prob)$. The
\emph{expectation} of $X$ is
\begin{equation}
  \E[X] = \int_\Omega X\, d\Prob,
  \label{eq:expectation}
\end{equation}
provided the integral exists (i.e., $\E[|X|] < \infty$).
\end{definition}

\begin{remark}
For a discrete random variable taking values $\{x_k\}_{k \geq 1}$ with
$p_k = \Prob(X = x_k)$, the Lebesgue integral reduces to
$\E[X] = \sum_k x_k p_k$ (provided $\sum_k |x_k| p_k < \infty$). This is
because $X = \lim_N \sum_{k=1}^N x_k \indicator_{\{X = x_k\}}$ and the
integral of each simple approximation is $\sum_{k=1}^N x_k p_k$.
\end{remark}

\section{Linearity of Expectation}
\label{sec:linearity}

\begin{keyresult}[title={Linearity of Expectation}]
\begin{theorem}[Linearity of Expectation]
\label{thm:linearity}
For random variables $X$ and $Y$ with finite expectations, and constants
$a, b \in \R$:
\begin{equation}
  \E[aX + bY] = a\,\E[X] + b\,\E[Y].
  \label{eq:linearity}
\end{equation}
This holds regardless of whether $X$ and $Y$ are independent.
\end{theorem}
\end{keyresult}

\begin{proof}
We prove linearity in stages, building from simple functions to the general
case.

\textbf{Step 1: Homogeneity, $\E[aX] = a\E[X]$.}
For $a \geq 0$ and $X \geq 0$: if $\varphi \leq X$ is simple, then
$a\varphi \leq aX$ is simple and $\int a\varphi = a \int \varphi$
(Proposition~\ref{prop:simple_integral_props}(a)). Taking the supremum:
$\int aX = a \int X$. For general $a$ and $X$, decompose using positive
and negative parts and consider the cases $a \geq 0$ and $a < 0$
separately.

\textbf{Step 2: Additivity for non-negative simple functions.}
This is Proposition~\ref{prop:simple_integral_props}(b).

\textbf{Step 3: Additivity for non-negative measurable functions.}
Let $X, Y \geq 0$ be measurable. By the simple function approximation
theorem, there exist simple $\varphi_n \uparrow X$ and
$\psi_n \uparrow Y$. Then $\varphi_n + \psi_n \uparrow X + Y$ and
\[
  \int (X + Y)\, d\Prob
  = \lim_n \int (\varphi_n + \psi_n)\, d\Prob
  = \lim_n \left(\int \varphi_n\, d\Prob + \int \psi_n\, d\Prob\right)
  = \int X\, d\Prob + \int Y\, d\Prob,
\]
where the first equality uses the monotone convergence theorem (the
integrals of an increasing sequence of non-negative functions converge to
the integral of the limit).

\textbf{Step 4: General case.}
Write $X = X^+ - X^-$ and $Y = Y^+ - Y^-$. Then
$X + Y = (X^+ + Y^+) - (X^- + Y^-)$. By Step~3 applied to the
non-negative functions $X^+ + Y^+$ and $X^- + Y^-$:
$\int(X^+ + Y^+) = \int X^+ + \int Y^+$ and similarly for the negative
parts. It follows that $\E[X+Y] = \E[X] + \E[Y]$.

\textbf{Combining:} $\E[aX + bY] = \E[aX] + \E[bY] = a\E[X] + b\E[Y]$.
\end{proof}

\begin{corollary}[Linearity for Finite Sums]
\label{cor:linearity_sum}
For random variables $X_1, \ldots, X_n$ with finite expectations and
constants $a_1, \ldots, a_n \in \R$:
\begin{equation}
  \E\!\left[\sum_{k=1}^n a_k X_k\right]
  = \sum_{k=1}^n a_k\, \E[X_k].
  \label{eq:linearity_sum}
\end{equation}
\end{corollary}

\begin{proof}
Induction on $n$. The base case $n = 1$ is homogeneity. For the inductive
step, let $Z = \sum_{k=1}^{n-1} a_k X_k$. By
Theorem~\ref{thm:linearity}:
\[
  \E\!\left[\sum_{k=1}^n a_k X_k\right]
  = \E[Z + a_n X_n]
  = \E[Z] + a_n \E[X_n]
  = \sum_{k=1}^{n-1} a_k \E[X_k] + a_n \E[X_n]
  = \sum_{k=1}^n a_k \E[X_k],
\]
where the third equality uses the inductive hypothesis.
\end{proof}

\begin{rlconnection}[title={Connection to RL}]
Linearity is the first step of every Bellman equation derivation. The
return $G_t = R_{t+1} + \gamma G_{t+1}$ gives
\[
  V^\pi(s) = \E_\pi[G_t \mid S_t = s]
  = \E_\pi[R_{t+1} \mid S_t = s]
    + \gamma\, \E_\pi[G_{t+1} \mid S_t = s].
\]
The second term is then expanded using the law of total expectation
(developed in Lesson~6) to obtain the full Bellman equation.
\end{rlconnection}

\section{Common Discrete Distributions}
\label{sec:distributions}

\begin{definition}[Bernoulli Distribution]
\label{def:bernoulli}
A random variable $X$ has a \emph{Bernoulli distribution} with parameter
$p \in [0,1]$, written $X \sim \operatorname{Bernoulli}(p)$, if
$\Prob(X = 1) = p$ and $\Prob(X = 0) = 1 - p$.
\end{definition}

\begin{proposition}[Expectation of Bernoulli]
\label{prop:bernoulli_E}
If $X \sim \operatorname{Bernoulli}(p)$, then $\E[X] = p$.
\end{proposition}

\begin{proof}
$\E[X] = 1 \cdot \Prob(X = 1) + 0 \cdot \Prob(X = 0) = p$.
\end{proof}

\begin{definition}[Categorical Distribution]
\label{def:categorical}
A random variable $X$ taking values in $\{1, 2, \ldots, k\}$ has a
\emph{categorical distribution} with parameters $(p_1, \ldots, p_k)$,
where $p_i \geq 0$ and $\sum_{i=1}^k p_i = 1$, if
$\Prob(X = i) = p_i$ for each $i$.
\end{definition}

\begin{proposition}[Expectation of Categorical]
\label{prop:categorical_E}
If $X$ has a categorical distribution with parameters $(p_1, \ldots, p_k)$,
then $\E[X] = \sum_{i=1}^k i\, p_i$.
\end{proposition}

\begin{proof}
$\E[X] = \sum_{i=1}^k i \cdot \Prob(X = i) = \sum_{i=1}^k i\, p_i$.
\end{proof}

\begin{rlconnection}[title={Connection to RL}]
The categorical distribution models action selection under a stochastic
policy: $A_t \sim \pi(\cdot \mid s)$ is categorical with
$\Prob(A_t = a) = \pi(a \mid s)$. The policy gradient theorem computes
$\nabla_\theta \E[R]$ by differentiating through these categorical
probabilities.
\end{rlconnection}

\begin{definition}[Geometric Distribution]
\label{def:geometric}
$X \sim \operatorname{Geometric}(p)$ for $p \in (0,1]$ if
$\Prob(X = k) = (1-p)^{k-1} p$ for $k = 1, 2, 3, \ldots$.
\end{definition}

\begin{proposition}[Expectation of Geometric]
\label{prop:geometric_E}
If $X \sim \operatorname{Geometric}(p)$, then $\E[X] = 1/p$.
\end{proposition}

\begin{proof}
We compute:
\begin{align}
  \E[X] &= \sum_{k=1}^\infty k\, (1-p)^{k-1} p
  = p \sum_{k=1}^\infty k\, (1-p)^{k-1}.
  \label{eq:geometric_E_start}
\end{align}
Let $q = 1 - p$. We evaluate $\sum_{k=1}^\infty k q^{k-1}$ by
differentiating the geometric series:
\[
  \sum_{k=0}^\infty q^k = \frac{1}{1 - q}
  \quad \implies \quad
  \frac{d}{dq} \sum_{k=0}^\infty q^k
  = \sum_{k=1}^\infty k q^{k-1} = \frac{1}{(1-q)^2}.
\]
The differentiation is justified because the power series converges
uniformly on $[0, q_0]$ for any $q_0 < 1$, and $q = 1 - p \in [0, 1)$.
Substituting $1 - q = p$:
\[
  \E[X] = p \cdot \frac{1}{p^2} = \frac{1}{p}. \qedhere
\]
\end{proof}

\begin{remark}
The geometric distribution models the number of trials until the first
success. In RL, it arises in episode length analysis: if each step has
termination probability $p$, the episode length is
$\operatorname{Geometric}(p)$ with expected length $1/p$.
\end{remark}

\section{Joint and Marginal Distributions}
\label{sec:joint_marginal}

\begin{definition}[Joint Distribution]
\label{def:joint}
For random variables $X$ and $Y$ on $(\Omega, \mathcal{F}, \Prob)$, the
\emph{joint distribution} is the probability measure $\mu_{X,Y}$ on
$(\R^2, \Borel(\R^2))$ defined by
\begin{equation}
  \mu_{X,Y}(B) = \Prob\bigl((X, Y) \in B\bigr)
  \quad \text{for all } B \in \Borel(\R^2).
\end{equation}
In the discrete case, the \emph{joint PMF} is
$p_{X,Y}(x,y) = \Prob(X = x, Y = y)$.
\end{definition}

\begin{definition}[Marginal Distribution]
\label{def:marginal}
The \emph{marginal PMF} of $X$ is obtained by summing the joint PMF over
all values of $Y$:
\begin{equation}
  p_X(x) = \sum_y p_{X,Y}(x, y).
  \label{eq:marginal_def}
\end{equation}
\end{definition}

\begin{proposition}[Marginalization via Total Probability]
\label{prop:marginalization}
Let $X$ and $Y$ be discrete random variables. Then for every $x$ in the
range of $X$:
\begin{equation}
  p_X(x) = \sum_y p_{X,Y}(x, y).
\end{equation}
\end{proposition}

\begin{proof}
Let $Y$ take values in $\{y_1, y_2, \ldots\}$. The events
$B_j = \{Y = y_j\}$ are pairwise disjoint with $\bigcup_j B_j = \Omega$.
For $A = \{X = x\}$: $A = \bigsqcup_j (A \cap B_j)$, so by countable
additivity $p_X(x) = \sum_j \Prob(X = x, Y = y_j)
= \sum_j p_{X,Y}(x, y_j)$.
\end{proof}

\section{Basic Properties of Expectation}
\label{sec:expect_props}

\begin{proposition}[Monotonicity of Expectation]
\label{prop:monotonicity}
If $X \leq Y$ almost surely and both expectations exist, then
$\E[X] \leq \E[Y]$.
\end{proposition}

\begin{proof}
$Y - X \geq 0$ a.s., so $\E[Y - X] \geq 0$ (the integral of a
non-negative function is non-negative). By linearity:
$\E[Y] - \E[X] = \E[Y - X] \geq 0$.
\end{proof}

\begin{proposition}[Triangle Inequality for Expectation]
\label{prop:triangle}
If $\E[|X|] < \infty$, then $|\E[X]| \leq \E[|X|]$.
\end{proposition}

\begin{proof}
Since $-|X| \leq X \leq |X|$, monotonicity gives
$-\E[|X|] \leq \E[X] \leq \E[|X|]$, which is $|\E[X]| \leq \E[|X|]$.
\end{proof}

\begin{proposition}[Expectation of an Indicator]
\label{prop:expect_indicator}
For any event $A \in \mathcal{F}$, $\E[\indicator_A] = \Prob(A)$.
\end{proposition}

\begin{proof}
$\indicator_A$ is a simple function with standard representation
$1 \cdot \indicator_A + 0 \cdot \indicator_{A^c}$. By
Definition~\ref{def:integral_simple}:
$\E[\indicator_A] = 1 \cdot \Prob(A) + 0 \cdot \Prob(A^c) = \Prob(A)$.
\end{proof}

\section{Exercises}
\label{sec:exercises}

\begin{exercise}
\label{ex:total_prob_countable}
Extend the law of total probability
(Theorem~\ref{thm:total_prob}) to countable partitions: if
$B_1, B_2, \ldots$ is a countable partition of $\Omega$ with
$\Prob(B_i) > 0$, show that $\Prob(A) = \sum_{i=1}^\infty
\Prob(A \mid B_i) \Prob(B_i)$.
\end{exercise}

\begin{exercise}
\label{ex:bayes_three}
In a three-state MDP, the agent starts in state $s_1$, $s_2$, or $s_3$
with prior probabilities $0.5$, $0.3$, $0.2$. The probability of
receiving a positive reward ($R > 0$) from each state is $0.9$, $0.4$,
$0.1$ respectively. Given that a positive reward was observed, compute
the posterior probability of each starting state using Bayes' theorem.
\end{exercise}

\begin{exercise}
\label{ex:linearity_return}
Let $G_t = \sum_{k=0}^{T-t-1} \gamma^k R_{t+k+1}$ be the discounted
return in a finite-horizon MDP with horizon $T$ and discount factor
$\gamma \in (0,1)$. Using linearity of expectation, show that
$\E[G_t] = \sum_{k=0}^{T-t-1} \gamma^k \E[R_{t+k+1}]$.
\end{exercise}

\begin{exercise}
\label{ex:geometric_memoryless}
Prove the \emph{memoryless property} of the geometric distribution: if
$X \sim \operatorname{Geometric}(p)$, then for all $m, n \geq 1$:
\[
  \Prob(X > m + n \mid X > m) = \Prob(X > n).
\]
\textit{Hint:} Compute $\Prob(X > k) = (1-p)^k$ and use the definition of
conditional probability.
\end{exercise}

\begin{exercise}
\label{ex:cdf_determines_expectation}
Let $X$ be a non-negative integer-valued random variable with CDF $F_X$.
Show that $\E[X] = \sum_{k=0}^\infty \Prob(X > k)
= \sum_{k=0}^\infty (1 - F_X(k))$.
\textit{Hint:} Write $X = \sum_{k=0}^\infty \indicator_{\{X > k\}}$ and
apply linearity.
\end{exercise}

\begin{exercise}
\label{ex:indicator_variance_preview}
For $A \in \mathcal{F}$, compute $\E[\indicator_A^2]$ and use it to find
$\E[\indicator_A^2] - (\E[\indicator_A])^2$. The quantity you obtain is
the variance of a Bernoulli random variable (to be formally defined in
Lesson~7).
\end{exercise}

\begin{exercise}
\label{ex:joint_marginal_compute}
Let $(X, Y)$ have joint PMF $p_{X,Y}(0,0) = 1/4$, $p_{X,Y}(0,1) = 1/4$,
$p_{X,Y}(1,0) = 1/4$, $p_{X,Y}(1,1) = 1/4$.
Compute the marginal PMFs $p_X$ and $p_Y$, verify $\E[XY] = \E[X]\E[Y]$,
and compute $\E[X + 2Y]$ using linearity.
\end{exercise}
