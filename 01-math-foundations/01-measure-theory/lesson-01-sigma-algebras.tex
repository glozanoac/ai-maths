\section{Motivation and Overview}
\label{sec:motivation}

The convergence proofs of TD-learning, Q-learning, and policy gradient methods
rest on stochastic approximation theory, which in turn requires martingale
convergence theorems. To state these theorems precisely, three ingredients
are needed:
\begin{enumerate}[nosep]
  \item \textbf{$\sigma$-algebras} to model the information available at each
    time step (filtrations).
  \item \textbf{Measures} to assign probabilities to events in a mathematically
    consistent way.
  \item \textbf{Measurable functions} to formalize random variables and value
    functions.
\end{enumerate}

Without measure theory, the expression $\E[X \mid \mathcal{F}_t]$ is merely
notation. With it, conditional expectation becomes a well-defined random
variable characterized by two precise properties, and the entire convergence
machinery follows.

This lesson develops the first ingredient---$\sigma$-algebras---and
culminates in Dynkin's $\pi$-$\lambda$ theorem, the principal tool for
proving that two measures agree on an entire $\sigma$-algebra.

\begin{rlconnection}[title={Why This Matters for RL}]
The TD(0) convergence proof uses:
\begin{itemize}[nosep]
  \item The filtration $\mathcal{F}_t = \sigma(S_0, A_0, R_1, \ldots, S_t)$
    --- a $\sigma$-algebra generated by random variables.
  \item The noise term $w_k(s)$ satisfying
    $\E[w_k(s) \mid \mathcal{F}_k] = 0$ --- conditional expectation with
    respect to a $\sigma$-algebra.
  \item The Robbins--Siegmund theorem, which requires a
    \emph{non-negative supermartingale} --- defined via $\sigma$-algebras
    and conditional expectation.
\end{itemize}
This lesson provides the language for all three.
\end{rlconnection}

\section{$\sigma$-Algebras}
\label{sec:sigma_algebras}

Throughout, $\Omega$ denotes a non-empty set (the \emph{sample space}).

\begin{definition}[$\sigma$-Algebra]
\label{def:sigma_algebra}
A collection $\mathcal{F} \subseteq \powerset(\Omega)$ of subsets of
$\Omega$ is a \emph{$\sigma$-algebra} (or \emph{$\sigma$-field}) on
$\Omega$ if:
\begin{enumerate}[nosep,label=(\roman*)]
  \item $\Omega \in \mathcal{F}$.
  \item If $A \in \mathcal{F}$, then $A^c := \Omega \setminus A \in
    \mathcal{F}$.
    \hfill(closure under complement)
  \item If $A_1, A_2, \ldots \in \mathcal{F}$, then
    $\bigcup_{n=1}^\infty A_n \in \mathcal{F}$.
    \hfill(closure under countable union)
\end{enumerate}
The pair $(\Omega, \mathcal{F})$ is called a \emph{measurable space}.
Elements of $\mathcal{F}$ are called \emph{measurable sets} (or
\emph{events} in probability).
\end{definition}

\begin{remark}[Why Countable, Not Arbitrary?]
\label{rem:countable}
Axiom (iii) requires closure under \emph{countable} unions, not arbitrary
ones. Closure under arbitrary unions would force
$\mathcal{F} = \powerset(\Omega)$ in most interesting cases, which is too
large: the power set of $\R$ admits no translation-invariant measure
extending the notion of ``length.'' The restriction to countable
operations is precisely what makes non-trivial measures possible.
\end{remark}

\begin{proposition}[Basic Properties of $\sigma$-Algebras]
\label{prop:sigma_basic}
Let $\mathcal{F}$ be a $\sigma$-algebra on $\Omega$. Then:
\begin{enumerate}[nosep,label=(\alph*)]
  \item $\emptyset \in \mathcal{F}$.
  \item If $A_1, A_2, \ldots \in \mathcal{F}$, then
    $\bigcap_{n=1}^\infty A_n \in \mathcal{F}$.
    \hfill(closure under countable intersection)
  \item If $A, B \in \mathcal{F}$, then $A \setminus B \in \mathcal{F}$
    and $A \triangle B \in \mathcal{F}$.
  \item $\mathcal{F}$ is closed under finite unions and finite intersections.
\end{enumerate}
\end{proposition}

\begin{proof}
\textbf{(a)} $\emptyset = \Omega^c \in \mathcal{F}$ by axioms (i) and (ii).

\textbf{(b)} By De Morgan's law,
$\bigcap_{n=1}^\infty A_n = \bigl(\bigcup_{n=1}^\infty A_n^c\bigr)^c$.
Each $A_n^c \in \mathcal{F}$ by (ii), so $\bigcup A_n^c \in \mathcal{F}$
by (iii), and $(\bigcup A_n^c)^c \in \mathcal{F}$ by (ii).

\textbf{(c)} $A \setminus B = A \cap B^c$. Since $B^c \in \mathcal{F}$ by
(ii), we have $A \cap B^c \in \mathcal{F}$ by (b) applied to a two-element
sequence ($A, B^c, \Omega, \Omega, \ldots$). For symmetric difference:
$A \triangle B = (A \setminus B) \cup (B \setminus A)$.

\textbf{(d)} For a finite union $A_1 \cup \cdots \cup A_n$, set
$A_k = \emptyset$ for $k > n$ in axiom (iii). Finite intersections follow
from (b) similarly.
\end{proof}

\begin{example}[Extreme $\sigma$-Algebras]
\label{ex:extreme}
For any non-empty $\Omega$:
\begin{itemize}[nosep]
  \item The \emph{trivial $\sigma$-algebra}
    $\{\emptyset, \Omega\}$ is the smallest $\sigma$-algebra on $\Omega$.
  \item The \emph{power set} $\powerset(\Omega)$ is the largest.
\end{itemize}
Every $\sigma$-algebra satisfies
$\{\emptyset, \Omega\} \subseteq \mathcal{F} \subseteq \powerset(\Omega)$.
\end{example}

\begin{example}[Partition-Generated $\sigma$-Algebra]
\label{ex:partition}
Let $\Omega = \{1,2,3,4\}$ and consider the partition
$\Pi = \bigl\{\{1,2\}, \{3,4\}\bigr\}$. The $\sigma$-algebra generated by
this partition consists of all possible unions of blocks:
\begin{equation}
  \mathcal{F} = \bigl\{\emptyset,\;\{1,2\},\;\{3,4\},\;\{1,2,3,4\}\bigr\}.
\end{equation}
This has $2^2 = 4$ elements. In general, a partition of $\Omega$ into $k$
blocks generates a $\sigma$-algebra with exactly $2^k$ elements.

To verify the claim, note that if $\Pi = \{B_1, \ldots, B_k\}$ is a
partition, then the collection of all unions of sub-families of $\Pi$
(including the empty union $\emptyset$) contains $\Omega$, is closed
under complement (the complement of a union of blocks is the union of the
remaining blocks), and is closed under countable union (a countable union
of such sets is again a union of blocks). So it is a $\sigma$-algebra.
Any $\sigma$-algebra containing $\Pi$ must contain all unions of blocks,
so this is the smallest one.
\end{example}

\begin{rlconnection}[title={Partition $\sigma$-Algebras in RL}]
In a finite MDP with state space $\mathcal{S} = \{s_1, \ldots, s_n\}$,
the ``information contained in knowing $S_t$'' is modeled by
$\sigma(S_t) = \sigma\bigl(\{S_t = s_1\}, \ldots, \{S_t = s_n\}\bigr)$.
Since the events $\{S_t = s_i\}$ partition $\Omega$, this is a
partition-generated $\sigma$-algebra with $2^n$ events. A function $f$ is
$\sigma(S_t)$-measurable if and only if $f$ is constant on each block
$\{S_t = s_i\}$---that is, $f$ depends only on the current state.
\end{rlconnection}

\section{Generated $\sigma$-Algebras}
\label{sec:generated}

Given an arbitrary collection of subsets, we want the ``smallest''
$\sigma$-algebra containing them. This construction is fundamental
because we typically specify a few key events or random variables and
need the full $\sigma$-algebra they generate.

\begin{lemma}[Intersection of $\sigma$-Algebras]
\label{lem:intersection}
If $\{\mathcal{F}_\alpha\}_{\alpha \in I}$ is any (possibly uncountable)
family of $\sigma$-algebras on $\Omega$, then
$\bigcap_{\alpha \in I} \mathcal{F}_\alpha$ is also a $\sigma$-algebra
on $\Omega$.
\end{lemma}

\begin{proof}
We verify the three axioms.
\begin{enumerate}[nosep,label=(\roman*)]
  \item $\Omega \in \mathcal{F}_\alpha$ for all $\alpha$, so
    $\Omega \in \bigcap_\alpha \mathcal{F}_\alpha$.
  \item If $A \in \bigcap_\alpha \mathcal{F}_\alpha$, then
    $A \in \mathcal{F}_\alpha$ for every $\alpha$, so
    $A^c \in \mathcal{F}_\alpha$ for every $\alpha$, and thus
    $A^c \in \bigcap_\alpha \mathcal{F}_\alpha$.
  \item If $A_1, A_2, \ldots \in \bigcap_\alpha \mathcal{F}_\alpha$, then
    for each $\alpha$, all $A_n \in \mathcal{F}_\alpha$, so
    $\bigcup_n A_n \in \mathcal{F}_\alpha$. Hence
    $\bigcup_n A_n \in \bigcap_\alpha \mathcal{F}_\alpha$.
    \qedhere
\end{enumerate}
\end{proof}

\begin{definition}[Generated $\sigma$-Algebra]
\label{def:generated}
For any collection $\mathcal{C} \subseteq \powerset(\Omega)$, the
$\sigma$-algebra \emph{generated by} $\mathcal{C}$ is
\begin{equation}
  \sigma(\mathcal{C}) := \bigcap\bigl\{\mathcal{G} :
    \mathcal{G} \text{ is a $\sigma$-algebra on } \Omega,\;
    \mathcal{C} \subseteq \mathcal{G}\bigr\}.
  \label{eq:generated}
\end{equation}
\end{definition}

\begin{remark}
This is well defined by Lemma~\ref{lem:intersection}: the collection
being intersected is non-empty (it includes $\powerset(\Omega)$), and the
intersection of $\sigma$-algebras is a $\sigma$-algebra. By construction,
$\sigma(\mathcal{C})$ is the \emph{smallest} $\sigma$-algebra containing
$\mathcal{C}$: it contains $\mathcal{C}$, and any $\sigma$-algebra
containing $\mathcal{C}$ must contain $\sigma(\mathcal{C})$.
\end{remark}

\begin{definition}[Borel $\sigma$-Algebra]
\label{def:borel}
The \emph{Borel $\sigma$-algebra} on $\R$, denoted $\Borel(\R)$, is the
$\sigma$-algebra generated by the open subsets of $\R$:
\begin{equation}
  \Borel(\R) := \sigma\bigl(\{U \subseteq \R : U \text{ is open}\}\bigr).
  \label{eq:borel}
\end{equation}
Elements of $\Borel(\R)$ are called \emph{Borel sets}.
\end{definition}

\begin{proposition}[Equivalent Generators of $\Borel(\R)$]
\label{prop:borel_generators}
The following collections all generate $\Borel(\R)$:
\begin{enumerate}[nosep,label=(\alph*)]
  \item Open subsets of $\R$.
  \item Closed subsets of $\R$.
  \item Open intervals $(a, b)$ with $a, b \in \Q$.
  \item Half-lines $(-\infty, x]$ with $x \in \R$.
  \item Half-lines $(-\infty, q)$ with $q \in \Q$.
\end{enumerate}
\end{proposition}

\begin{proof}
We show mutual containment among the generated $\sigma$-algebras.

\textbf{(a) $\Leftrightarrow$ (b):} A set is closed if and only if its
complement is open. Since $\sigma$-algebras are closed under complement,
$\sigma(\text{open sets}) = \sigma(\text{closed sets})$.

\textbf{(c) $\Rightarrow$ (a):} Every open set $U \subseteq \R$ can be
written as a countable union of open intervals with rational endpoints.
Indeed, for each $x \in U$, the density of $\Q$ provides rationals
$a_x < x < b_x$ with $(a_x, b_x) \subseteq U$. Then
$U = \bigcup_{x \in U}(a_x, b_x)$, and since the set of pairs
$(a_x, b_x) \in \Q \times \Q$ is countable, this is a countable union.
Hence $\sigma(\text{rational open intervals}) \supseteq \text{open sets}$.

\textbf{(a) $\Rightarrow$ (c):} Every open interval with rational
endpoints is open, so it belongs to $\Borel(\R)$.

\textbf{(d) $\Rightarrow$ (a):} Observe that
$(-\infty, b) = \bigcup_{n=1}^\infty (-\infty, b - 1/n]$ and
$(a, b) = (-\infty, b) \setminus (-\infty, a]$. So open intervals belong
to $\sigma(\{(-\infty, x] : x \in \R\})$. By the implication
(c) $\Rightarrow$ (a), all open sets are obtained.

\textbf{(a) $\Rightarrow$ (d):} Since
$(-\infty, x] = \R \setminus (x, \infty)$ and $(x, \infty)$ is open,
each half-line $(-\infty, x]$ is a Borel set.

\textbf{(d) $\Leftrightarrow$ (e):} For the forward direction,
$(-\infty, x] = \bigcap_{n=1}^\infty (-\infty, q_n)$ where
$(q_n)_{n \geq 1}$ is a sequence in $\Q$ decreasing to $x$. For the
reverse, $(-\infty, q) = \bigcup_{n=1}^\infty (-\infty, q - 1/n]$.
\end{proof}

\begin{remark}
\label{rem:borel_cdf}
Generator (d) is particularly important: to verify that a function
$f\colon \Omega \to \R$ is $\mathcal{F}$/$\Borel(\R)$-measurable, it
suffices to check that
$f^{-1}\bigl((-\infty, x]\bigr) = \{\omega : f(\omega) \leq x\}
\in \mathcal{F}$ for every $x \in \R$. This is the standard criterion
for random variables.
\end{remark}

\section{$\pi$-Systems and $\lambda$-Systems}
\label{sec:pi_lambda}

The $\pi$-$\lambda$ theorem (Dynkin's theorem) is one of the most
powerful tools in measure theory. It reduces the verification of
identities on an entire $\sigma$-algebra to checking them on a much
smaller ``$\pi$-system.''

\begin{definition}[$\pi$-System]
\label{def:pi_system}
A collection $\mathcal{P} \subseteq \powerset(\Omega)$ is a
\emph{$\pi$-system} if it is closed under finite intersection:
$A, B \in \mathcal{P}$ implies $A \cap B \in \mathcal{P}$.
\end{definition}

\begin{definition}[$\lambda$-System (Dynkin System)]
\label{def:lambda_system}
A collection $\mathcal{L} \subseteq \powerset(\Omega)$ is a
\emph{$\lambda$-system} if:
\begin{enumerate}[nosep,label=(\roman*)]
  \item $\Omega \in \mathcal{L}$.
  \item If $A, B \in \mathcal{L}$ and $A \subseteq B$, then
    $B \setminus A \in \mathcal{L}$.
    \hfill(closure under proper difference)
  \item If $A_1 \subseteq A_2 \subseteq \cdots$ are in $\mathcal{L}$,
    then $\bigcup_{n=1}^\infty A_n \in \mathcal{L}$.
    \hfill(closure under increasing union)
\end{enumerate}
\end{definition}

\begin{remark}
Every $\sigma$-algebra is both a $\pi$-system and a $\lambda$-system.
The converse also holds: a collection that is simultaneously a
$\pi$-system and a $\lambda$-system must be a $\sigma$-algebra. This
fact is the key lemma enabling the $\pi$-$\lambda$ theorem.
\end{remark}

\begin{lemma}[$\pi + \lambda = \sigma$]
\label{lem:pi_lambda_sigma}
If $\mathcal{F}$ is both a $\pi$-system and a $\lambda$-system, then
$\mathcal{F}$ is a $\sigma$-algebra.
\end{lemma}

\begin{proof}
We verify the three $\sigma$-algebra axioms.

\textit{Axiom (i):} $\Omega \in \mathcal{F}$ by the $\lambda$-system
property.

\textit{Axiom (ii):} If $A \in \mathcal{F}$, then $A \subseteq \Omega$
with both in $\mathcal{F}$, so $A^c = \Omega \setminus A \in \mathcal{F}$
by the proper difference property.

\textit{Axiom (iii):} Let $A_1, A_2, \ldots \in \mathcal{F}$. Define
$B_n = \bigcup_{k=1}^n A_k$. We show $B_n \in \mathcal{F}$ by
induction. For $n = 1$: $B_1 = A_1 \in \mathcal{F}$. For the
inductive step, assuming $B_n \in \mathcal{F}$:
\begin{equation}
  B_n \cup A_{n+1} = (B_n^c \cap A_{n+1}^c)^c.
\end{equation}
Since $\mathcal{F}$ is closed under complement (just proved) and
finite intersection ($\pi$-system), $B_n^c \cap A_{n+1}^c \in \mathcal{F}$,
so $(B_n^c \cap A_{n+1}^c)^c \in \mathcal{F}$.

The sequence $(B_n)$ is increasing and each $B_n \in \mathcal{F}$, so
$\bigcup_{n=1}^\infty A_n = \bigcup_{n=1}^\infty B_n \in \mathcal{F}$
by the $\lambda$-system closure under increasing unions.
\end{proof}

\begin{keyresult}[title={Dynkin's $\pi$-$\lambda$ Theorem}]
\begin{theorem}[$\pi$-$\lambda$ Theorem (Dynkin)]
\label{thm:pi_lambda}
If $\mathcal{P}$ is a $\pi$-system and $\mathcal{L}$ is a
$\lambda$-system with $\mathcal{P} \subseteq \mathcal{L}$, then
\begin{equation}
  \sigma(\mathcal{P}) \subseteq \mathcal{L}.
\end{equation}
\end{theorem}
\end{keyresult}

\begin{proof}
Let $\ell(\mathcal{P})$ denote the smallest $\lambda$-system containing
$\mathcal{P}$. This exists by the same intersection argument used for
generated $\sigma$-algebras: $\powerset(\Omega)$ is a $\lambda$-system,
and the intersection of any family of $\lambda$-systems is a
$\lambda$-system. Since $\mathcal{L}$ is a $\lambda$-system containing
$\mathcal{P}$, we have $\ell(\mathcal{P}) \subseteq \mathcal{L}$.

It suffices to show that $\ell(\mathcal{P})$ is a $\sigma$-algebra, for
then $\sigma(\mathcal{P}) \subseteq \ell(\mathcal{P}) \subseteq
\mathcal{L}$. By Lemma~\ref{lem:pi_lambda_sigma}, it is enough to prove
that $\ell(\mathcal{P})$ is a $\pi$-system, i.e., closed under finite
intersection.

\textbf{Step 1.} For each $B \in \ell(\mathcal{P})$, define
\begin{equation}
  \mathcal{G}_B = \{A \in \ell(\mathcal{P}) :
    A \cap B \in \ell(\mathcal{P})\}.
\end{equation}
We claim $\mathcal{G}_B$ is a $\lambda$-system:
\begin{itemize}[nosep]
  \item $\Omega \in \mathcal{G}_B$ since
    $\Omega \cap B = B \in \ell(\mathcal{P})$.
  \item If $A_1 \subseteq A_2$ with $A_1, A_2 \in \mathcal{G}_B$, then
    $(A_2 \setminus A_1) \cap B = (A_2 \cap B) \setminus (A_1 \cap B)$.
    Since $A_1 \cap B \subseteq A_2 \cap B$ and both are in
    $\ell(\mathcal{P})$, the proper difference property gives
    $(A_2 \setminus A_1) \cap B \in \ell(\mathcal{P})$.
  \item If $A_n \uparrow A$ with each $A_n \in \mathcal{G}_B$, then
    $A_n \cap B \uparrow A \cap B$, so
    $A \cap B = \bigcup_n (A_n \cap B) \in \ell(\mathcal{P})$.
\end{itemize}

\textbf{Step 2.} For any $B \in \mathcal{P}$: since $\mathcal{P}$ is a
$\pi$-system, $A \cap B \in \mathcal{P} \subseteq \ell(\mathcal{P})$
for all $A \in \mathcal{P}$. So $\mathcal{P} \subseteq \mathcal{G}_B$.
Since $\mathcal{G}_B$ is a $\lambda$-system containing $\mathcal{P}$
and $\ell(\mathcal{P})$ is the smallest such, we conclude
$\ell(\mathcal{P}) \subseteq \mathcal{G}_B$. This means:
\begin{equation}
  \text{for all } B \in \mathcal{P} \text{ and }
  A \in \ell(\mathcal{P}),\quad
  A \cap B \in \ell(\mathcal{P}).
  \label{eq:step2}
\end{equation}

\textbf{Step 3.} Now fix any $A \in \ell(\mathcal{P})$ and consider
$\mathcal{G}_A = \{B \in \ell(\mathcal{P}) :
A \cap B \in \ell(\mathcal{P})\}$. By~\eqref{eq:step2},
$\mathcal{P} \subseteq \mathcal{G}_A$ (for any $B \in \mathcal{P}$,
$A \cap B \in \ell(\mathcal{P})$). By Step~1, $\mathcal{G}_A$ is a
$\lambda$-system, so $\ell(\mathcal{P}) \subseteq \mathcal{G}_A$.

This means: for all $A, B \in \ell(\mathcal{P})$,
$A \cap B \in \ell(\mathcal{P})$. So $\ell(\mathcal{P})$ is a
$\pi$-system. By Lemma~\ref{lem:pi_lambda_sigma},
$\ell(\mathcal{P})$ is a $\sigma$-algebra, and therefore
$\sigma(\mathcal{P}) \subseteq \ell(\mathcal{P}) \subseteq \mathcal{L}$.
\end{proof}

\section{Uniqueness of Measures}
\label{sec:uniqueness}

The principal application of the $\pi$-$\lambda$ theorem is that a
probability measure is uniquely determined by its values on any
generating $\pi$-system.

\begin{corollary}[Uniqueness of Probability Measures]
\label{cor:uniqueness}
Let $\mathcal{P}$ be a $\pi$-system on $\Omega$. If two probability
measures $\Prob_1$ and $\Prob_2$ on
$(\Omega, \sigma(\mathcal{P}))$ satisfy
$\Prob_1(A) = \Prob_2(A)$ for all $A \in \mathcal{P}$, then
$\Prob_1 = \Prob_2$ on all of $\sigma(\mathcal{P})$.
\end{corollary}

\begin{proof}
Define the collection
\begin{equation}
  \mathcal{L} = \{A \in \sigma(\mathcal{P}) :
    \Prob_1(A) = \Prob_2(A)\}.
\end{equation}
By hypothesis, $\mathcal{P} \subseteq \mathcal{L}$. We verify that
$\mathcal{L}$ is a $\lambda$-system:
\begin{itemize}[nosep]
  \item $\Omega \in \mathcal{L}$ since
    $\Prob_1(\Omega) = \Prob_2(\Omega) = 1$.
  \item If $A \subseteq B$ with $A, B \in \mathcal{L}$, then
    $\Prob_i(B \setminus A) = \Prob_i(B) - \Prob_i(A)$ for $i = 1, 2$
    (using $\Prob_i(A) \leq \Prob_i(B) \leq 1 < \infty$). Since
    $\Prob_1(B) = \Prob_2(B)$ and $\Prob_1(A) = \Prob_2(A)$, the
    differences are equal, so $B \setminus A \in \mathcal{L}$.
  \item If $A_n \uparrow A$ with each $A_n \in \mathcal{L}$, then
    $\Prob_i(A) = \lim_{n} \Prob_i(A_n)$ by continuity from below.
    Since $\Prob_1(A_n) = \Prob_2(A_n)$ for each $n$, the limits agree:
    $A \in \mathcal{L}$.
\end{itemize}
By the $\pi$-$\lambda$ theorem (Theorem~\ref{thm:pi_lambda}),
$\sigma(\mathcal{P}) \subseteq \mathcal{L}$, which means
$\Prob_1 = \Prob_2$ on $\sigma(\mathcal{P})$.
\end{proof}

\begin{warning}[title={The $\pi$-System Hypothesis is Essential}]
The corollary fails if $\mathcal{P}$ is not a $\pi$-system. For
example, on $\Omega = \{1, 2, 3, 4\}$, let
$\mathcal{C} = \bigl\{\{1,2\}, \{2,3\}\bigr\}$. This is not a
$\pi$-system since $\{1,2\} \cap \{2,3\} = \{2\} \notin \mathcal{C}$.
One can construct two distinct probability measures on
$\sigma(\mathcal{C})$ that agree on $\mathcal{C}$ but disagree on
$\{2\} \in \sigma(\mathcal{C})$.
\end{warning}

\begin{rlconnection}[title={CDFs Determine Distributions}]
Corollary~\ref{cor:uniqueness} explains why the cumulative distribution
function $F_X(x) = \Prob(X \leq x)$ uniquely determines the
distribution of a random variable $X$. The half-lines
$\{(-\infty, x] : x \in \R\}$ form a $\pi$-system (since
$(-\infty, x] \cap (-\infty, y] = (-\infty, \min(x,y)]$)
that generates $\Borel(\R)$
(Proposition~\ref{prop:borel_generators}(d)). So any two measures
agreeing on all half-lines must agree on all Borel sets. In RL, this
means the distribution of the return $G_t = \sum_{k=0}^\infty
\gamma^k R_{t+k+1}$ is fully characterized by its CDF---there is no
``hidden'' probabilistic information beyond what the CDF captures.
\end{rlconnection}

\begin{remark}[Extension to $\sigma$-Finite Measures]
\label{rem:sigma_finite}
Corollary~\ref{cor:uniqueness} extends to $\sigma$-finite measures:
if $\mu_1$ and $\mu_2$ are measures on $(\Omega, \sigma(\mathcal{P}))$
that agree on a $\pi$-system $\mathcal{P}$, and there exist sets
$\Omega_1 \subseteq \Omega_2 \subseteq \cdots$ in $\mathcal{P}$ with
$\bigcup_n \Omega_n = \Omega$ and $\mu_1(\Omega_n) = \mu_2(\Omega_n)
< \infty$ for each $n$, then $\mu_1 = \mu_2$ on
$\sigma(\mathcal{P})$. The proof applies the probability-measure
version to the normalized restrictions
$\mu_i^{(n)}(\cdot) = \mu_i(\cdot \cap \Omega_n) / \mu_i(\Omega_n)$.
\end{remark}

\section{Exercises}
\label{sec:exercises}

\begin{exercise}[Countable-Cocountable $\sigma$-Algebra]
\label{ex:countable_cocountable}
Let $\Omega$ be an uncountable set. Define
\begin{equation}
  \mathcal{F} = \{A \subseteq \Omega : A \text{ is countable or }
    A^c \text{ is countable}\}.
\end{equation}
\begin{enumerate}[label=(\alph*)]
  \item Prove that $\mathcal{F}$ is a $\sigma$-algebra on $\Omega$.
  \item Show that $\mathcal{F}$ is strictly between the trivial
    $\sigma$-algebra and $\powerset(\Omega)$.
  \item Define $\mu(A) = 0$ if $A$ is countable and $\mu(A) = 1$ if $A^c$
    is countable. Prove that $\mu$ is a probability measure on
    $(\Omega, \mathcal{F})$.
\end{enumerate}
\textit{Hint for (a):} A countable union of countable sets is countable.
For closure under complement, consider both cases.
\end{exercise}

\begin{exercise}[$\pi$-Systems and $\lambda$-Systems]
\label{ex:pi_lambda_check}
Determine which of the following collections on $\Omega = \{1,2,3,4\}$
are $\pi$-systems, $\lambda$-systems, or $\sigma$-algebras:
\begin{enumerate}[label=(\alph*)]
  \item $\mathcal{C}_1 = \bigl\{\{1,2\},\; \{2,3\},\; \{2\}\bigr\}$.
  \item $\mathcal{C}_2 = \bigl\{\Omega,\; \emptyset,\;
    \{1\},\; \{2,3,4\}\bigr\}$.
  \item $\mathcal{C}_3 = \bigl\{\Omega,\; \emptyset,\;
    \{1,2\},\; \{3,4\},\; \{1,3\},\; \{2,4\}\bigr\}$.
\end{enumerate}
For each, give a complete justification.
\end{exercise}

\begin{exercise}[Uniqueness Failure Without the $\pi$-System Property]
\label{ex:uniqueness_failure}
Let $\Omega = \{1, 2, 3\}$ and
$\mathcal{C} = \bigl\{\{1,2\},\; \{2,3\}\bigr\}$.
\begin{enumerate}[label=(\alph*)]
  \item Compute $\sigma(\mathcal{C})$ explicitly.
  \item Construct two distinct probability measures on
    $(\Omega, \sigma(\mathcal{C}))$ that agree on $\mathcal{C}$.
  \item Explain why this does not contradict
    Corollary~\ref{cor:uniqueness}.
\end{enumerate}
\end{exercise}

\begin{exercise}[Application to RL: Filtration Verification]
\label{ex:filtration}
Consider a Markov chain $(X_n)_{n \geq 0}$ on a finite state space
$\mathcal{S} = \{s_1, s_2, s_3\}$ with sample space
$\Omega = \mathcal{S}^{\N}$ (the set of all infinite sequences).
Define $\mathcal{F}_n = \sigma(X_0, X_1, \ldots, X_n)$.
\begin{enumerate}[label=(\alph*)]
  \item Show that $\mathcal{F}_0 \subseteq \mathcal{F}_1 \subseteq
    \mathcal{F}_2 \subseteq \cdots$ (i.e., $(\mathcal{F}_n)$ is a
    filtration).
  \item Show that
    $\mathcal{P}_n = \bigl\{\{X_0 = a_0, \ldots, X_n = a_n\} :
    a_i \in \mathcal{S}\bigr\}$ is a $\pi$-system.
  \item Using Corollary~\ref{cor:uniqueness}, conclude that the
    distribution of $(X_0, \ldots, X_n)$ is uniquely determined by
    the probabilities $\Prob(X_0 = a_0, \ldots, X_n = a_n)$.
\end{enumerate}
\end{exercise}

\begin{exercise}[Borel Sets]
\label{ex:borel_sets}
Show that each of the following subsets of $\R$ is a Borel set by
expressing it using countable set operations on open or closed sets:
\begin{enumerate}[label=(\alph*)]
  \item The set of rationals $\Q$.
  \item The set of irrationals $\R \setminus \Q$.
  \item The Cantor set $C = \bigcap_{n=0}^\infty C_n$, where $C_0 = [0,1]$
    and $C_{n+1}$ is obtained from $C_n$ by removing the open middle
    third of each interval.
\end{enumerate}
\end{exercise}
