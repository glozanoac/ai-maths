\section{Motivation and Overview}
\label{sec:motivation}

The value function in reinforcement learning is defined as
$V^\pi(s) = \E_\pi[G_t \mid S_t = s]$, a conditional expectation.
Every Bellman equation---the recursive backbone of dynamic
programming and temporal-difference learning---is derived by
splitting a conditional expectation across two successive time
steps.  The algebraic engine behind this splitting is the
\emph{tower property} (law of total expectation):
\[
  \E[X] = \E\bigl[\E[X \mid Y]\bigr].
\]
This lesson develops conditional expectation rigorously,
establishes its main algebraic properties (linearity, pull-out,
monotonicity, tower), and shows how the tower property,
combined with the Markov property of MDPs, produces the Bellman
expectation equation.

\begin{rlconnection}[title={Why This Matters for RL}]
The tower property is the single most-used probabilistic tool in
RL theory.  It enables:
\begin{itemize}[nosep]
  \item The derivation of the Bellman equation from the return
    definition.
  \item The proof that the TD error $\delta_t^\pi$ has zero
    conditional mean (a martingale difference property).
  \item The expansion of multi-step expectations into single-step
    sums over actions and next states.
\end{itemize}
\end{rlconnection}

\medskip
\noindent\textbf{Prerequisites.}
Lesson~4 (probability spaces) and Lesson~5 (random variables,
expectation, linearity of expectation, conditional probability).
We assume throughout that $(\Omega, \mathcal{F}, \Prob)$ is a
probability space and all random variables are discrete unless
otherwise stated.

\section{Conditional Expectation Given an Event}
\label{sec:cond_event}

\begin{definition}[Conditional Expectation Given an Event]
\label{def:cond_exp_event}
Let $X$ be a random variable with $\E[|X|] < \infty$ and let
$B \in \mathcal{F}$ with $\Prob(B) > 0$.  The
\emph{conditional expectation of $X$ given $B$} is the number
\begin{equation}
  \E[X \mid B]
  = \frac{\E[X \cdot \indicator_B]}{\Prob(B)}
  = \sum_x x\, \Prob(X = x \mid B)
  = \sum_x x\, \frac{\Prob(X = x,\, B)}{\Prob(B)}.
  \label{eq:cond_exp_event}
\end{equation}
\end{definition}

\begin{remark}
$\E[X \mid B]$ is a \emph{number}, not a random variable.
It answers the question: ``What is the average value of $X$
among outcomes in $B$?''
\end{remark}

\begin{example}
\label{ex:die_event}
Let $X$ be the outcome of a fair six-sided die and
$B = \{X \text{ is even}\} = \{2, 4, 6\}$.  Then $\Prob(B) = 1/2$
and
\[
  \E[X \mid B]
  = \frac{2 \cdot \tfrac{1}{6} + 4 \cdot \tfrac{1}{6}
          + 6 \cdot \tfrac{1}{6}}{1/2}
  = \frac{12/6}{1/2} = 4.
\]
\end{example}

\begin{proposition}[Conditional Expectation via Total Probability]
\label{prop:total_exp_events}
If $B_1, B_2, \ldots, B_n$ partition $\Omega$ with
$\Prob(B_i) > 0$, then
\begin{equation}
  \E[X] = \sum_{i=1}^n \E[X \mid B_i]\, \Prob(B_i).
  \label{eq:total_exp_events}
\end{equation}
\end{proposition}

\begin{proof}
Expanding each conditional expectation:
\begin{align}
  \sum_{i=1}^n \E[X \mid B_i]\, \Prob(B_i)
  &= \sum_{i=1}^n \left(\sum_x x\,
    \frac{\Prob(X = x, B_i)}{\Prob(B_i)}\right) \Prob(B_i)
  \nonumber\\
  &= \sum_{i=1}^n \sum_x x\, \Prob(X = x, B_i)
  = \sum_x x \sum_{i=1}^n \Prob(X = x, B_i).
  \label{eq:total_exp_step}
\end{align}
Since $\{B_i\}$ partitions $\Omega$,
$\sum_i \Prob(X = x, B_i) = \Prob(X = x)$, giving
$\sum_x x\, \Prob(X = x) = \E[X]$.
\end{proof}

\section{Conditional Expectation Given a Random Variable}
\label{sec:cond_rv}

\begin{definition}[Conditional Expectation Given a Random Variable]
\label{def:cond_exp_rv}
For discrete random variables $X$ and $Y$ with
$\E[|X|] < \infty$, the \emph{conditional expectation of $X$
given $Y$} is the random variable $\E[X \mid Y]$ defined by
\begin{equation}
  \E[X \mid Y](\omega) = \E[X \mid Y = Y(\omega)]
  = \sum_x x\, \Prob(X = x \mid Y = Y(\omega)).
  \label{eq:cond_exp_rv}
\end{equation}
Equivalently, $\E[X \mid Y] = g(Y)$ where
$g(y) = \E[X \mid Y = y]$ for each $y$ in the range of $Y$.
\end{definition}

\begin{remark}[Key conceptual point]
Unlike $\E[X \mid B]$ (a number), $\E[X \mid Y]$ is a
\emph{random variable}: its value depends on the outcome
$\omega$ through $Y(\omega)$.  It is always a function of $Y$,
hence $\sigma(Y)$-measurable.
\end{remark}

\begin{definition}[Conditional Expectation Given a $\sigma$-Algebra]
\label{def:cond_exp_sigma}
More generally, if $\mathcal{G} \subseteq \mathcal{F}$ is a
sub-$\sigma$-algebra, the \emph{conditional expectation
$\E[X \mid \mathcal{G}]$} is the unique (a.s.)
$\mathcal{G}$-measurable random variable satisfying
\begin{equation}
  \int_G \E[X \mid \mathcal{G}]\, d\Prob
  = \int_G X\, d\Prob
  \quad \text{for all } G \in \mathcal{G}.
  \label{eq:cond_exp_sigma}
\end{equation}
Existence and uniqueness (up to a.s.\ equality) follow from the
Radon--Nikodym theorem \citep{durrett2019, billingsley1995}.
When $\mathcal{G} = \sigma(Y)$, this recovers
Definition~\ref{def:cond_exp_rv}.
\end{definition}

\begin{rlconnection}[title={Connection to RL}]
The value function $V^\pi(s) = \E_\pi[G_t \mid S_t = s]$ is
$\E_\pi[G_t \mid S_t]$ evaluated at $S_t = s$.
Equivalently, $V^\pi(S_t) = \E_\pi[G_t \mid S_t]$ is the random
variable that assigns to each trajectory the expected return from
the state visited at time $t$.  The fact that $V^\pi(S_t)$ is
$\sigma(S_t)$-measurable is precisely the statement that it is a
function of $S_t$ alone.
\end{rlconnection}

\section{Properties of Conditional Expectation}
\label{sec:properties}

Throughout this section, $X$, $Y$, $Z$ are discrete random
variables with $\E[|X|], \E[|Y|] < \infty$.

\subsection{Linearity}
\label{subsec:linearity}

\begin{theorem}[Linearity of Conditional Expectation]
\label{thm:cond_linearity}
For constants $a, b \in \R$:
\begin{equation}
  \E[aX + bY \mid Z] = a\,\E[X \mid Z] + b\,\E[Y \mid Z].
  \label{eq:cond_linearity}
\end{equation}
\end{theorem}

\begin{proof}
Fix $Z = z$ with $\Prob(Z = z) > 0$.  Expanding by definition:
\begin{align}
  \E[aX + bY \mid Z = z]
  &= \sum_{x,y} (ax + by)\, \Prob(X = x, Y = y \mid Z = z)
  \nonumber\\
  &= a \sum_{x,y} x\, \Prob(X = x, Y = y \mid Z = z)
    + b \sum_{x,y} y\, \Prob(X = x, Y = y \mid Z = z).
  \label{eq:lin_step1}
\end{align}
We marginalize the first sum over $y$:
\begin{equation}
  \sum_{x,y} x\, \Prob(X = x, Y = y \mid Z = z)
  = \sum_x x \sum_y \Prob(X = x, Y = y \mid Z = z)
  = \sum_x x\, \Prob(X = x \mid Z = z)
  = \E[X \mid Z = z].
  \label{eq:lin_marginalize}
\end{equation}
Similarly, the second sum equals $\E[Y \mid Z = z]$.  Since the
identity holds for every $z$, we have
$\E[aX + bY \mid Z] = a\,\E[X \mid Z] + b\,\E[Y \mid Z]$ as
random variables.
\end{proof}

\subsection{Pull-Out Property}
\label{subsec:pullout}

\begin{keyresult}[title={Pull-Out Property}]
\begin{theorem}[Pull-Out Property]
\label{thm:pull_out}
If $h(Z)$ is a function of $Z$ (i.e., $\sigma(Z)$-measurable),
then
\begin{equation}
  \E[h(Z) \cdot X \mid Z] = h(Z) \cdot \E[X \mid Z].
  \label{eq:pull_out}
\end{equation}
\end{theorem}
\end{keyresult}

\begin{proof}
Fix $Z = z$.  Then $h(Z) = h(z)$ is a constant under this
conditioning:
\begin{align}
  \E[h(Z) \cdot X \mid Z = z]
  &= \sum_x h(z) \cdot x \cdot \Prob(X = x \mid Z = z)
  \nonumber\\
  &= h(z) \sum_x x\, \Prob(X = x \mid Z = z)
  = h(z) \cdot \E[X \mid Z = z].
  \label{eq:pull_out_proof}
\end{align}
Since this holds for every $z$, the identity holds as random
variables.
\end{proof}

\begin{rlconnection}[title={Connection to RL}]
When proving that the TD error has zero conditional mean, we
condition on $S_t = s$ and pull out $V(S_t) = V(s)$:
\[
  \E[\delta_t \mid S_t = s]
  = \E[R_{t+1} + \gamma V(S_{t+1}) \mid S_t = s] - V(s).
\]
The term $V(s)$ exits the expectation because it is fully
determined by $S_t$.  This is the pull-out property with
$h(S_t) = V(S_t)$ and $Z = S_t$.
\end{rlconnection}

\subsection{Monotonicity}
\label{subsec:monotonicity}

\begin{proposition}[Monotonicity of Conditional Expectation]
\label{prop:cond_mono}
If $X \leq Y$ almost surely, then
$\E[X \mid Z] \leq \E[Y \mid Z]$ almost surely.
\end{proposition}

\begin{proof}
Fix $Z = z$.  Since $X \leq Y$ a.s., we have
$\Prob(X \leq Y \mid Z = z) = 1$ (because
$\Prob(X > Y) = 0$ implies $\Prob(X > Y, Z = z) = 0$).
Therefore:
\begin{equation}
  \E[Y \mid Z = z] - \E[X \mid Z = z]
  = \E[Y - X \mid Z = z] \geq 0,
\end{equation}
where the equality uses linearity and the inequality uses that
$Y - X \geq 0$ a.s.\ conditionally.
\end{proof}

\subsection{Conditional Expectation of Measurable Functions}
\label{subsec:measurable}

\begin{proposition}[Conditioning on Known Information]
\label{prop:cond_known}
If $X$ is $\sigma(Z)$-measurable (i.e., $X = g(Z)$ for some
function $g$), then
$\E[X \mid Z] = X$ almost surely.
\end{proposition}

\begin{proof}
If $X = g(Z)$, then fixing $Z = z$ gives $X = g(z)$, which is
constant.  Hence $\E[X \mid Z = z] = g(z) = X$ on the event
$\{Z = z\}$.
\end{proof}

\begin{corollary}
\label{cor:cond_constant}
If $c$ is a constant, $\E[c \mid Z] = c$.
\end{corollary}

\section{The Tower Property (Law of Total Expectation)}
\label{sec:tower}

\begin{keyresult}[title={Law of Total Expectation (Tower Property)}]
\begin{theorem}[Law of Total Expectation]
\label{thm:tower}
For discrete random variables $X$ and $Y$ with
$\E[|X|] < \infty$:
\begin{equation}
  \E[X] = \E\!\left[\E[X \mid Y]\right]
  = \sum_y \E[X \mid Y = y]\, \Prob(Y = y).
  \label{eq:tower}
\end{equation}
\end{theorem}
\end{keyresult}

\begin{proof}
We expand the right-hand side and show it equals $\E[X]$.

\textbf{Step 1: Expand the inner conditional expectation.}
\begin{align}
  \sum_y \E[X \mid Y = y]\, \Prob(Y = y)
  &= \sum_y \left(\sum_x x\,
    \Prob(X = x \mid Y = y)\right) \Prob(Y = y).
  \label{eq:tower_step1}
\end{align}

\textbf{Step 2: Use the definition of conditional probability.}
\begin{align}
  &= \sum_y \sum_x x\,
    \frac{\Prob(X = x, Y = y)}{\Prob(Y = y)} \cdot \Prob(Y = y)
  = \sum_y \sum_x x\, \Prob(X = x, Y = y).
  \label{eq:tower_step2}
\end{align}

\textbf{Step 3: Interchange sums and marginalize.}
\begin{align}
  &= \sum_x x \sum_y \Prob(X = x, Y = y)
  = \sum_x x\, \Prob(X = x) = \E[X].
  \label{eq:tower_step3}
\end{align}
The interchange of summation is justified by absolute convergence
(since $\E[|X|] < \infty$), and the marginalization uses
$\sum_y \Prob(X = x, Y = y) = \Prob(X = x)$.
\end{proof}

\begin{remark}[Interpretation]
The tower property says: to compute $\E[X]$, we may first
compute $\E[X \mid Y]$ (which is a random variable, a function
of $Y$) and then average over $Y$.  The ``tower'' name comes
from thinking of $\E\bigl[\E[\cdot \mid \cdot]\bigr]$ as a
stack of two operations.
\end{remark}

\begin{theorem}[Generalized Tower Property]
\label{thm:tower_general}
If $\mathcal{G} \subseteq \mathcal{H} \subseteq \mathcal{F}$
are sub-$\sigma$-algebras, then
\begin{equation}
  \E\!\left[\E[X \mid \mathcal{H}]
    \;\middle|\; \mathcal{G}\right]
  = \E[X \mid \mathcal{G}].
  \label{eq:tower_general}
\end{equation}
In particular, for random variables with
$\sigma(Z) \subseteq \sigma(Y, Z)$:
\begin{equation}
  \E\!\left[\E[X \mid Y, Z] \;\middle|\; Z\right]
  = \E[X \mid Z].
  \label{eq:tower_yz}
\end{equation}
\end{theorem}

\begin{proof}
We prove~\eqref{eq:tower_yz} in the discrete case.  Fix $Z = z$
with $\Prob(Z = z) > 0$:
\begin{align}
  \E\!\left[\E[X \mid Y, Z] \;\middle|\; Z = z\right]
  &= \sum_y \E[X \mid Y = y, Z = z]\, \Prob(Y = y \mid Z = z)
  \nonumber\\
  &= \sum_y \left(\sum_x x\,
    \Prob(X = x \mid Y = y, Z = z)\right)
    \Prob(Y = y \mid Z = z)
  \nonumber\\
  &= \sum_x x \sum_y
    \Prob(X = x \mid Y = y, Z = z)\, \Prob(Y = y \mid Z = z).
  \label{eq:tower_gen_step1}
\end{align}
Applying the chain rule to the inner product:
\begin{align}
  \Prob(X = x \mid Y = y, Z = z)\, \Prob(Y = y \mid Z = z)
  &= \frac{\Prob(X = x, Y = y, Z = z)}{\Prob(Y = y, Z = z)}
    \cdot \frac{\Prob(Y = y, Z = z)}{\Prob(Z = z)}
  \nonumber\\
  &= \frac{\Prob(X = x, Y = y, Z = z)}{\Prob(Z = z)}.
  \label{eq:tower_gen_chain}
\end{align}
Summing over $y$:
\begin{equation}
  \sum_y \frac{\Prob(X = x, Y = y, Z = z)}{\Prob(Z = z)}
  = \frac{\Prob(X = x, Z = z)}{\Prob(Z = z)}
  = \Prob(X = x \mid Z = z).
  \label{eq:tower_gen_marginalize}
\end{equation}
Substituting back:
$\E[\E[X \mid Y, Z] \mid Z = z]
= \sum_x x\, \Prob(X = x \mid Z = z)
= \E[X \mid Z = z]$.
\end{proof}

\begin{remark}
The generalized tower property says: if we first condition on
``more information'' ($\mathcal{H}$) and then average back to
``less information'' ($\mathcal{G} \subseteq \mathcal{H}$), the
result is the same as conditioning on the less informative
$\sigma$-algebra directly.  Conditioning on more information and
then averaging away the excess is the same as never having
conditioned on the excess.
\end{remark}

\section{MDP Application: Deriving the Bellman Equation}
\label{sec:bellman}

We now apply the tower property and MDP expansion to derive the
Bellman expectation equation, the central recursive equation of
reinforcement learning.

\subsection{MDP Expectation Expansion}
\label{subsec:mdp_expansion}

\begin{theorem}[MDP Expectation Expansion]
\label{thm:mdp_expansion}
Let $(\mathcal{S}, \mathcal{A}, P, r, \gamma)$ be a finite MDP
with policy $\pi$.  For any function
$f: \mathcal{A} \times \mathcal{S} \times \R \to \R$:
\begin{equation}
  \E_\pi[f(A_t, S_{t+1}, R_{t+1}) \mid S_t = s]
  = \sum_{a \in \mathcal{A}} \pi(a \mid s)
    \sum_{s' \in \mathcal{S}} P(s' \mid s, a)\,
    f(a, s', r(s,a,s')).
  \label{eq:mdp_expansion}
\end{equation}
\end{theorem}

\begin{proof}
By the definition of conditional expectation:
\begin{align}
  &\E_\pi[f(A_t, S_{t+1}, R_{t+1}) \mid S_t = s]
  \nonumber\\
  &\quad= \sum_{a \in \mathcal{A}} \sum_{s' \in \mathcal{S}}
    f(a, s', r(s,a,s'))\,
    \Prob_\pi(A_t = a, S_{t+1} = s' \mid S_t = s).
  \label{eq:expand_step1}
\end{align}
By the chain rule for conditional probability and the MDP
dynamics:
\begin{align}
  \Prob_\pi(A_t = a, S_{t+1} = s' \mid S_t = s)
  &= \Prob_\pi(A_t = a \mid S_t = s)
    \cdot \Prob(S_{t+1} = s' \mid S_t = s, A_t = a)
  \nonumber\\
  &= \pi(a \mid s) \cdot P(s' \mid s, a).
  \label{eq:expand_step2}
\end{align}
Substituting \eqref{eq:expand_step2}
into~\eqref{eq:expand_step1}
gives~\eqref{eq:mdp_expansion}.
\end{proof}

\subsection{The Bellman Expectation Equation}
\label{subsec:bellman_eq}

\begin{keyresult}[title={Bellman Expectation Equation}]
\begin{corollary}[Bellman Expectation Equation]
\label{cor:bellman}
For the value function
$V^\pi(s) = \E_\pi[G_t \mid S_t = s]$ with
$G_t = R_{t+1} + \gamma G_{t+1}$:
\begin{equation}
  V^\pi(s) = \sum_{a} \pi(a \mid s) \sum_{s'}
  P(s' \mid s, a)\, \bigl[r(s,a,s') + \gamma V^\pi(s')\bigr].
  \label{eq:bellman}
\end{equation}
\end{corollary}
\end{keyresult}

\begin{proof}
\textbf{Step 1: Decompose the return.}
By the definition of the discounted return:
\begin{equation}
  V^\pi(s) = \E_\pi[G_t \mid S_t = s]
  = \E_\pi[R_{t+1} + \gamma G_{t+1} \mid S_t = s].
  \label{eq:bellman_step1}
\end{equation}

\textbf{Step 2: Apply linearity of conditional expectation.}
\begin{equation}
  = \E_\pi[R_{t+1} \mid S_t = s]
    + \gamma\, \E_\pi[G_{t+1} \mid S_t = s].
  \label{eq:bellman_step2}
\end{equation}

\textbf{Step 3: Apply the tower property to the second term.}
By the generalized tower property
(Theorem~\ref{thm:tower_general}) and the Markov property:
\begin{align}
  \E_\pi[G_{t+1} \mid S_t = s]
  &= \E_\pi\!\left[
    \E_\pi[G_{t+1} \mid S_{t+1}]
    \;\middle|\; S_t = s\right]
  \nonumber\\
  &= \E_\pi[V^\pi(S_{t+1}) \mid S_t = s].
  \label{eq:bellman_tower}
\end{align}
The inner conditioning uses the Markov property: given $S_{t+1}$,
$G_{t+1}$ is conditionally independent of $S_t$, so
$\E_\pi[G_{t+1} \mid S_{t+1}, S_t] = \E_\pi[G_{t+1} \mid S_{t+1}]
= V^\pi(S_{t+1})$.

\textbf{Step 4: Expand via MDP dynamics.}
Combining \eqref{eq:bellman_step2} and \eqref{eq:bellman_tower}:
\begin{equation}
  V^\pi(s) = \E_\pi[R_{t+1} + \gamma V^\pi(S_{t+1})
    \mid S_t = s].
\end{equation}
Setting $f(a, s', r) = r + \gamma V^\pi(s')$ in
Theorem~\ref{thm:mdp_expansion}:
\begin{equation}
  V^\pi(s) = \sum_a \pi(a \mid s) \sum_{s'}
  P(s' \mid s, a)\, \bigl[r(s,a,s') + \gamma V^\pi(s')\bigr].
  \qedhere
\end{equation}
\end{proof}

\begin{remark}[Operator notation]
In operator notation, the Bellman equation reads
$V^\pi = \mathcal{T}^\pi V^\pi$, where the Bellman operator
$\mathcal{T}^\pi$ is defined by the right-hand side
of~\eqref{eq:bellman}.  The fact that $V^\pi$ is a fixed point
of $\mathcal{T}^\pi$ is a direct consequence of the tower
property.
\end{remark}

\begin{rlconnection}[title={Connection to TD Learning}]
The Bellman equation derivation relies on three tools from this
lesson:
\begin{enumerate}[nosep]
  \item \textbf{Linearity} (Theorem~\ref{thm:cond_linearity})
    to split $\E[R_{t+1} + \gamma G_{t+1}]$.
  \item \textbf{Tower property}
    (Theorem~\ref{thm:tower_general}) to replace
    $\E[G_{t+1} \mid S_t]$ with
    $\E[V^\pi(S_{t+1}) \mid S_t]$.
  \item \textbf{MDP expansion}
    (Theorem~\ref{thm:mdp_expansion}) to write the result as
    an explicit sum over actions and next states.
\end{enumerate}
TD learning bootstraps by replacing the unknown $V^\pi(S_{t+1})$
with a current estimate $V(S_{t+1})$, producing the TD target
$R_{t+1} + \gamma V(S_{t+1})$.  The tower property guarantees
that $\E_\pi[\delta_t^\pi \mid S_t = s] = 0$ when $V = V^\pi$,
the foundation of all TD convergence proofs.
\end{rlconnection}

\section{Exercises}
\label{sec:exercises}

\begin{exercise}
\label{ex:cond_exp_compute}
Let $X$ and $Y$ be independent with
$X \sim \operatorname{Bernoulli}(p)$ and
$Y \sim \operatorname{Bernoulli}(q)$.
Define $Z = X + Y$.  Compute $\E[X \mid Z]$ for each possible
value of $Z$.
\end{exercise}

\begin{exercise}
\label{ex:pullout_application}
Let $X$ and $Y$ be random variables with finite second moments.
Using the pull-out property and the tower property, show that
$\E[Y \cdot \E[X \mid Y]] = \E[XY]$.
\textit{Hint:} Start from the right-hand side and apply the
tower property with the inner conditioning on $Y$.
\end{exercise}

\begin{exercise}
\label{ex:iterated_tower}
Let $X$, $Y$, $Z$ be discrete random variables with
$\E[|X|] < \infty$.  Show that
$\E\bigl[\E[\E[X \mid Y, Z] \mid Y] \bigr] = \E[X]$ by
applying the tower property twice.
\end{exercise}

\begin{exercise}
\label{ex:bellman_two_state}
Consider a two-state MDP with
$\mathcal{S} = \{s_1, s_2\}$,
$\mathcal{A} = \{a\}$ (single action),
$P(s_1 \mid s_1, a) = 0.6$,
$P(s_2 \mid s_1, a) = 0.4$,
$P(s_1 \mid s_2, a) = 0.3$,
$P(s_2 \mid s_2, a) = 0.7$,
$r(s_1, a, \cdot) = 1$, $r(s_2, a, \cdot) = 2$, and
$\gamma = 0.9$.
Write the Bellman equation~\eqref{eq:bellman} for each state
and solve the resulting $2 \times 2$ linear system for
$V^\pi(s_1)$ and $V^\pi(s_2)$.
\end{exercise}

\begin{exercise}
\label{ex:tower_conditional}
Prove the \emph{conditional tower property}: if
$\mathcal{G} \subseteq \mathcal{H}$ are sub-$\sigma$-algebras,
then for any event $A \in \mathcal{G}$:
\[
  \E\bigl[\E[X \mid \mathcal{H}] \cdot \indicator_A\bigr]
  = \E[X \cdot \indicator_A].
\]
\textit{Hint:} Use the defining
property~\eqref{eq:cond_exp_sigma} of conditional expectation.
\end{exercise}

\begin{exercise}
\label{ex:cond_exp_constant}
Let $X$ and $Y$ be independent with $\E[|X|] < \infty$.
Show that $\E[X \mid Y] = \E[X]$ a.s.
\textit{Hint:} For independent discrete RVs,
$\Prob(X = x \mid Y = y) = \Prob(X = x)$.
\end{exercise}

\begin{exercise}
\label{ex:td_error_zero_mean}
In an MDP with policy $\pi$ and true value function $V^\pi$,
the TD error is $\delta_t^\pi = R_{t+1} + \gamma V^\pi(S_{t+1})
- V^\pi(S_t)$.  Using linearity, the pull-out property, and the
Bellman equation, show that
$\E_\pi[\delta_t^\pi \mid S_t = s] = 0$.
\end{exercise}
