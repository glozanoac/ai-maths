\section{Motivation and Overview}
\label{sec:motivation}

Lessons 1--3 developed the general theory of sigma-algebras, measures, and
measurable functions. This lesson specializes that framework to
\emph{probability}. The key observation is that a probability space is
simply a measure space of total mass one, and a random variable is simply a
measurable function on that space. This identification lets every theorem
from measure theory -- Caratheodory extension, the approximation theorem,
the Doob--Dynkin lemma -- carry over directly.

\medskip\noindent\textbf{Goals of this lesson.}
\begin{enumerate}[nosep]
  \item Define probability spaces as a special case of measure spaces
    (Kolmogorov axioms).
  \item Define random variables, distributions (pushforward measures), and
    cumulative distribution functions.
  \item Prove that the CDF uniquely determines the distribution
    (via the $\pi$-$\lambda$ theorem).
  \item Construct the probability space for a finite Markov decision process,
    including the natural filtration.
\end{enumerate}

\section{Probability Spaces}
\label{sec:prob_spaces}

\begin{definition}[Probability Space]
\label{def:prob_space}
A \emph{probability space} is a triple $(\Omega, \mathcal{F}, \Prob)$ where:
\begin{enumerate}[nosep]
  \item $\Omega$ is a non-empty set called the \emph{sample space},
  \item $\mathcal{F}$ is a sigma-algebra on $\Omega$ whose elements are
    called \emph{events},
  \item $\Prob: \mathcal{F} \to [0, 1]$ is a \emph{probability measure},
    i.e., a measure with $\Prob(\Omega) = 1$.
\end{enumerate}
Explicitly, $\Prob$ satisfies the \emph{Kolmogorov axioms}:
\begin{enumerate}[nosep,label=(K\arabic*)]
  \item $\Prob(A) \geq 0$ for all $A \in \mathcal{F}$.
  \item $\Prob(\Omega) = 1$.
  \item For pairwise disjoint $A_1, A_2, \ldots \in \mathcal{F}$:
    $\displaystyle\Prob\!\left(\bigcup_{n=1}^\infty A_n\right)
    = \sum_{n=1}^\infty \Prob(A_n)$.
\end{enumerate}
\end{definition}

\begin{remark}
Every probability measure is a measure in the sense of Lesson~2.
Conversely, any finite measure $\mu$ with $\mu(\Omega) > 0$ can be
normalized: $\Prob(A) = \mu(A)/\mu(\Omega)$ defines a probability measure.
The Kolmogorov axioms are exactly the measure axioms with the additional
constraint $\Prob(\Omega) = 1$.
\end{remark}

\begin{theorem}[Basic Properties of Probability]
\label{thm:prob_basic}
Let $(\Omega, \mathcal{F}, \Prob)$ be a probability space and let
$A, B \in \mathcal{F}$. Then:
\begin{enumerate}[nosep,label=(\alph*)]
  \item $\Prob(\emptyset) = 0$.
  \item $\Prob(A^c) = 1 - \Prob(A)$.
  \item If $A \subseteq B$, then $\Prob(A) \leq \Prob(B)$.
  \item $\Prob(A \cup B) = \Prob(A) + \Prob(B) - \Prob(A \cap B)$.
  \item $\Prob\!\left(\bigcup_{n=1}^\infty A_n\right)
    \leq \sum_{n=1}^\infty \Prob(A_n)$ \quad (union bound / Boole's
    inequality).
\end{enumerate}
\end{theorem}

\begin{proof}
These follow from the corresponding properties of general measures
(Lesson~2), but we give self-contained proofs for completeness.

\textbf{(a)} Write $\Omega = \Omega \cup \emptyset \cup \emptyset \cup
\cdots$ (pairwise disjoint). By (K3),
$1 = \Prob(\Omega) = \Prob(\Omega) + \sum_{n=2}^\infty \Prob(\emptyset)$.
Since $1 < \infty$, all additional terms must vanish, so
$\Prob(\emptyset) = 0$.

\textbf{(b)} $\Omega = A \sqcup A^c$ (disjoint union), so
$1 = \Prob(A) + \Prob(A^c)$ by (K3) applied to two sets (extend with
$\emptyset$).

\textbf{(c)} $B = A \sqcup (B \setminus A)$, so
$\Prob(B) = \Prob(A) + \Prob(B \setminus A) \geq \Prob(A)$.

\textbf{(d)} Write $A \cup B = A \sqcup (B \setminus A)$ and
$B = (A \cap B) \sqcup (B \setminus A)$. The second gives
$\Prob(B \setminus A) = \Prob(B) - \Prob(A \cap B)$. Substituting:
$\Prob(A \cup B) = \Prob(A) + \Prob(B) - \Prob(A \cap B)$.

\textbf{(e)} Define $B_1 = A_1$ and
$B_n = A_n \setminus \bigcup_{k=1}^{n-1} A_k$ for $n \geq 2$. The $B_n$
are pairwise disjoint with $\bigcup_n B_n = \bigcup_n A_n$ and
$B_n \subseteq A_n$. By (K3) and monotonicity:
$\Prob(\bigcup A_n) = \sum_n \Prob(B_n) \leq \sum_n \Prob(A_n)$.
\end{proof}

\begin{theorem}[Continuity of Probability]
\label{thm:continuity}
Let $(\Omega, \mathcal{F}, \Prob)$ be a probability space.
\begin{enumerate}[nosep,label=(\alph*)]
  \item \textbf{Continuity from below:} If
    $A_1 \subseteq A_2 \subseteq \cdots$, then
    $\Prob\!\left(\bigcup_{n} A_n\right)
    = \lim_{n \to \infty} \Prob(A_n)$.
  \item \textbf{Continuity from above:} If
    $A_1 \supseteq A_2 \supseteq \cdots$, then
    $\Prob\!\left(\bigcap_{n} A_n\right)
    = \lim_{n \to \infty} \Prob(A_n)$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textbf{(a)} Define $B_1 = A_1$ and $B_n = A_n \setminus A_{n-1}$ for
$n \geq 2$. The $B_n$ are pairwise disjoint (if $i < j$ and
$\omega \in B_i$ then $\omega \in A_i \subseteq A_{j-1}$, but
$\omega \in B_j$ implies $\omega \notin A_{j-1}$, a contradiction).
By induction, $\bigcup_{k=1}^N B_k = A_N$, so
$\sum_{k=1}^N \Prob(B_k) = \Prob(A_N)$ by finite additivity. Taking
$N \to \infty$:
\[
  \Prob\!\left(\bigcup_n A_n\right)
  = \Prob\!\left(\bigcup_n B_n\right)
  = \sum_{n=1}^\infty \Prob(B_n)
  = \lim_{N \to \infty} \Prob(A_N).
\]

\textbf{(b)} If $A_n$ is decreasing, then $A_n^c$ is increasing. By (a),
$\Prob(\bigcup_n A_n^c) = \lim \Prob(A_n^c) = \lim (1 - \Prob(A_n))$.
By De Morgan, $\bigcup_n A_n^c = (\bigcap_n A_n)^c$, so
$1 - \Prob(\bigcap_n A_n) = 1 - \lim \Prob(A_n)$, giving the result.
\end{proof}

\begin{rlconnection}[title={Connection to RL}]
Continuity from above is essential for ``almost sure'' convergence
statements. The event ``$V_t \to V^\pi$'' is
$\bigcap_{k=1}^\infty \bigcup_{N=1}^\infty
\bigcap_{t=N}^\infty \{|V_t - V^\pi| < 1/k\}$, and computing its
probability requires continuity of $\Prob$ along nested sequences.
\end{rlconnection}

\section{Random Variables}
\label{sec:rv}

\begin{definition}[Random Variable]
\label{def:rv}
A \emph{random variable} on a probability space
$(\Omega, \mathcal{F}, \Prob)$ is a measurable function
$X: (\Omega, \mathcal{F}) \to (\R, \Borel(\R))$.

More generally, an \emph{$S$-valued random variable} (or \emph{random
element}) is a measurable function
$X: (\Omega, \mathcal{F}) \to (S, \mathcal{S})$ for some measurable space
$(S, \mathcal{S})$.
\end{definition}

\begin{remark}
By the standard criterion (Lesson~3, Corollary~3.2), a function
$X: \Omega \to \R$ is a random variable if and only if
$\{X \leq x\} \in \mathcal{F}$ for all $x \in \R$. This ensures we can
assign probabilities to events like ``$X$ is at most $x$.''
\end{remark}

\begin{example}[Indicator Random Variable]
\label{ex:indicator_rv}
For any event $A \in \mathcal{F}$, the indicator function
$\indicator_A: \Omega \to \{0, 1\}$ is a random variable with
$\Prob(\indicator_A = 1) = \Prob(A)$ and
$\Prob(\indicator_A = 0) = 1 - \Prob(A)$.
\end{example}

\begin{example}[State as a Random Variable]
\label{ex:state_rv}
In a finite MDP with states $\mathcal{S} = \{s_1, \ldots, s_m\}$, the
state at time $t$ is a random variable
$S_t: \Omega \to \mathcal{S}$. Equipping $\mathcal{S}$ with the discrete
sigma-algebra $\powerset(\mathcal{S})$, measurability is automatic: for
every $B \subseteq \mathcal{S}$,
$\{S_t \in B\} = S_t^{-1}(B) \in \mathcal{F}$.
\end{example}

\section{Distributions (Pushforward Measures)}
\label{sec:distribution}

\begin{definition}[Distribution / Pushforward Measure]
\label{def:pushforward}
Let $X: (\Omega, \mathcal{F}, \Prob) \to (S, \mathcal{S})$ be a random
element. The \emph{distribution} (or \emph{law}) of $X$ is the probability
measure $\mu_X$ on $(S, \mathcal{S})$ defined by
\begin{equation}
  \mu_X(B) = \Prob(X \in B) = \Prob\bigl(X^{-1}(B)\bigr)
  \quad \text{for all } B \in \mathcal{S}.
  \label{eq:pushforward}
\end{equation}
This is also called the \emph{pushforward} of $\Prob$ by $X$, written
$\Prob \circ X^{-1}$ or $X_\# \Prob$.
\end{definition}

\begin{proposition}[$\mu_X$ is a Probability Measure]
\label{prop:pushforward_is_prob}
The pushforward $\mu_X$ defined in~\eqref{eq:pushforward} is indeed a
probability measure on $(S, \mathcal{S})$.
\end{proposition}

\begin{proof}
We verify the three Kolmogorov axioms.

\textit{(K1) Non-negativity:} $\mu_X(B) = \Prob(X^{-1}(B)) \geq 0$
since $\Prob$ is non-negative.

\textit{(K2) Normalization:} $\mu_X(S) = \Prob(X^{-1}(S))
= \Prob(\Omega) = 1$.

\textit{(K3) Countable additivity:} Let $B_1, B_2, \ldots$ be pairwise
disjoint in $\mathcal{S}$. Then $X^{-1}(B_1), X^{-1}(B_2), \ldots$
are pairwise disjoint in $\mathcal{F}$ (if
$\omega \in X^{-1}(B_i) \cap X^{-1}(B_j)$ then $X(\omega) \in B_i \cap B_j$,
contradicting disjointness of $B_i, B_j$). Therefore:
\[
  \mu_X\!\left(\bigcup_n B_n\right)
  = \Prob\!\left(X^{-1}\!\left(\bigcup_n B_n\right)\right)
  = \Prob\!\left(\bigcup_n X^{-1}(B_n)\right)
  = \sum_n \Prob(X^{-1}(B_n))
  = \sum_n \mu_X(B_n). \qedhere
\]
\end{proof}

\begin{remark}
Two random variables $X$ and $Y$ (possibly defined on different probability
spaces) have the \emph{same distribution} if $\mu_X = \mu_Y$, i.e.,
$\Prob(X \in B) = \Prob(Y \in B)$ for all Borel sets $B$. This is written
$X \stackrel{d}{=} Y$ or $\text{Law}(X) = \text{Law}(Y)$.
\end{remark}

\section{Cumulative Distribution Functions}
\label{sec:cdf}

\begin{definition}[Cumulative Distribution Function]
\label{def:cdf}
The \emph{cumulative distribution function} (CDF) of a real-valued random
variable $X$ is the function $F_X: \R \to [0, 1]$ defined by
\begin{equation}
  F_X(x) = \Prob(X \leq x) = \mu_X\bigl((-\infty, x]\bigr),
  \quad x \in \R.
  \label{eq:cdf}
\end{equation}
\end{definition}

\begin{proposition}[Properties of the CDF]
\label{prop:cdf_props}
Every CDF $F = F_X$ satisfies:
\begin{enumerate}[nosep,label=(\alph*)]
  \item $F$ is non-decreasing: $x \leq y \implies F(x) \leq F(y)$.
  \item $\lim_{x \to -\infty} F(x) = 0$ and
    $\lim_{x \to +\infty} F(x) = 1$.
  \item $F$ is right-continuous: $\lim_{y \downarrow x} F(y) = F(x)$.
\end{enumerate}
\end{proposition}

\begin{proof}
\textbf{(a)} If $x \leq y$, then $\{X \leq x\} \subseteq \{X \leq y\}$,
so $F(x) = \Prob(X \leq x) \leq \Prob(X \leq y) = F(y)$ by monotonicity
of $\Prob$.

\textbf{(b)} For the left limit, let $A_n = \{X \leq -n\}$. These sets
are decreasing ($A_{n+1} \subseteq A_n$) and
$\bigcap_{n=1}^\infty A_n = \emptyset$ (no real number is $\leq -n$ for all
$n$, by the Archimedean property). By continuity from above
(Theorem~\ref{thm:continuity}(b)):
\[
  \lim_{n \to \infty} F(-n) = \lim_{n \to \infty} \Prob(A_n)
  = \Prob\!\left(\bigcap_n A_n\right) = \Prob(\emptyset) = 0.
\]
Since $F$ is non-decreasing, $\lim_{x \to -\infty} F(x)
= \lim_{n \to \infty} F(-n) = 0$.

For the right limit, $A_n = \{X \leq n\}$ is increasing with
$\bigcup_n A_n = \Omega$. By continuity from below:
$\lim F(n) = \Prob(\Omega) = 1$.

\textbf{(c)} Let $y_n \downarrow x$ (i.e., $y_n$ decreases to $x$). Then
$\{X \leq y_n\}$ is a decreasing sequence with
\[
  \bigcap_{n=1}^\infty \{X \leq y_n\} = \{X \leq x\}.
\]
To see the equality: if $X(\omega) \leq x$, then $X(\omega) \leq y_n$ for
all $n$ (since $y_n \geq x$), so $\omega$ is in the intersection.
Conversely, if $\omega$ is in the intersection, then $X(\omega) \leq y_n$
for all $n$, and taking the limit gives $X(\omega) \leq x$.
By continuity from above:
$\lim F(y_n) = \Prob(\bigcap_n \{X \leq y_n\}) = \Prob(X \leq x) = F(x)$.
\end{proof}

\begin{keyresult}[title={CDF Determines Distribution}]
\begin{proposition}[CDF Determines Distribution]
\label{prop:cdf_determines}
The CDF $F_X$ uniquely determines the distribution $\mu_X$. That is, if
$X$ and $Y$ are random variables with $F_X(x) = F_Y(x)$ for all $x$,
then $\mu_X = \mu_Y$.
\end{proposition}
\end{keyresult}

\begin{proof}
We apply the uniqueness theorem from Lesson~1 (via the
$\pi$-$\lambda$ theorem).

\textit{Step 1: Identify a $\pi$-system.}
Let $\mathcal{P} = \{(-\infty, x] : x \in \R\}$. This is a $\pi$-system
because $(-\infty, x] \cap (-\infty, y] = (-\infty, \min(x,y)] \in
\mathcal{P}$.

\textit{Step 2: $\sigma(\mathcal{P}) = \Borel(\R)$.}
The half-lines $(-\infty, x]$ generate the Borel sigma-algebra
(see Lesson~1).

\textit{Step 3: $\mu_X$ and $\mu_Y$ agree on $\mathcal{P}$.}
For every $x \in \R$:
$\mu_X((-\infty, x]) = F_X(x) = F_Y(x) = \mu_Y((-\infty, x])$.

\textit{Step 4: Apply uniqueness.}
By the uniqueness corollary of the $\pi$-$\lambda$ theorem (Lesson~1),
two probability measures that agree on a $\pi$-system agree on the
generated sigma-algebra. Since $\mu_X$ and $\mu_Y$ agree on $\mathcal{P}$,
they agree on $\sigma(\mathcal{P}) = \Borel(\R)$. Hence $\mu_X = \mu_Y$.
\end{proof}

\begin{remark}
This result shows that the CDF is a complete description of a random
variable's distributional behavior. Two random variables with the same
CDF assign the same probability to every Borel set.
\end{remark}

\begin{proposition}[Point Probabilities from the CDF]
\label{prop:cdf_point_mass}
For any $x \in \R$,
\begin{equation}
  \Prob(X = x) = F_X(x) - F_X(x^-)
  \label{eq:cdf_jump}
\end{equation}
where $F_X(x^-) = \lim_{y \uparrow x} F_X(y)$ is the left limit.
In particular, $\Prob(X = x) > 0$ if and only if $F_X$ has a jump
discontinuity at $x$.
\end{proposition}

\begin{proof}
Let $y_n \uparrow x$ with $y_n < x$. Then
$\{X \leq y_n\} \uparrow \{X < x\}$ (the increasing union
$\bigcup_n \{X \leq y_n\}$ equals $\{X < x\}$ because for any
$\omega$ with $X(\omega) < x$, there exists $n$ with
$X(\omega) \leq y_n < x$). By continuity from below:
\[
  F_X(x^-) = \lim_n F_X(y_n) = \lim_n \Prob(X \leq y_n)
  = \Prob(X < x).
\]
Since $\{X \leq x\} = \{X < x\} \sqcup \{X = x\}$:
\[
  \Prob(X = x) = \Prob(X \leq x) - \Prob(X < x)
  = F_X(x) - F_X(x^-). \qedhere
\]
\end{proof}

\section{Probability Space for a Finite MDP}
\label{sec:mdp}

We now construct the probability space underlying a Markov decision process
with finite state and action spaces. This example ties together all the
concepts of this lesson and provides the foundation for the formal
treatment of reinforcement learning.

\begin{example}[Probability Space for a Finite MDP]
\label{ex:mdp_prob_space}
Consider a finite MDP specified by:
\begin{itemize}[nosep]
  \item State space $\mathcal{S} = \{s_1, \ldots, s_m\}$.
  \item Action space $\mathcal{A} = \{a_1, \ldots, a_k\}$.
  \item Reward set $\mathcal{R} \subset \R$ (finite).
  \item Transition dynamics: $p(s', r \mid s, a)
    = \Prob(S_{t+1} = s', R_{t+1} = r \mid S_t = s, A_t = a)$.
  \item Policy: $\pi(a \mid s) = \Prob(A_t = a \mid S_t = s)$.
  \item Initial distribution: $p_0(s) = \Prob(S_0 = s)$.
\end{itemize}

\textbf{The probability space $(\Omega, \mathcal{F}, \Prob_\pi)$.}

\textit{Sample space.} A trajectory is an infinite sequence
\[
  \omega = (s_0, a_0, r_1, s_1, a_1, r_2, s_2, a_2, \ldots).
\]
The sample space is the set of all such trajectories:
\[
  \Omega = (\mathcal{S} \times \mathcal{A} \times \mathcal{R})^\infty.
\]

\textit{Sigma-algebra.} Equip $\mathcal{S}$, $\mathcal{A}$, $\mathcal{R}$
each with the discrete sigma-algebra (the power set). The product
sigma-algebra on $\Omega$ is
\[
  \mathcal{F} = \sigma(S_0, A_0, R_1, S_1, A_1, R_2, \ldots),
\]
where $S_t, A_t, R_{t+1}$ are the coordinate projection functions.

\textit{Probability measure.} By the Ionescu-Tulcea extension theorem
(the infinite-product version of Caratheodory), there exists a unique
probability measure $\Prob_\pi$ on $(\Omega, \mathcal{F})$ satisfying
\begin{equation}
  \Prob_\pi(S_0 = s_0, A_0 = a_0, R_1 = r_1, S_1 = s_1, \ldots,
  R_T = r_T, S_T = s_T)
  = p_0(s_0) \prod_{t=0}^{T-1} \pi(a_t \mid s_t)\,
    p(s_{t+1}, r_{t+1} \mid s_t, a_t).
  \label{eq:mdp_prob}
\end{equation}
The consistency of finite-dimensional distributions (each marginal obtained
from the next by summing over the added coordinate) ensures that the
extension is well-defined.
\end{example}

\begin{definition}[Natural Filtration of an MDP]
\label{def:mdp_filtration}
The \emph{natural filtration} $(\mathcal{F}_t)_{t \geq 0}$ of the MDP is
\begin{equation}
  \mathcal{F}_t = \sigma(S_0, A_0, R_1, S_1, \ldots, A_{t-1}, R_t, S_t).
  \label{eq:mdp_filtration}
\end{equation}
This is a \emph{filtration}: $\mathcal{F}_0 \subseteq \mathcal{F}_1
\subseteq \mathcal{F}_2 \subseteq \cdots$, and each $\mathcal{F}_t$
represents the information available at time $t$.
\end{definition}

\begin{proposition}[Filtration is Increasing]
\label{prop:filtration_increasing}
$\mathcal{F}_t \subseteq \mathcal{F}_{t+1}$ for all $t \geq 0$.
\end{proposition}

\begin{proof}
$\mathcal{F}_{t+1} = \sigma(S_0, A_0, R_1, \ldots, S_t, A_t, R_{t+1},
S_{t+1})$ is generated by a larger collection of random variables than
$\mathcal{F}_t$. Formally, every generator of $\mathcal{F}_t$ is also a
generator of $\mathcal{F}_{t+1}$, so the generated sigma-algebra satisfies
$\mathcal{F}_t \subseteq \mathcal{F}_{t+1}$.
\end{proof}

\begin{rlconnection}[title={Connection to RL}]
The filtration $\mathcal{F}_t$ captures the agent's history at time $t$.
The Markov property says that the conditional distribution of the future
given $\mathcal{F}_t$ depends only on $S_t$. Formally:
$\Prob(S_{t+1} = s' \mid \mathcal{F}_t) = \Prob(S_{t+1} = s' \mid S_t)$
$\Prob$-almost surely. This will be stated precisely once conditional
expectation is defined (Lesson~6).
\end{rlconnection}

\section{Additional Results}
\label{sec:additional}

\begin{proposition}[Distribution of a Function of a Random Variable]
\label{prop:dist_function}
If $X$ is a random variable and $g: \R \to \R$ is Borel measurable, then
$Y = g(X)$ is a random variable with distribution
\begin{equation}
  \mu_Y(B) = \mu_X(g^{-1}(B)) \quad \text{for all } B \in \Borel(\R).
\end{equation}
\end{proposition}

\begin{proof}
$Y = g \circ X$ is measurable as the composition of measurable functions
(Lesson~3, Example~3.6). For any Borel set $B$:
\[
  \mu_Y(B) = \Prob(Y \in B) = \Prob(g(X) \in B)
  = \Prob(X \in g^{-1}(B)) = \mu_X(g^{-1}(B)). \qedhere
\]
\end{proof}

\begin{proposition}[CDF of a Discrete Random Variable]
\label{prop:cdf_discrete}
If $X$ takes values in a countable set $\{x_1, x_2, \ldots\}$ with
$p_k = \Prob(X = x_k)$, then
\begin{equation}
  F_X(x) = \sum_{k:\, x_k \leq x} p_k.
  \label{eq:cdf_discrete}
\end{equation}
In particular, $F_X$ is a right-continuous step function with jumps of
size $p_k$ at each $x_k$.
\end{proposition}

\begin{proof}
$\{X \leq x\} = \bigsqcup_{k:\, x_k \leq x} \{X = x_k\}$ (disjoint
since the $x_k$ are distinct). By countable additivity:
$F_X(x) = \sum_{k:\, x_k \leq x} \Prob(X = x_k) = \sum_{k:\, x_k \leq x} p_k$.
\end{proof}

\section{Exercises}
\label{sec:exercises}

\begin{exercise}
\label{ex:sigma_finite}
Show that $|\mathcal{F}|$ (the number of elements in a finite
sigma-algebra on a finite set $\Omega$) is always a power of $2$.
\textit{Hint:} Show that every finite sigma-algebra is generated by a
partition of $\Omega$ into \emph{atoms}, where an atom is a minimal
non-empty element of $\mathcal{F}$.
\end{exercise}

\begin{exercise}
\label{ex:borel_cantelli}
Let $(\Omega, \mathcal{F}, \Prob)$ be a probability space and
$A_n \in \mathcal{F}$. Prove the \emph{first Borel--Cantelli lemma}: if
$\sum_n \Prob(A_n) < \infty$, then
$\Prob(\limsup_n A_n) = 0$, where
$\limsup_n A_n = \bigcap_{N=1}^\infty \bigcup_{n=N}^\infty A_n$.
\textit{Hint:} Use continuity from above and the union bound.
\end{exercise}

\begin{exercise}
\label{ex:cdf_construct}
Let $F: \R \to [0,1]$ be defined by
\[
  F(x) = \begin{cases}
    0 & \text{if } x < 0, \\
    x/2 & \text{if } 0 \leq x < 1, \\
    3/4 & \text{if } 1 \leq x < 2, \\
    1 & \text{if } x \geq 2.
  \end{cases}
\]
\begin{enumerate}[nosep,label=(\alph*)]
  \item Verify that $F$ satisfies all three CDF properties
    (non-decreasing, correct limits, right-continuous at every point).
  \item Compute $\Prob(X = 1)$, $\Prob(X = 2)$, $\Prob(0.3 \leq X < 0.7)$.
  \item Does $X$ have a density on $[0, 1)$? Explain.
\end{enumerate}
\end{exercise}

\begin{exercise}
\label{ex:coin_flip}
Construct a probability measure on $(\{0,1\}^\N, \mathcal{F})$
corresponding to an infinite sequence of fair coin flips, where
$\mathcal{F}$ is the sigma-algebra generated by the cylinder sets.
Verify $\Prob(\Omega) = 1$ and countable additivity on cylinder sets.
\end{exercise}

\begin{exercise}
\label{ex:pushforward_uniform}
Let $U \sim \text{Uniform}[0,1]$ (i.e., $\Prob(U \leq x) = x$ for
$x \in [0,1]$). Let $Y = -\ln(1 - U)$.
\begin{enumerate}[nosep,label=(\alph*)]
  \item Compute the CDF of $Y$.
  \item Identify the distribution of $Y$.
  \item Generalize: for a continuous CDF $F$ with inverse $F^{-1}$, show
    that $X = F^{-1}(U)$ has CDF $F$.
\end{enumerate}
\end{exercise}

\begin{exercise}
\label{ex:monotone_class}
Prove the \emph{Monotone Class Theorem}: if $\mathcal{A}$ is an algebra
on $\Omega$, then the smallest monotone class containing $\mathcal{A}$
equals $\sigma(\mathcal{A})$.
\textit{Hint:} Adapt the proof of the $\pi$-$\lambda$ theorem from
Lesson~1.
\end{exercise}

\begin{exercise}[RL Application: MDP Filtration]
\label{ex:rl_filtration}
Consider an MDP with $\mathcal{S} = \{s_1, s_2, s_3\}$ and
$\mathcal{A} = \{a_1, a_2\}$.
\begin{enumerate}[nosep,label=(\alph*)]
  \item Write out explicitly all elements of $\sigma(S_0)$.
  \item Write out all elements of $\sigma(S_0, A_0)$ as a sigma-algebra on
    the appropriate product space.
  \item Verify that $\sigma(S_0) \subseteq \sigma(S_0, A_0)$.
  \item For a value function $V: \mathcal{S} \to \R$, verify that
    $V(S_0)$ is $\sigma(S_0)$-measurable by identifying the Borel
    function $g$ in the Doob--Dynkin lemma.
\end{enumerate}
\end{exercise}
