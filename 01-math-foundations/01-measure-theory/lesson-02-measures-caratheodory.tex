\section{Motivation and Overview}
\label{sec:motivation}

In the previous lesson we constructed $\sigma$-algebras to specify
which subsets of a sample space are ``measurable.'' The next step is to
assign sizes---lengths, areas, probabilities---to these sets. A
\emph{measure} is the mathematical object that does this, and the
Carath\'{e}odory extension theorem is the fundamental tool for
constructing measures rigorously.

The construction matters for two reasons: (1)~every probability space
$(\Omega, \mathcal{F}, \Prob)$ is a measure space, so properties of
measures (monotonicity, continuity) become properties of probability
used constantly in convergence proofs; and (2)~the expected discounted
return $\E[\sum_{k=0}^\infty \gamma^k R_{t+k+1}]$ is an integral
with respect to $\Prob$, requiring the Lebesgue integral built on
measures.

\begin{rlconnection}[title={Why This Matters for RL}]
Many RL convergence proofs apply the Borel--Cantelli lemma, the
dominated convergence theorem, or Fubini's theorem---all of which
require a rigorously constructed measure. The Carath\'{e}odory
extension theorem guarantees that the probability measure underlying an
MDP with continuous state/action spaces actually exists. Without it,
expressions like
$\Prob(S_{t+1} \in B \mid S_t = s, A_t = a) = \int_B p(s' \mid s, a)
\,ds'$ would be formal manipulations without foundation.
\end{rlconnection}

\section{Measures}
\label{sec:measures}

\begin{definition}[Measure]
\label{def:measure}
A \emph{measure} on a measurable space $(\Omega, \mathcal{F})$ is a
function $\mu\colon \mathcal{F} \to [0, \infty]$ satisfying:
\begin{enumerate}[nosep,label=(\roman*)]
  \item $\mu(\emptyset) = 0$.
  \item \textbf{Countable additivity:} If $A_1, A_2, \ldots \in
    \mathcal{F}$ are pairwise disjoint, then
    \begin{equation}
      \mu\!\left(\bigsqcup_{n=1}^\infty A_n\right)
        = \sum_{n=1}^\infty \mu(A_n).
      \label{eq:countable_additivity}
    \end{equation}
\end{enumerate}
The triple $(\Omega, \mathcal{F}, \mu)$ is called a \emph{measure
space}. If $\mu(\Omega) = 1$, then $\mu$ is a \emph{probability
measure} and we write $\Prob$ instead of $\mu$.
\end{definition}

\begin{example}[Standard Measures]
\label{ex:standard_measures}
\begin{enumerate}[nosep,label=(\alph*)]
  \item \textbf{Counting measure:} On any set $\Omega$ with
    $\mathcal{F} = \powerset(\Omega)$, define $\mu(A) = |A|$ (the
    cardinality of $A$, with $|A| = \infty$ if $A$ is infinite).
  \item \textbf{Dirac measure:} For $x \in \Omega$, define
    $\delta_x(A) = \indicator_A(x) =
    \begin{cases} 1 & \text{if } x \in A, \\
    0 & \text{if } x \notin A. \end{cases}$
  \item \textbf{Lebesgue measure:} On $(\R, \Borel(\R))$, the unique
    measure with $\lambda((a, b]) = b - a$ for all $a < b$.
    (Existence proved in Section~\ref{sec:lebesgue}.)
  \item \textbf{Probability on a finite set:} For
    $\Omega = \{s_1, \ldots, s_n\}$, any function
    $p\colon \Omega \to [0,1]$ with $\sum p(s_i) = 1$ defines a
    probability measure $\Prob(A) = \sum_{s_i \in A} p(s_i)$.
\end{enumerate}
\end{example}

\begin{theorem}[Properties of Measures]
\label{thm:measure_props}
Let $(\Omega, \mathcal{F}, \mu)$ be a measure space. For all
$A, B, A_1, A_2, \ldots \in \mathcal{F}$:
\begin{enumerate}[nosep,label=(\alph*)]
  \item \textbf{Monotonicity:}
    $A \subseteq B \implies \mu(A) \leq \mu(B)$.
  \item \textbf{Subtractivity:}
    $A \subseteq B$ and $\mu(A) < \infty \implies
    \mu(B \setminus A) = \mu(B) - \mu(A)$.
  \item \textbf{Countable subadditivity:}
    $\mu\bigl(\bigcup_{n=1}^\infty A_n\bigr) \leq
    \sum_{n=1}^\infty \mu(A_n)$.
  \item \textbf{Continuity from below:}
    $A_n \uparrow A \implies \mu(A) = \lim_{n \to \infty} \mu(A_n)$.
  \item \textbf{Continuity from above:}
    $A_n \downarrow A$ and $\mu(A_1) < \infty \implies
    \mu(A) = \lim_{n \to \infty} \mu(A_n)$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textbf{(a)} Write $B = A \sqcup (B \setminus A)$. By countable
additivity (applied to a sequence with all but the first two terms
equal to $\emptyset$), $\mu(B) = \mu(A) + \mu(B \setminus A) \geq
\mu(A)$.

\textbf{(b)} From $\mu(B) = \mu(A) + \mu(B \setminus A)$ and
$\mu(A) < \infty$, subtract $\mu(A)$ from both sides.

\textbf{(c)} Define $B_1 = A_1$ and
$B_n = A_n \setminus \bigcup_{k=1}^{n-1} A_k$ for $n \geq 2$. The
sets $B_n$ are pairwise disjoint, $\bigcup_n B_n = \bigcup_n A_n$,
and $B_n \subseteq A_n$. By countable additivity and monotonicity:
\begin{equation}
  \mu\!\left(\bigcup_n A_n\right)
    = \mu\!\left(\bigsqcup_n B_n\right)
    = \sum_n \mu(B_n) \leq \sum_n \mu(A_n).
\end{equation}

\textbf{(d)} Suppose $A_n \uparrow A$ (i.e.,
$A_1 \subseteq A_2 \subseteq \cdots$ and
$A = \bigcup_n A_n$). Set $A_0 = \emptyset$ and
$C_n = A_n \setminus A_{n-1}$. The sets $C_n$ are pairwise disjoint
with $A = \bigsqcup_{k=1}^\infty C_k$ and
$A_n = \bigsqcup_{k=1}^n C_k$. By countable additivity:
\begin{equation}
  \mu(A) = \sum_{k=1}^\infty \mu(C_k)
    = \lim_{n \to \infty} \sum_{k=1}^n \mu(C_k)
    = \lim_{n \to \infty} \mu(A_n).
\end{equation}

\textbf{(e)} Define $B_n = A_1 \setminus A_n$. Then
$B_n \uparrow A_1 \setminus A$. By part (d):
\begin{equation}
  \mu(A_1 \setminus A) = \lim_{n \to \infty} \mu(B_n)
    = \lim_{n \to \infty} \bigl(\mu(A_1) - \mu(A_n)\bigr),
\end{equation}
where the second equality uses part (b) (valid since
$\mu(A_n) \leq \mu(A_1) < \infty$). Therefore
$\mu(A_1) - \mu(A) = \mu(A_1) - \lim_n \mu(A_n)$, so
$\mu(A) = \lim_n \mu(A_n)$.
\end{proof}

\begin{warning}[title={Finiteness in Continuity from Above}]
The condition $\mu(A_1) < \infty$ in part (e) cannot be dropped.
Consider counting measure on $\N$ with $A_n = \{n, n+1, n+2, \ldots\}$.
Then $A_n \downarrow \emptyset$, but $\mu(A_n) = \infty$ for every
$n$, while $\mu(\emptyset) = 0$. For probability measures, this issue
never arises since $\Prob(\Omega) = 1 < \infty$.
\end{warning}

\subsection{Null Sets and Complete Measures}
\label{subsec:null_complete}

\begin{definition}[Null Set, Complete Measure]
\label{def:null_complete}
Let $(\Omega, \mathcal{F}, \mu)$ be a measure space. A set
$N \in \mathcal{F}$ is a \emph{$\mu$-null set} if $\mu(N) = 0$. The
measure $\mu$ is \emph{complete} if whenever $N$ is $\mu$-null and
$A \subseteq N$, then $A \in \mathcal{F}$ (and hence $\mu(A) = 0$ by
monotonicity).
\end{definition}

\begin{proposition}[Completion of a Measure]
\label{prop:completion}
Every measure space $(\Omega, \mathcal{F}, \mu)$ has a
\emph{completion}: a complete measure space
$(\Omega, \overline{\mathcal{F}}, \bar{\mu})$ with
$\overline{\mathcal{F}} \supseteq \mathcal{F}$ and
$\bar{\mu}|_{\mathcal{F}} = \mu$, defined by
\begin{equation}
  \overline{\mathcal{F}} = \{A \cup N' : A \in \mathcal{F},\;
    N' \subseteq N \text{ for some $\mu$-null }
    N \in \mathcal{F}\},
  \qquad
  \bar{\mu}(A \cup N') = \mu(A).
  \label{eq:completion}
\end{equation}
\end{proposition}

\begin{proof}
\textit{Well-definedness of $\bar{\mu}$.} If
$A_1 \cup N_1' = A_2 \cup N_2'$ with $N_i' \subseteq N_i$ and
$\mu(N_i) = 0$, then $A_1 \triangle A_2 \subseteq N_1 \cup N_2$
(a null set), so $\mu(A_1) = \mu(A_2)$.

\textit{$\overline{\mathcal{F}}$ is a $\sigma$-algebra.}
For complement: if $E = A \cup N'$ with $N' \subseteq N$,
$\mu(N) = 0$, then
$E^c = (A^c \cap N^c) \cup (A^c \cap (N \setminus N'))$; the first
part is in $\mathcal{F}$ and the second is a subset of $N$.
For countable union: $\bigcup_n (A_n \cup N_n')
= (\bigcup_n A_n) \cup (\bigcup_n N_n')$ with
$\bigcup N_n' \subseteq \bigcup N_n$ and
$\mu(\bigcup N_n) \leq \sum \mu(N_n) = 0$.

\textit{Completeness.} If $\bar{\mu}(E) = 0$ with $E = A \cup N'$
and $F \subseteq E$, then $F \subseteq A \cup N$ with
$\mu(A \cup N) = 0$, so
$F = \emptyset \cup F \in \overline{\mathcal{F}}$.
\end{proof}

\section{Algebras and Pre-Measures}
\label{sec:algebras}

To construct measures on $\sigma$-algebras (which are typically
enormous), we first define them on smaller ``algebras'' and then extend.

\begin{definition}[Algebra of Sets]
\label{def:algebra}
A collection $\mathcal{A} \subseteq \powerset(\Omega)$ is an
\emph{algebra} (or \emph{field of sets}) if:
\begin{enumerate}[nosep,label=(\roman*)]
  \item $\Omega \in \mathcal{A}$.
  \item $A \in \mathcal{A} \implies A^c \in \mathcal{A}$.
  \item $A, B \in \mathcal{A} \implies A \cup B \in \mathcal{A}$.
    \hfill(closure under \emph{finite} union)
\end{enumerate}
\end{definition}

\begin{remark}
An algebra differs from a $\sigma$-algebra only in axiom (iii): finite
versus countable closure. Every $\sigma$-algebra is an algebra, but
not conversely.
\end{remark}

\begin{example}[The Algebra of Finite Unions of Half-Open Intervals]
\label{ex:algebra_intervals}
On $\R$, the collection $\mathcal{A}$ of all finite disjoint unions
of intervals of the form $(a, b]$ (including $\emptyset$ and
$\R = (-\infty, \infty)$ as limiting cases) is an algebra but not a
$\sigma$-algebra. For instance,
$\bigcup_{n=1}^\infty (0, 1 - 1/n] = (0, 1)$ is not a finite union
of half-open intervals, so $(0, 1) \notin \mathcal{A}$.
\end{example}

\begin{definition}[Pre-Measure]
\label{def:premeasure}
A \emph{pre-measure} on an algebra $\mathcal{A}$ is a function
$\mu_0\colon \mathcal{A} \to [0, \infty]$ satisfying:
\begin{enumerate}[nosep,label=(\roman*)]
  \item $\mu_0(\emptyset) = 0$.
  \item \textbf{Countable additivity on $\mathcal{A}$:} If
    $A_1, A_2, \ldots \in \mathcal{A}$ are pairwise disjoint and
    $\bigsqcup_{n=1}^\infty A_n \in \mathcal{A}$, then
    $\mu_0\bigl(\bigsqcup_n A_n\bigr) = \sum_n \mu_0(A_n)$.
\end{enumerate}
\end{definition}

\begin{example}[Length as a Pre-Measure]
\label{ex:length_premeasure}
On the algebra $\mathcal{A}$ of Example~\ref{ex:algebra_intervals},
define $\mu_0\bigl(\bigsqcup_{i=1}^k (a_i, b_i]\bigr) = \sum_{i=1}^k
(b_i - a_i)$. Finite additivity is immediate. Countable additivity
(the non-trivial step) uses the Heine--Borel theorem to show
$b - a \leq \sum (b_n - a_n)$ whenever
$(a,b] = \bigsqcup_n (a_n, b_n]$; the reverse follows from
monotonicity. So $\mu_0$ is a pre-measure.
\end{example}

\section{Outer Measures and Carath\'{e}odory's Theorem}
\label{sec:caratheodory}

\begin{definition}[Outer Measure]
\label{def:outer_measure}
An \emph{outer measure} on $\Omega$ is a function
$\mu^*\colon \powerset(\Omega) \to [0, \infty]$ (defined on
\emph{all} subsets) satisfying: (i)~$\mu^*(\emptyset) = 0$;
(ii)~\textbf{monotonicity:}
$A \subseteq B \implies \mu^*(A) \leq \mu^*(B)$;
(iii)~\textbf{countable subadditivity:}
$\mu^*(\bigcup_{n=1}^\infty A_n) \leq \sum_{n=1}^\infty \mu^*(A_n)$.
\end{definition}

\begin{proposition}[Pre-Measure Induces Outer Measure]
\label{prop:induced_outer}
Let $\mu_0$ be a pre-measure on an algebra $\mathcal{A}$. For any
$E \subseteq \Omega$, define
\begin{equation}
  \mu^*(E) = \inf\left\{\sum_{n=1}^\infty \mu_0(A_n) :
    A_n \in \mathcal{A},\;
    E \subseteq \bigcup_{n=1}^\infty A_n\right\}.
  \label{eq:outer_measure}
\end{equation}
Then $\mu^*$ is an outer measure, and $\mu^*(A) = \mu_0(A)$ for all
$A \in \mathcal{A}$.
\end{proposition}

\begin{proof}
\textit{Property (i):} Cover $\emptyset$ by $A_n = \emptyset$ for all
$n$. Then $\sum \mu_0(A_n) = 0$, so $\mu^*(\emptyset) = 0$.

\textit{Property (ii):} If $A \subseteq B$, every cover of $B$ is a
cover of $A$, so the infimum over covers of $A$ is at most the infimum
over covers of $B$.

\textit{Property (iii):} Let $E = \bigcup_n E_n$. For each
$\varepsilon > 0$ and each $n$, choose a cover
$\{A_{n,k}\}_{k \geq 1} \subseteq \mathcal{A}$ with
$E_n \subseteq \bigcup_k A_{n,k}$ and
$\sum_k \mu_0(A_{n,k}) \leq \mu^*(E_n) + \varepsilon / 2^n$. Then
$\{A_{n,k}\}_{n,k}$ covers $E$, so
\begin{equation}
  \mu^*(E) \leq \sum_n \sum_k \mu_0(A_{n,k})
    \leq \sum_n \bigl(\mu^*(E_n) + \varepsilon / 2^n\bigr)
    = \sum_n \mu^*(E_n) + \varepsilon.
\end{equation}
Since $\varepsilon > 0$ was arbitrary,
$\mu^*(E) \leq \sum_n \mu^*(E_n)$.

\textit{$\mu^*$ extends $\mu_0$:} For $A \in \mathcal{A}$, the
single-set cover $A_1 = A$, $A_n = \emptyset$ for $n \geq 2$ gives
$\mu^*(A) \leq \mu_0(A)$. For the reverse inequality, suppose
$A \subseteq \bigcup_n A_n$ with $A_n \in \mathcal{A}$. Define
$B_1 = A \cap A_1$ and $B_n = A \cap A_n \cap
\bigl(\bigcup_{k=1}^{n-1} A_k\bigr)^c$ for $n \geq 2$. Since
$\mathcal{A}$ is an algebra, each $B_n \in \mathcal{A}$. The sets
$B_n$ are pairwise disjoint with $A = \bigsqcup_n B_n$ and
$B_n \subseteq A_n$. By countable additivity of $\mu_0$ on
$\mathcal{A}$ and monotonicity of $\mu_0$ (which follows from finite
additivity):
\begin{equation}
  \mu_0(A) = \sum_n \mu_0(B_n) \leq \sum_n \mu_0(A_n).
\end{equation}
Taking the infimum over all covers: $\mu_0(A) \leq \mu^*(A)$.
\end{proof}

\begin{definition}[Carath\'{e}odory Measurability]
\label{def:cara_measurable}
Given an outer measure $\mu^*$ on $\Omega$, a set
$A \subseteq \Omega$ is \emph{$\mu^*$-measurable} (or
\emph{Carath\'{e}odory measurable}) if for every
$E \subseteq \Omega$:
\begin{equation}
  \mu^*(E) = \mu^*(E \cap A) + \mu^*(E \cap A^c).
  \label{eq:cara_condition}
\end{equation}
\end{definition}

\begin{remark}
By subadditivity, $\mu^*(E) \leq \mu^*(E \cap A) + \mu^*(E \cap A^c)$
always holds. The content of~\eqref{eq:cara_condition} is the reverse
inequality: $A$ ``splits'' every test set $E$ additively.
\end{remark}

\begin{keyresult}[title={Carath\'{e}odory Extension Theorem}]
\begin{theorem}[Carath\'{e}odory Extension]
\label{thm:caratheodory}
Let $\mu^*$ be an outer measure on $\Omega$. Then:
\begin{enumerate}[nosep,label=(\alph*)]
  \item The collection $\mathcal{M} = \{A \subseteq \Omega :
    A \text{ is $\mu^*$-measurable}\}$ is a $\sigma$-algebra.
  \item The restriction $\mu = \mu^*|_{\mathcal{M}}$ is a complete
    measure on $(\Omega, \mathcal{M})$.
\end{enumerate}
Consequently, if $\mu_0$ is a pre-measure on an algebra $\mathcal{A}$
and $\mu^*$ is the induced outer measure~\eqref{eq:outer_measure},
then $\mathcal{A} \subseteq \mathcal{M}$ and
$\mu|_{\mathcal{A}} = \mu_0$. If $\mu_0$ is $\sigma$-finite, the
extension to $\sigma(\mathcal{A})$ is unique.
\end{theorem}
\end{keyresult}

\begin{proof}
\textbf{Part (a): $\mathcal{M}$ is a $\sigma$-algebra.}

\textit{Step 1: $\Omega \in \mathcal{M}$.} For any $E$:
$\mu^*(E \cap \Omega) + \mu^*(E \cap \emptyset)
= \mu^*(E) + 0 = \mu^*(E)$.

\textit{Step 2: Closure under complement.} The
condition~\eqref{eq:cara_condition} is symmetric in $A$ and $A^c$, so
$A \in \mathcal{M}$ if and only if $A^c \in \mathcal{M}$.

\textit{Step 3: Closure under finite union.} Let
$A, B \in \mathcal{M}$. For any test set $E$, applying the
measurability of $A$ and then of $B$ (to the test set $E \cap A^c$):
\begin{align}
  \mu^*(E)
    &= \mu^*(E \cap A) + \mu^*(E \cap A^c) \notag \\
    &= \mu^*(E \cap A)
      + \mu^*(E \cap A^c \cap B)
      + \mu^*(E \cap A^c \cap B^c). \label{eq:cara_three_terms}
\end{align}
We need to show $\mu^*(E) = \mu^*(E \cap (A \cup B))
+ \mu^*(E \cap (A \cup B)^c)$. Note that
$(A \cup B)^c = A^c \cap B^c$, so the third term
in~\eqref{eq:cara_three_terms} is already
$\mu^*(E \cap (A \cup B)^c)$. For the remaining two terms:
$E \cap (A \cup B) = (E \cap A) \cup (E \cap A^c \cap B)$, so by
subadditivity:
\begin{equation}
  \mu^*(E \cap (A \cup B)) \leq \mu^*(E \cap A)
    + \mu^*(E \cap A^c \cap B).
\end{equation}
Combining with~\eqref{eq:cara_three_terms}:
\begin{equation}
  \mu^*(E) \geq \mu^*(E \cap (A \cup B))
    + \mu^*(E \cap (A \cup B)^c).
\end{equation}
The reverse inequality holds by subadditivity of $\mu^*$. Therefore
$A \cup B \in \mathcal{M}$.

\textit{Step 4: Closure under countable disjoint union.} Let
$A_1, A_2, \ldots \in \mathcal{M}$ be pairwise disjoint. Set
$S_n = \bigsqcup_{k=1}^n A_k$ and
$S = \bigsqcup_{k=1}^\infty A_k$. By Step~3 and closure under
complement (hence under intersection), each $S_n \in \mathcal{M}$.

We prove by induction that for any $E$:
\begin{equation}
  \mu^*(E \cap S_n) = \sum_{k=1}^n \mu^*(E \cap A_k).
  \label{eq:finite_additivity_outer}
\end{equation}
For $n = 1$: $\mu^*(E \cap S_1) = \mu^*(E \cap A_1)$. For the
inductive step, apply the measurability of $A_n$ to the test set
$E \cap S_n$:
\begin{equation}
  \mu^*(E \cap S_n)
    = \mu^*(E \cap S_n \cap A_n) + \mu^*(E \cap S_n \cap A_n^c)
    = \mu^*(E \cap A_n) + \mu^*(E \cap S_{n-1}).
\end{equation}

Since $S_n \in \mathcal{M}$:
$\mu^*(E) = \mu^*(E \cap S_n) + \mu^*(E \cap S_n^c)$. Since
$S^c \subseteq S_n^c$, monotonicity gives
$\mu^*(E \cap S_n^c) \geq \mu^*(E \cap S^c)$.
Using~\eqref{eq:finite_additivity_outer}:
\begin{equation}
  \mu^*(E) \geq \sum_{k=1}^n \mu^*(E \cap A_k)
    + \mu^*(E \cap S^c).
\end{equation}
Letting $n \to \infty$:
\begin{equation}
  \mu^*(E) \geq \sum_{k=1}^\infty \mu^*(E \cap A_k)
    + \mu^*(E \cap S^c)
    \geq \mu^*(E \cap S) + \mu^*(E \cap S^c),
\end{equation}
where the last inequality uses subadditivity:
$\mu^*(E \cap S) \leq \sum_k \mu^*(E \cap A_k)$. The reverse
inequality holds by subadditivity of $\mu^*$. So
$S \in \mathcal{M}$.

\textit{Step 5: General countable unions.} For arbitrary (not
necessarily disjoint) $A_n \in \mathcal{M}$, define
$B_n = A_n \setminus \bigcup_{k=1}^{n-1} A_k \in \mathcal{M}$ (by
closure under complement and finite intersection). Then
$\bigcup_n A_n = \bigsqcup_n B_n \in \mathcal{M}$ by Step~4.

\textbf{Part (b): $\mu$ is a complete measure.}

\textit{Countable additivity.} For pairwise disjoint
$A_n \in \mathcal{M}$, set $S = \bigsqcup_n A_n$ and $E = S$
in~\eqref{eq:finite_additivity_outer}. Letting $n \to \infty$:
$\mu(S) = \mu^*(S) \geq \sum_k \mu^*(A_k) = \sum_k \mu(A_k)$.
Subadditivity gives the reverse, so
$\mu(S) = \sum_k \mu(A_k)$.

\textit{Completeness.} Suppose $\mu^*(N) = 0$ and
$A \subseteq N$. For any test set $E$:
$\mu^*(E \cap A) \leq \mu^*(A) \leq \mu^*(N) = 0$, so
$\mu^*(E \cap A) + \mu^*(E \cap A^c) = \mu^*(E \cap A^c)
\leq \mu^*(E)$ by monotonicity (since $E \cap A^c \subseteq E$).
The reverse $\mu^*(E) \leq \mu^*(E \cap A) + \mu^*(E \cap A^c)$
holds by subadditivity. So $A \in \mathcal{M}$.

\textbf{$\mathcal{A} \subseteq \mathcal{M}$ and uniqueness.}
For $A \in \mathcal{A}$ and any $E \subseteq \Omega$: let
$\{C_n\} \subseteq \mathcal{A}$ cover $E$ with
$\sum \mu_0(C_n) \leq \mu^*(E) + \varepsilon$. Then
$C_n \cap A, C_n \cap A^c \in \mathcal{A}$ and they cover
$E \cap A$ and $E \cap A^c$ respectively. By finite additivity of
$\mu_0$: $\mu_0(C_n) = \mu_0(C_n \cap A) + \mu_0(C_n \cap A^c)$.
So $\mu^*(E \cap A) + \mu^*(E \cap A^c) \leq
\sum \mu_0(C_n \cap A) + \sum \mu_0(C_n \cap A^c) =
\sum \mu_0(C_n) \leq \mu^*(E) + \varepsilon$. Since $\varepsilon$
was arbitrary, $A \in \mathcal{M}$.

Uniqueness under $\sigma$-finiteness follows from the
$\pi$-$\lambda$ theorem (Lesson~1, Corollary): $\mathcal{A}$ is a
$\pi$-system generating $\sigma(\mathcal{A})$, and two measures
agreeing on a $\pi$-system with a $\sigma$-finite exhaustion agree
on the generated $\sigma$-algebra.
\end{proof}

\section{Lebesgue Measure}
\label{sec:lebesgue}

\begin{theorem}[Existence and Uniqueness of Lebesgue Measure]
\label{thm:lebesgue}
There exists a unique measure $\lambda$ on $(\R, \Borel(\R))$ such
that $\lambda((a, b]) = b - a$ for all $a < b$. This measure is
$\sigma$-finite and translation-invariant:
$\lambda(A + x) = \lambda(A)$ for all $A \in \Borel(\R)$ and
$x \in \R$.
\end{theorem}

\begin{proof}
Let $\mathcal{A}$ be the algebra of finite disjoint unions of
half-open intervals $(a_i, b_i]$
(Example~\ref{ex:algebra_intervals}), and let $\mu_0$ be the
pre-measure of Example~\ref{ex:length_premeasure}. The key
verification is that $\mu_0$ is countably additive on $\mathcal{A}$.

Suppose $(a, b] = \bigsqcup_{n=1}^\infty (a_n, b_n]$ with the union
in $\mathcal{A}$. Finite additivity gives
$b - a \geq \sum_{n=1}^N (b_n - a_n)$ for each $N$ (since
$\bigsqcup_{n=1}^N (a_n, b_n] \subseteq (a, b]$), hence
$b - a \geq \sum_{n=1}^\infty (b_n - a_n)$.

For the reverse, fix $\varepsilon > 0$. Consider the closed interval
$[a + \varepsilon, b] \subseteq (a, b]$ and the open intervals
$(a_n, b_n + \varepsilon/2^n) \supseteq (a_n, b_n]$. These open
intervals cover the compact set $[a + \varepsilon, b]$, so by the
Heine--Borel theorem, finitely many suffice:
$[a + \varepsilon, b] \subseteq \bigcup_{n=1}^N
(a_n, b_n + \varepsilon/2^n)$. Therefore
\begin{equation}
  b - a - \varepsilon \leq b - (a + \varepsilon)
    \leq \sum_{n=1}^N (b_n + \varepsilon/2^n - a_n)
    \leq \sum_{n=1}^\infty (b_n - a_n) + \varepsilon.
\end{equation}
Since $\varepsilon$ was arbitrary,
$b - a \leq \sum (b_n - a_n)$.

By Carath\'{e}odory's theorem (Theorem~\ref{thm:caratheodory}),
$\mu_0$ extends to a measure $\lambda$ on $\sigma(\mathcal{A})$. Since
$\sigma(\mathcal{A}) = \Borel(\R)$ (the half-open intervals generate
the Borel $\sigma$-algebra), $\lambda$ is defined on $\Borel(\R)$.

The extension is unique because $\mu_0$ is $\sigma$-finite:
$\R = \bigcup_{n=1}^\infty (-n, n]$ with
$\mu_0((-n, n]) = 2n < \infty$.

Translation invariance: $\mu_0((a + x, b + x]) = b - a =
\mu_0((a, b])$, so $\lambda$ and the shifted measure
$A \mapsto \lambda(A + x)$ agree on $\mathcal{A}$. By uniqueness,
they agree on $\Borel(\R)$.
\end{proof}

\section{Non-Measurable Sets}
\label{sec:vitali}

\begin{theorem}[Vitali Non-Measurable Set]
\label{thm:vitali}
There exists a set $V \subseteq [0, 1)$ that is not Lebesgue
measurable.
\end{theorem}

\begin{proof}
Define an equivalence relation on $[0, 1)$ by $x \sim y$ if and only
if $x - y \in \Q$. This partitions $[0, 1)$ into equivalence classes
(each a coset of $\Q$ intersected with $[0, 1)$). By the Axiom of
Choice, select exactly one representative from each class; call this
set $V$.

For each $q \in \Q \cap [0, 1)$, define the translate
$V_q = \{(v + q) \bmod 1 : v \in V\}$, where $x \bmod 1$ denotes
the fractional part of $x$.

\textit{Claim 1: The sets $\{V_q\}$ are pairwise disjoint.} If
$x \in V_{q_1} \cap V_{q_2}$, then
$x = (v_1 + q_1) \bmod 1 = (v_2 + q_2) \bmod 1$, so
$v_1 - v_2 \in \Q$, hence $v_1 \sim v_2$, giving $v_1 = v_2$ and
$q_1 = q_2$.

\textit{Claim 2:
$[0, 1) = \bigsqcup_{q \in \Q \cap [0,1)} V_q$.} Each
$x \in [0, 1)$ satisfies $x \sim v$ for some $v \in V$, and
$q = (x - v) \bmod 1 \in \Q \cap [0,1)$ gives $x \in V_q$.

\textit{Claim 3: All $V_q$ have the same measure.} The map
$x \mapsto (x + q) \bmod 1$ preserves $\lambda$ on $[0,1)$.

Now suppose $V$ were measurable. Then each $V_q$ is measurable with
$\lambda(V_q) = \lambda(V) =: c$. By countable additivity:
\begin{equation}
  1 = \lambda([0, 1))
    = \sum_{q \in \Q \cap [0,1)} \lambda(V_q)
    = \sum_{q \in \Q \cap [0,1)} c.
\end{equation}
But the right side is $0$ if $c = 0$ and $\infty$ if $c > 0$---neither
equals $1$. This contradiction shows $V$ is not Lebesgue measurable.
\end{proof}

\begin{remark}
Vitali's theorem shows that no translation-invariant measure can be
defined on all of $\powerset(\R)$---the fundamental reason for
restricting to $\sigma$-algebras. In RL with continuous state spaces,
all practically arising sets are Borel, but Vitali's result justifies
the measure-theoretic formalism.
\end{remark}

\section{Exercises}
\label{sec:exercises}

\begin{exercise}[Continuity of Measures]
\label{ex:continuity}
Let $(\Omega, \mathcal{F}, \mu)$ be a measure space and let
$(A_n)_{n \geq 1}$ be a sequence in $\mathcal{F}$.
\begin{enumerate}[label=(\alph*)]
  \item Prove that
    $\mu\bigl(\liminf_{n} A_n\bigr) \leq
    \liminf_{n} \mu(A_n)$,
    where $\liminf_n A_n = \bigcup_{n=1}^\infty
    \bigcap_{k=n}^\infty A_k$.
  \item Show by example that the inequality can be strict.
  \item If $\mu(\Omega) < \infty$, prove
    $\mu\bigl(\limsup_{n} A_n\bigr) \geq
    \limsup_{n} \mu(A_n)$,
    where $\limsup_n A_n = \bigcap_{n=1}^\infty
    \bigcup_{k=n}^\infty A_k$.
\end{enumerate}
\textit{Hint:} Define $B_n = \bigcap_{k=n}^\infty A_k$ and apply
continuity from below.
\end{exercise}

\begin{exercise}[Verifying Outer Measure Properties]
\label{ex:outer_measure}
Let $\Omega = \{1, 2, 3\}$ and define
$\mu^*\colon \powerset(\Omega) \to [0, \infty]$ by
$\mu^*(\emptyset) = 0$, $\mu^*(\{i\}) = 1$ for each $i$,
$\mu^*(\{1,2\}) = 1$, $\mu^*(\{1,3\}) = 2$, $\mu^*(\{2,3\}) = 2$,
and $\mu^*(\{1,2,3\}) = 2$.
\begin{enumerate}[label=(\alph*)]
  \item Verify that $\mu^*$ is an outer measure.
  \item Determine which subsets of $\Omega$ are
    $\mu^*$-measurable (i.e., satisfy the Carath\'{e}odory
    condition for \emph{every} test set $E$).
  \item What is the resulting $\sigma$-algebra $\mathcal{M}$, and what
    is $\mu = \mu^*|_{\mathcal{M}}$?
\end{enumerate}
\end{exercise}

\begin{exercise}[Borel--Cantelli Application]
\label{ex:borel_cantelli_prep}
Let $(\Omega, \mathcal{F}, \mu)$ be a measure space with
$\mu(\Omega) < \infty$. Using only the properties proved in
Theorem~\ref{thm:measure_props}, prove:
\begin{enumerate}[label=(\alph*)]
  \item If $\sum_{n=1}^\infty \mu(A_n) < \infty$, then
    $\mu(\limsup_n A_n) = 0$.
  \item Interpret part (a) when $\mu = \Prob$ is a probability
    measure: ``if the total probability is summable, then the event
    that infinitely many $A_n$ occur has probability zero.''
\end{enumerate}
This is the first Borel--Cantelli lemma, which is fundamental in RL
convergence proofs.

\textit{Hint:} Use subadditivity on
$\bigcup_{k=n}^\infty A_k$ and then continuity from above.
\end{exercise}

\begin{exercise}[Completion of a Measure]
\label{ex:counting_completion}
Let $\Omega = \{1, 2, 3\}$ with
$\mathcal{F} = \bigl\{\emptyset, \{1\}, \{2,3\}, \Omega\bigr\}$ and
$\mu(\emptyset) = 0$, $\mu(\{1\}) = 1$, $\mu(\{2,3\}) = 2$,
$\mu(\Omega) = 3$. Is $\mu$ complete? Compute the completion
$(\Omega, \overline{\mathcal{F}}, \bar{\mu})$ explicitly.
\end{exercise}
